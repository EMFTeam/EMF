# -*- ck2.decisions -*-

decisions = {

	make_rome_capital = {
		is_high_prio = yes
		ai = no

		potential = {
			has_dlc = "Legacy of Rome"
			has_landed_title = e_roman_empire
			capital_scope = {
				NOT = { province_id = 333 }
			}
			emf_hiprio_decision_basic_potential = yes
		}
		allow = {
			333 = {
				owner_under_ROOT = yes
				owner = {
					OR = {
						ai = yes
						character = ROOT
					}
				}
				any_province_holding = {
					is_preferred_holding_type = ROOT
					owner_under_ROOT = yes
					owner = {
						OR = {
							ai = yes
							character = ROOT
						}
					}
				}
			}
		}
		effect = {
			custom_tooltip = {
				text = rome_capital
				hidden_effect = {
					c_roma = {
						# Do this first so that the cultural name of c_roma is set properly
						usurp_title_plus_barony_if_unlanded_and_vassals = ROOT
						if = {
							limit = {
								location = {
									capital_holding = {
										NOT = { is_preferred_holding_type = ROOT }
									}
								}
							}
							# Prefer to use an existing holding named "Rome" if possible instead of renaming a different holding
							if = {
								limit = {
									location = {
										capital_holding = {
											NOT = { title = b_roma }
										}
									}
									b_roma = {
										has_holder = yes
										is_preferred_holding_type = ROOT
										owner_under_ROOT = yes
										owner = {
											OR = {
												ai = yes
												character = ROOT
											}
										}
									}
								}
								b_roma = { make_capital_holding = yes }
							}
							else_if = {
								limit = {
									location = {
										capital_holding = {
											NOT = { title = b_cittadiroma }
										}
									}
									b_cittadiroma = {
										has_holder = yes
										is_preferred_holding_type = ROOT
										owner_under_ROOT = yes
										owner = {
											OR = {
												ai = yes
												character = ROOT
											}
										}
									}
								}
								b_cittadiroma = { make_capital_holding = yes }
							}
							else = {
								location = {
									random_province_holding = {
										limit = {
											is_preferred_holding_type = ROOT
											owner = {
												under_ROOT = yes
												OR = {
													ai = yes
													character = ROOT
												}
											}
										}
										copy_name = PREVPREV # Not the best solution because the new name won't change should the holding's owner's culture change, but it'll do.
										make_capital_holding = yes
									}
								}
							}
						}
						else_if = {
							limit = {
								location = {
									capital_holding = {
										NOR = {
											title = b_roma
											title = b_cittadiroma
										}
									}
								}
							}
							location = {
								capital_holding = {
									copy_name = PREVPREV # Not the best solution because the new name won't change should the holding's owner's culture change, but it'll do.
								}
							}
						}
						ROOT = { capital = PREV }
					}
				}
			}
		}
		ai_will_do = {
			factor = 0
		}
	}
}
