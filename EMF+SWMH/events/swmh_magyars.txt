# -*- ck2.events -*-

# Christianization of Magyars (Moger -> Magyar) and Khavars (Qavar -> Székely)

namespace = SWMH_magyars

# Christianization begins for Magyars
character_event = {
	id = SWMH_magyars.1
	desc = "EVTDESCSWMH_magyars.1"
	picture = GFX_evt_stone_church
	
	only_playable = yes
	religion_group = christian
	culture = old_hungarian
	
	trigger = {
		is_nomadic = no
		OR = {
			is_tribal = no
			year = 920
			has_global_flag = emf_magyar_migration_completed
		}
		OR = {
			any_demesne_province = {
				NOT = { terrain = steppe }
				culture = old_hungarian
				religion_group = christian
				any_province_holding = { NOT = { holding_type = nomad } }
				OR = {
					NOT = { any_province_holding = { holding_type = tribal } }
					year = 920
					has_global_flag = emf_magyar_migration_completed
				}
			}
			NOT = { # Makes sure Christian Mogers ruling over purely settled, non-steppe, Christian lands also convert
				any_demesne_province = {
					OR = {
						terrain = steppe
						NOR = {
							religion_group = christian
							any_province_holding = { NOT = { holding_type = nomad } }
						}
						AND = {
							any_province_holding = { holding_type = tribal }
							NOR = {
								year = 920
								has_global_flag = emf_magyar_migration_completed
							}
						}
					}
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 120
		modifier = {
			factor = 2
			is_tribal = yes
		}
	}

	option = {
		name = "EVTOPTASWMH_magyars.1"
		culture = hungarian
		random_demesne_province = {
			limit = {
				NOT = { terrain = steppe }
				culture = old_hungarian
				religion_group = christian
				any_province_holding = { NOT = { holding_type = nomad } }
				OR = {
					NOT = { any_province_holding = { holding_type = tribal } }
					year = 920
					has_global_flag = emf_magyar_migration_completed
				}
			}
			preferred_limit = {
				ROOT = { capital_scope = { province = PREVPREV } }
				NOT = { any_province_holding = { holding_type = tribal } }
			}
			preferred_limit = {
				NOT = { any_province_holding = { holding_type = tribal } }
			}
			preferred_limit = {
				ROOT = { capital_scope = { province = PREVPREV } }
				any_province_holding = { NOT = { holding_type = tribal } }
			}
			preferred_limit = {
				any_province_holding = { NOT = { holding_type = tribal } }
			}
			preferred_limit = {
				ROOT = { capital_scope = { province = PREVPREV } }
			}
			culture = hungarian
		}
	}
}

# Appearance of Székely culture
character_event = {
	id = SWMH_magyars.6
	desc = "EVTDESCSWMH_magyars.6"
	picture = GFX_evt_stone_church
	
	only_playable = yes
	
	trigger = {
		is_nomadic = no
		OR = {
			is_tribal = no
			year = 920
			has_global_flag = emf_magyar_migration_completed
		}
		OR = {
			AND = {
				culture = hungarian
				NOR = {
					any_demesne_province = {
						culture = szekely
					}
					any_vassal = {
						tier = BARON
						culture = szekely
					}
				}
				OR = {
					any_demesne_province = {
						NOT = { terrain = steppe }
						culture = khavar
						religion_group = christian
						any_province_holding = { NOT = { holding_type = nomad } }
						OR = {
							NOT = { any_province_holding = { holding_type = tribal } }
							year = 920
							has_global_flag = emf_magyar_migration_completed
						}
					}
					any_vassal = {
						tier = BARON
						culture = khavar
						religion_group = christian
						capital_scope = {
							NOT = { terrain = steppe }
						}
						OR = {
							is_tribal = no
							year = 920
							has_global_flag = emf_magyar_migration_completed
						}
					}
				}
			}
			AND = {
				independent = no
				culture = khavar
				religion_group = christian
				top_liege = {
					culture = hungarian
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 120
		modifier = {
			factor = 2
			is_tribal = yes
		}
	}
	
	option = {
		name = "EVTOPTASWMH_magyars.6"
		trigger = {
			NOT = { culture = hungarian }
		}
		culture = szekely
		random_demesne_province = {
			limit = {
				NOT = { terrain = steppe }
				culture = khavar
				religion_group = christian
				any_province_holding = { NOT = { holding_type = nomad } }
				OR = {
					NOT = { any_province_holding = { holding_type = tribal } }
					year = 920
					has_global_flag = emf_magyar_migration_completed
				}
			}
			preferred_limit = {
				ROOT = { capital_scope = { province = PREVPREV } }
				NOT = { any_province_holding = { holding_type = tribal } }
			}
			preferred_limit = {
				NOT = { any_province_holding = { holding_type = tribal } }
			}
			preferred_limit = {
				ROOT = { capital_scope = { province = PREVPREV } }
				any_province_holding = { NOT = { holding_type = tribal } }
			}
			preferred_limit = {
				any_province_holding = { NOT = { holding_type = tribal } }
			}
			preferred_limit = {
				ROOT = { capital_scope = { province = PREVPREV } }
			}
			culture = szekely
		}
	}

	option = {
		name = I_SEE
		trigger = {
			culture = hungarian
		}
		if = {
			limit = {
				any_demesne_province = {
					NOT = { terrain = steppe }
					culture = khavar
					religion_group = christian
					any_province_holding = { NOT = { holding_type = nomad } }
					OR = {
						NOT = { any_province_holding = { holding_type = tribal } }
						year = 920
						has_global_flag = emf_magyar_migration_completed
					}
				}
			}
			random_demesne_province = {
				limit = {
					NOT = { terrain = steppe }
					culture = khavar
					religion_group = christian
					any_province_holding = { NOT = { holding_type = nomad } }
					OR = {
						NOT = { any_province_holding = { holding_type = tribal } }
						year = 920
						has_global_flag = emf_magyar_migration_completed
					}
				}
				preferred_limit = {
					NOT = { any_province_holding = { holding_type = tribal } }
				}
				preferred_limit = {
					any_province_holding = { NOT = { holding_type = tribal } }
				}
				culture = szekely
			}
		}
		else = {
			random_vassal = {
				limit = {
					tier = BARON
					culture = khavar
					religion_group = christian
					capital_scope = {
						NOT = { terrain = steppe }
					}
					OR = {
						is_tribal = no
						year = 920
						has_global_flag = emf_magyar_migration_completed
					}
				}
				preferred_limit = {
					dynasty = ROOT
					is_tribal = no
				}
				preferred_limit = {
					is_tribal = no
				}
				preferred_limit = {
					dynasty = ROOT
				}
				culture = szekely
			}
		}
	}
}
