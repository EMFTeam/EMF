# Appearance of Szekely culture
province_event = {
	id = emf_magyar.1015
	desc = EVTDESC_emf_magyar_3015
	picture = GFX_evt_magyars2
	
	trigger = {
		county = {
			is_occupied = no
			OR = {
				title = c_bihar
				title = c_szekelyfold
				title = c_pecs
			}
		}
		owner = {
			OR = {
				culture = hungarian
				culture = szekely
			}
			top_liege = {
				OR = {
					culture = hungarian
					culture = szekely
				}
				num_of_count_titles_in_realm = 10
				any_realm_province = {
					culture = hungarian
					count = 4
				}
			}
		}
		has_global_flag = emf_magyar_migration_completed
		NOT = { has_province_flag = szekely_conversion }
	}
	
	mean_time_to_happen = {
		years = 10
		modifier = {
			factor = 0.5
			owner = { culture = szekely }
		}
		modifier = {
			factor = 0.5
			any_neighbor_province = { culture = szekely }
		}
	}

	option = {
		name = I_SEE
		set_province_flag = szekely_conversion
		culture = szekely
		hidden_tooltip = {
			owner = {
				create_random_soldier = {
					random_traits = yes
					dynasty = none
					religion = ROOT
					culture = szekely
					female = no
					age = 25
				}
				create_random_soldier = {
					random_traits = yes
					dynasty = none
					religion = ROOT
					culture = szekely
					female = no
					age = 28
				}
				create_character = {
					random_traits = yes
					dynasty = none
					religion = ROOT
					culture = szekely
					female = no
					age = 30
				}
				create_character = {
					random_traits = yes
					dynasty = none
					religion = ROOT
					culture = szekely
					female = yes
					age = 16
				}
				liege = {
					create_random_soldier = {
						random_traits = yes
						dynasty = none
						religion = ROOT
						culture = szekely
						female = no
						age = 25
					}
					create_character = {
						random_traits = yes
						dynasty = none
						religion = ROOT
						culture = szekely
						female = no
						age = 22
					}
					create_character = {
						random_traits = yes
						dynasty = none
						religion = ROOT
						culture = szekely
						female = yes
						age = 16
					}
				}
			}
		}
	}
}

# Szekely conversion for Khazar courtiers
character_event = {
	id = emf_magyar.1016
	
	culture = hungarian
	religion = tengri_pagan
	only_playable = yes
	has_global_flag = emf_magyar_migration_completed

	hide_window = yes
	
	trigger = {
		capital_scope = {
			kingdom = { title = k_hungary }
		}
		any_courtier = {
			ai = yes
			is_ruler = no
			culture = khazar
		}
	}
	
	mean_time_to_happen = {
		years = 10
	}
	
	immediate = {
		any_courtier = {
			limit = {
				ai = yes
				is_ruler = no
				culture = khazar
			}
			culture = szekely
		}
	}
}
