# emf_nomad.1401 [401 - Hlynov]
province_event = {
	id = emf_nomad.1401

	is_triggered_only = yes
	hide_window = yes

	trigger = {
		owner = { is_nomadic = yes }
		NOT = { any_province_holding = { NOT = { holding_type = nomad } } }
		has_empty_holding = yes
	}

	immediate = {
		if = {
			limit = { NOT = { year = 1250 } }
			if = {
				limit = { culture = mari religion = finnish_pagan }
				break = yes # Not necessary to build a settlement
			}
			build_holding = { type = temple }
			culture  = mari
			religion = finnish_pagan
			emf_nomad_antinomad_temple_effect = yes
			log = "emf_antinomad(401, '[Root.GetName]') => temple / mari / finnish_pagan"
			break = yes
		}
		if = {
			limit = { culture = russian religion = orthodox }
			break = yes # Not necessary to build a settlement
		}
		build_holding = { type = temple }
		culture  = russian
		religion = orthodox
		emf_nomad_antinomad_temple_effect = yes
		log = "emf_antinomad(401, '[Root.GetName]') => temple / russian / orthodox"
	}
}

	id = emf_nomad.1607

	is_triggered_only = yes
	hide_window = yes

	trigger = {
		owner = { is_nomadic = yes }
		NOT = { any_province_holding = { NOT = { holding_type = nomad } } }
		has_empty_holding = yes
	}

	immediate = {
		if = {
			limit = { NOT = { year = 1060 } }
			if = {
				limit = { culture = khazar religion = tengri_pagan }
				break = yes # Not necessary to build a settlement
			}
			build_holding = { type = tribal }
			culture  = khazar
			religion = tengri_pagan
			emf_nomad_antinomad_tribal_effect = yes
			log = "emf_antinomad(607, '[Root.GetName]') => tribal / khazar / tengri_pagan"
			break = yes
		}
		if = {
			limit = { NOT = { year = 1227 } }
			if = {
				limit = { culture = oghuz religion = tengri_pagan }
				break = yes # Not necessary to build a settlement
			}
			build_holding = { type = tribal }
			culture  = oghuz
			religion = tengri_pagan
			emf_nomad_antinomad_tribal_effect = yes
			log = "emf_antinomad(607, '[Root.GetName]') => tribal / oghuz / tengri_pagan"
			break = yes
		}
		if = {
			limit = { NOT = { year = 1238 } }
			if = {
				limit = { culture = oghuz religion = sunni }
				break = yes # Not necessary to build a settlement
			}
			build_holding = { type = tribal }
			culture  = oghuz
			religion = sunni
			emf_nomad_antinomad_tribal_effect = yes
			log = "emf_antinomad(607, '[Root.GetName]') => tribal / oghuz / sunni"
			break = yes
		}
		if = {
			limit = { culture = mongol religion = sunni }
			break = yes # Not necessary to build a settlement
		}
		build_holding = { type = tribal }
