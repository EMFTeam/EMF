# emf_nomad.1401 [401 - Hlynov]
province_event = {
	id = emf_nomad.1401
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		owner = { is_nomadic = yes }
		NOT = { any_province_holding = { NOT = { holding_type = nomad } } }
		has_empty_holding = yes
	}

	immediate = {
		if = {
			limit = { NOT = { year = 1250 } }
			if = {
				limit = { culture = mari religion = finnish_pagan }
				break = yes # Not necessary to build a settlement
			}
			build_holding = { type = temple }
			culture  = mari
			religion = finnish_pagan
			emf_nomad_antinomad_temple_effect = yes
			log = "emf_antinomad(401, '[Root.GetName]') => temple / mari / finnish_pagan"
			break = yes
		}
		if = {
			limit = { culture = russian religion = orthodox }
			break = yes # Not necessary to build a settlement
		}
		build_holding = { type = temple }
		culture  = russian
		religion = orthodox
		emf_nomad_antinomad_temple_effect = yes
		log = "emf_antinomad(401, '[Root.GetName]') => temple / russian / orthodox"
	}
}

	id = emf_nomad.2342
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		owner = { is_nomadic = yes }
		NOT = { any_province_holding = { NOT = { holding_type = nomad } } }
		has_empty_holding = yes
	}

	immediate = {
		if = {
			limit = { NOT = { year = 1066 } }
			if = {
				limit = { culture = zanata religion = miaphysite }
				break = yes # Not necessary to build a settlement
			}
			build_holding = { type = tribal }
			culture  = zanata
			religion = miaphysite
			emf_nomad_antinomad_tribal_effect = yes
			log = "emf_antinomad(1342, '[Root.GetName]') => tribal / zanata / miaphysite"
			break = yes
		}
		if = {
			limit = { culture = zanata religion = sunni }
			break = yes # Not necessary to build a settlement
		}
		build_holding = { type = tribal }
		culture  = zanata
		religion = sunni
		emf_nomad_antinomad_tribal_effect = yes
		log = "emf_antinomad(1342, '[Root.GetName]') => tribal / zanata / sunni"
	}
}

	hide_window = yes
	is_triggered_only = yes

	trigger = {
		owner = { is_nomadic = yes }
		NOT = { any_province_holding = { NOT = { holding_type = nomad } } }
		has_empty_holding = yes
	}

	immediate = {
		if = {
			limit = { culture = kirghiz religion = tengri_pagan }
			break = yes # Not necessary to build a settlement
		}
		build_holding = { type = temple }
		culture  = kirghiz
		religion = tengri_pagan
		emf_nomad_antinomad_temple_effect = yes
		log = "emf_antinomad(1575, '[Root.GetName]') => temple / kirghiz / tengri_pagan"
	}
}
