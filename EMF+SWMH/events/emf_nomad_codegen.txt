# emf_nomad.1511 [511 - Galaz]
province_event = {
	id = emf_nomad.1511

	is_triggered_only = yes
	hide_window = yes

	trigger = {
		owner = { is_nomadic = yes }
		NOT = { any_province_holding = { NOT = { holding_type = nomad } } }
		has_empty_holding = yes
	}

	immediate = {
		if = {
			limit = { NOT = { year = 936 } }
			if = {
				limit = { culture = bulgarian religion = tengri_pagan }
				break = yes # Not necessary to build a settlement
			}
			build_holding = { type = tribal }
			culture  = bulgarian
			religion = tengri_pagan
			emf_nomad_antinomad_tribal = yes
			log = "emf_antinomad(511, '[Root.GetName]') => tribal / bulgarian / tengri_pagan"
			break = yes
		}
		if = {
			limit = { NOT = { year = 1125 } }
			if = {
				limit = { culture = pecheneg religion = tengri_pagan }
				break = yes # Not necessary to build a settlement
			}
			build_holding = { type = tribal }

			log = "emf_antinomad(511, '[Root.GetName]') => tribal / pecheneg / tengri_pagan"
