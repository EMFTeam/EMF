	emf_create_hre = {
		is_high_prio = yes

		potential = {
			emf_hiprio_decision_basic_potential_trigger = yes
			has_global_flag = charlemagne_hre
			independent = yes
			tier = king
			is_feudal = yes
			OR = {
				religion = catholic
				AND = {
					religion = fraticelli
					is_heretic = no
				}
			}
			e_hre = {
				has_holder = no
			}
			e_hre_french = {
				has_holder = no
			}
			OR = {
				has_landed_title = k_germany
				has_landed_title = k_bavaria
				has_landed_title = k_italy
				has_landed_title = k_lotharingia
			}
		}
		allow = {
			has_landed_title = k_germany
			has_landed_title = k_lotharingia
			has_landed_title = k_italy
			realm_size = 180
			prestige = 1000
			war = no
			has_regent = no
		}
		revoke_allowed = {
			always = no
		}
		effect = {
			set_global_flag = emf_hre_formed
			set_character_flag = emf_hre_founder
			prestige = 1000

			if = {
				limit = { has_nickname = no }
				random_list = {
					60 = { give_nickname = nick_the_great }
					20 = { give_nickname = nick_the_restorer }
					20 = { give_nickname = nick_the_lawgiver }
				}
			}

			set_character_flag = emf_no_law_penalties
			primary_title = {
				e_hre = {
					grant_title = ROOT
					copy_title_laws = PREV
					copy_title_history = PREV
					tooltip = { succession = feudal_elective }
					hidden_tooltip = { add_law = succ_feudal_elective }
				}
			}
			clr_character_flag = emf_no_law_penalties

			# Destroy kingdoms that are no longer needed.
			any_demesne_title = {
				limit = {
					tier = KING
					OR = {
						title = k_lotharingia
						title = k_bohemia
						de_jure_liege = e_hre
					}
				}
				destroy_landed_title = THIS
			}

			hidden_tooltip = { recalc_succession = yes }

			# Map-specific de jure adjustments post-formation
			hidden_tooltip = { character_event = { id = emf_core.30 } }

			hidden_tooltip = { # TODO: Custom tooltip.
				if = {
					limit = { ai = no }
					add_character_modifier = {
						name = emf_cb_mod_dejure_kingdom_claim
						duration = 5475
					}
				}
				if = {
					limit = { ai = yes }
					add_character_modifier = {
						name = emf_cb_mod_dejure_kingdom_claim
						duration = 10950
						inherit = yes
					}
				}
			}
		}
		ai_will_do = {
			factor = 1
		}
	}

	emf_restore_byzantium = {
		is_high_prio = yes

		potential = {
			emf_hiprio_decision_basic_potential_trigger = yes
			independent = yes
			tier = KING
			has_landed_title = c_byzantion
			is_feudal = yes
			e_byzantium = { has_holder = no }
			e_roman_empire = { has_holder = no }
			OR = {
				culture_group = byzantine
				AND = {
					ai = no
					culture_group = south_slavic
				}
			}
			OR = {
				religion = orthodox
				religion = bogomilist
				religion = paulician
				religion = monothelite
				religion = iconoclast
				AND = {
					ai = no
					OR = {
						religion = nestorian
						religion = messalian
						religion = miaphysite
						religion = monophysite
					}
				}
			}
		}
		allow = {
			war = no
			has_regent = no
			prestige = 1000
			realm_size = 75
			emf_empires_no_empire_same_religion_trigger = yes
			k_byzantium = {
				de_jure_liege = e_byzantium
				emf_ROOT_completely_controls_counties_trigger = yes
			}
			c_byzantion = {
				custom_tooltip = {
					text = emf_empires_ctt_completely_controls_county_besides_hagiasophia
					hidden_tooltip = {
						NOT = {
							any_direct_de_jure_vassal_title = {
								NOT = { title = b_hagiasophia }
								holder_scope = {
									NOR = {
										character = ROOT
										is_liege_or_above = ROOT
									}
								}
							}
						}
					}
				}
			}
			OR = {
				NOT = { religion = orthodox }
				b_hagiasophia = {
					holder_scope = {
						OR = {
							character = ROOT
							is_liege_or_above = ROOT
							has_landed_title = k_orthodox
						}
					}
				}
			}
		}
		revoke_allowed = {
			always = no
		}
		effect = {
			clr_global_flag = emf_byz_restored
			set_global_flag = emf_byz_restored
			set_character_flag = emf_achievement_byz_restored

			if = {
				limit = { has_nickname = no }
				random_list = {
					60 = { give_nickname = nick_the_restorer }
					20 = { give_nickname = nick_the_great }
					20 = { }
				}
			}

			if = {
				limit = { NOT = { is_title_active = e_byzantium } }
				activate_title = { title = e_byzantium status = yes }
			}

			set_character_flag = emf_no_law_penalties
			primary_title = {
				e_byzantium = {
					grant_title = ROOT
					copy_title_laws = PREV
					add_law = administration_2
					if = {
						limit = {
							PREV = {
								NOR = {
									has_law = crown_authority_2
									has_law = crown_authority_3
									has_law = crown_authority_4
								}
							}
						}
						add_law = crown_authority_2
						if = {
							limit = {
								has_dlc = "Zeus"
								NOT = { has_law = ze_revokation_2 }
							}
							add_law = ze_revokation_2
						}
						if = {
							limit = {
								NOT = { has_dlc = "Zeus" }
								NOT = { has_law = revokation_2 }
							}
							add_law = revokation_2
						}
					}
					# We don't use add_law so that it doesn't consume the lifetime cooldown upon
					# succession law changes, should a player NOT want to be semi-irreversibly forced
					# into primogeniture. Hence, the recalc_succession afterward.
					succession = primogeniture
				}
			}
			clr_character_flag = emf_no_law_penalties
			hidden_tooltip = { recalc_succession = yes }
			
			# Should the empire disintegrate again, we want the timer to start over for how long the
			# emf_reclaim_byz_capital CB can be used before expiring.
			clr_global_flag = emf_reclaim_byz_capital_success
			
			# Potentially give Hagia Sophia to our religious head if appropriate and if it's not theirs already,
			# and vassalize the religious head title
			
			if = {
				limit = { religion = orthodox }
				k_orthodox = { holder_scope = { save_event_target_as = emf_relhead } }
			}
			if = {
				limit = { religion = paulician }
				d_paulician = { holder_scope = { save_event_target_as = emf_relhead } }
			}
			if = {
				limit = { religion = monothelite }
				d_monothelite = { holder_scope = { save_event_target_as = emf_relhead } }
			}
			if = {
				limit = { religion = iconoclast }
				d_iconoclast = { holder_scope = { save_event_target_as = emf_relhead } }
			}
			if = {
				limit = { religion = miaphysite }
				d_miaphysite = { holder_scope = { save_event_target_as = emf_relhead } }
			}
			if = {
				limit = { religion = monophysite }
				d_monophysite = { holder_scope = { save_event_target_as = emf_relhead } }
			}
			if = {
				limit = { religion = nestorian }
				d_nestorian = { holder_scope = { save_event_target_as = emf_relhead } }
			}
			
			b_hagiasophia = { holder_scope = { save_event_target_as = emf_temple_owner } }
			
			if = {
				limit = {
					event_target:emf_relhead = { # The religious head is indeed valid
						NOT = { # No other player is liege to the religious head
							any_liege = {
								NOT = { character = ROOT }
								ai = no
							}
						}
						OR = { # If not orthodox or iconoclast, rel. head must not hate ruler
							religion = orthodox
							religion = iconoclast
							opinion = { who = ROOT value = -50 }
						}
					}
				}
				if = {
					limit = {
						event_target:emf_temple_owner = {
							OR = {
								character = ROOT
								is_liege_or_above = ROOT
								ai = yes
							}
						}
					}
					b_hagiasophia = {
						grant_title_no_opinion = event_target:emf_relhead
						remove_claim = event_target:emf_temple_owner
					}
				}
				event_target:emf_relhead = {
					set_defacto_liege = ROOT
					pf_liege_change_effect = yes
				}
			}
			
			# Varangian Guard vassalization, if they have indeed been formed
			d_varangian_guard = {
				if = {
					limit = {
						is_title_active = THIS
						holder_scope = {
							OR = {
								independent = yes
								liege = {
									ai = yes
								}
							}
						}
					}
					holder_scope = {
						set_defacto_liege = ROOT
						pf_liege_change_effect = yes
					}
				}
			}

			# TODO: Do similar to the de jure kingdom claim as below with temporarily free access to
			# king-tier emf_subjugation.

			hidden_tooltip = { # TODO: Custom tooltip.
				if = {
					limit = { ai = no }
					add_character_modifier = {
						name = emf_cb_mod_dejure_kingdom_claim
						duration = 5475
					}
				}
				if = {
					limit = { ai = yes }
					add_character_modifier = {
						name = emf_cb_mod_dejure_kingdom_claim
						duration = 10950
						inherit = yes
					}
				}
			}
		}
		ai_will_do = {
			factor = 1
		}
	}
