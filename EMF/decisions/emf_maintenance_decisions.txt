# -*- ck2.decisions -*-

decisions = {

	emf_quarterly_pulse = {
		only_independent = yes
		is_high_prio = yes

		potential = {
			has_landed_title = e_hip
			OR = {
				NOT = { has_global_flag = emf_quarter_pulse }
				had_global_flag = { flag = emf_quarter_pulse days = 45 }
			}
			OR = {
				month = 11 # Winter
				AND = {
					month = 2 # Spring
					NOT = { month = 3 }
				}
				AND = {
					month = 5 # Summer
					NOT = { month = 6 }
				}
				AND = {
					month = 8 # Autumn
					NOT = { month = 9 }
				}
			}
		}
		allow = {}
		effect = {
			clr_global_flag = emf_quarter_pulse
			set_global_flag = emf_quarter_pulse

			### Weather Module ###
			narrative_event = { id = emf_regional.0 }
			any_player = {
				character_event = { id = emf_regional.1 }
			}
		}
		ai_will_do = {
			factor = 100
		}
	}

	ai_annual_maintenance = {
		only_independent = yes
		potential = {
			has_landed_title = e_hip
			OR = {
				NOT = { has_global_flag = emf_annual_pulse }
				had_global_flag = { flag = emf_annual_pulse days = 365 }
			}
		}
		allow = {}
		effect = {
			clr_global_flag = emf_annual_pulse
			set_global_flag = emf_annual_pulse

			# clean up flags (essentially all rulers in the game, ideally would
			# be amortized, but ideal is far from working ATM)
			any_playable_ruler = {
				clr_flag = pf_pulse
				if = {
					limit = { war = no }
					clr_flag = faction_court_war
					clr_flag = faction_prosperity_war
					clr_flag = faction_glory_war
					clr_flag = faction_tradition_war
				}
				if = {
					limit = {
						pf_leader_trigger = no
						OR = {
							has_flag = faction_meeting
							has_flag = faction_meeting_called
							has_flag = faction_meeting_begun
							has_flag = faction_meeting_postponed
							has_flag = faction_leadership_challenge
						}
					}
					clr_flag = faction_meeting
					clr_flag = can_make_demand
					clr_flag = faction_meeting_called
					clr_flag = faction_meeting_begun
					clr_flag = faction_meeting_postponed
					clr_flag = faction_leadership_challenge
					clr_flag = faction_is_happy
					clr_flag = faction_is_content
					clr_flag = faction_is_unhappy
					clr_flag = faction_is_angry
					clr_flag = made_great_speech
					clr_flag = made_good_speech
					clr_flag = made_fair_speech
					clr_flag = made_poor_speech
					clr_flag = made_terrible_speech
					clr_flag = liege_opinion_increased
					clr_flag = liege_opinion_slightly_increased
					clr_flag = liege_opinion_slightly_decreased
					clr_flag = liege_opinion_decreased
					clr_flag = demand_money_liege
					clr_flag = demand_support_liege
					clr_flag = faction_demand_refused
				}
				if = {
					limit = {
						OR = {
							had_flag = { flag = faction_meeting_called days = 365 }
							had_flag = { flag = faction_meeting_begun days = 365 }
							had_flag = { flag = faction_leadership_challenge days = 365 }
						}
					}
					clr_flag = can_make_demand
					clr_flag = faction_meeting_called
					clr_flag = faction_meeting_begun
					clr_flag = faction_meeting_postponed
					clr_flag = faction_leadership_challenge
					clr_flag = faction_is_happy
					clr_flag = faction_is_content
					clr_flag = faction_is_unhappy
					clr_flag = faction_is_angry
					clr_flag = made_great_speech
					clr_flag = made_good_speech
					clr_flag = made_fair_speech
					clr_flag = made_poor_speech
					clr_flag = made_terrible_speech
					clr_flag = liege_opinion_increased
					clr_flag = liege_opinion_slightly_increased
					clr_flag = liege_opinion_slightly_decreased
					clr_flag = liege_opinion_decreased
					clr_flag = demand_money_liege
					clr_flag = demand_support_liege
					clr_flag = faction_demand_refused
				}
				if = {
					limit = { had_flag = { flag = former_faction_leader days = 7 } }
					clr_flag = former_faction_leader
				}
				if = {
					limit = { had_flag = { flag = faction_reboot days = 7 } }
					clr_flag = faction_reboot
				}
				# Clear do_not_disturb flag if it's been set for an obviously-errant duration (> 3 years)
				if = {
					limit = {
						NOT = { trait = in_hiding }
						NOT = {
							primary_title = {
								OR = {
									adventurer = yes
									has_flag = emf_revolt
								}
							}
						}
						had_flag = { flag = do_not_disturb days = 1095 }
					}
					clr_flag = do_not_disturb
				}
				# Years of peace (supplemented by emf_cb_on_add to reset it upon war declarations)
				if = {
					limit = {
						higher_real_tier_than = COUNT
						is_primary_war_participant = no
					}
					change_variable = { which = emf_peace_years value = 1 }
				}
				if = {
					limit = {
						lower_real_tier_than = DUKE
						check_variable = { which = emf_peace_years value = 0.1 }
					}
					set_variable = { which = emf_peace_years value = 0 }
				}
				# Remove inappropriate tributaries
				if = {
					limit = {
						independent = no
						is_tributary = yes
					}
					any_suzerain = { remove_tributary = PREV }
				}
				# Enable PF maintenance pulse
				if = {
					limit = {
						higher_real_tier_than = COUNT
						is_nomadic = no
						is_landed = yes
						any_vassal = { pf_member_trigger = yes }
					}
					set_flag = pf_pulse
				}
			}
		}
		ai_will_do = {
			factor = 100
		}
	}
}
