decisions = {
	#Marriage decisions
	toggle_marriage_decisions = {
		potential = {
			ai = no
		}
		effect = {
			if = {
				limit = {
					NOT = { has_character_flag = enable_marriage_decisions }
				}
				hidden_tooltip = {
					set_character_flag = enable_marriage_decisions
					set_character_flag = enable_marriage_decisions_temp
				}
			}
			if = {
				limit = {
					has_character_flag = enable_marriage_decisions
					NOT = { has_character_flag = enable_marriage_decisions_temp }
				}
				hidden_tooltip = {
					clr_character_flag = enable_marriage_decisions
				}
			}
			clr_character_flag = enable_marriage_decisions_temp
		}
	}
	marry_courtiers = {
		potential = {
			has_character_flag = enable_marriage_decisions
		}
		allow = {
			any_courtier = {
				can_marry = yes
				NOT = { dynasty = ROOT }
				is_married = no
				age = 16
				is_female = no
				ROOT = {
					any_courtier = {
						can_marry = PREVPREV
						NOT = { dynasty = ROOT }
						is_married = no
						age = 16
						NOT = { age = 45 }
						is_female = yes
					}
				}
			}
		}
		effect = {
			any_courtier = {
				limit = {
					can_marry = yes
					NOT = { dynasty = ROOT }
					is_married = no
					age = 16
					NOT = { age = 45 }
					is_female = yes
				}
				ROOT = {
					random_courtier = {
						limit = {
							can_marry = yes
							NOT = { dynasty = ROOT }
							can_marry = PREVPREV
							is_married = no
							age = 16
							is_female = no
						}
						add_spouse = PREVPREV
					}
				}
			}
		}
	}
}
