			NOT = { is_title_active = d_fraticelli }
			has_dlc = "Sons of Abraham"
		}
		
		allow = {

			}
		}
		
		effect = {
			activate_title = { title = d_fraticelli status = yes }
			hidden_tooltip = {

				new_character = {

						grant_title = PREV

						modifier = opinion_creator_of_rel_heal
					}
				}

					modifier = papacy_instituted
					years = 20
				}
			}

		}

			always = no
		}

			factor = 1
		}

	restore_pope_in_rome = {
		is_high_prio = yes
	
		potential = {
			is_playable = yes
			independent = yes
			has_dlc = "Sons of Abraham"
			had_global_flag = { flag = HIP days = 1 }
			
			OR = {
				religion = catholic
				AND = {
					religion = fraticelli
					is_title_active = d_fraticelli
				}
			}
			
			controls_religion = no

			rightful_religious_head_scope = {
				OR = {
					controls_religion = yes
					AND = {
						religion = catholic
						has_claim = k_papal_state
						k_papal_state = {
							holder_scope = {
								is_landed = no
							}
						}
					}
					AND = {
						religion = fraticelli
						has_claim = d_fraticelli
						d_fraticelli = {
							holder_scope = {
								is_landed = no
							}
						}
					}
				}
			}
			
			c_roma = {
				holder_scope = {
					OR = {
						character = ROOT
						is_liege_or_above = ROOT
					}
					rightful_religious_head_scope = {
						NOR = {
							character = PREV
							is_vassal_or_below = PREV
						}
					}
				}
			}
		}
		
		effect = {
			rightful_religious_head_scope = {
				if = {
					limit = {
						ROOT = { religion = catholic }
						has_claim = k_papal_state
						k_papal_state = {
							holder_scope = {
								is_landed = no
							}
						}
					}
					k_papal_state = {
						holder_scope = {
							transfer_scaled_wealth = {
								to = PREVPREV
								value = all
							}
							abdicate_to = PREVPREV
						}
					}
					hidden_tooltip = {
						if = {
							limit = {
								ROOT = { higher_real_tier_than = PREV }
							}
							set_defacto_liege = ROOT
							pf_liege_change_effect = yes
						}
					}
				}
				if = {
					limit = {
						ROOT = { religion = fraticelli }
						has_claim = d_fraticelli
						d_fraticelli = {
							holder_scope = {
								is_landed = no
							}
						}
					}
					d_fraticelli = {
						holder_scope = {
							transfer_scaled_wealth = {
								to = PREVPREV
								value = all
							}
							abdicate_to = PREVPREV
						}
					}
					hidden_tooltip = {
						if = {
							limit = {
								ROOT = { higher_real_tier_than = PREV }
							}
							set_defacto_liege = ROOT
							pf_liege_change_effect = yes
						}
					}
				}
				c_roma = {
					grant_title = PREV
					any_de_jure_vassal_title = {
						limit = {
							holder_scope = {
								OR = {
									character = ROOT
									is_liege_or_above = ROOT
								}
							}
						}
						grant_title = PREVPREV
					}
				}
			}
			if = {
				limit = {
					religion = catholic
					NOT = { has_character_flag = restored_catholic_pope_in_rome }
				}
				k_papal_state = {
					holder_scope = {
						opinion = {
							who = ROOT
							modifier = opinion_restored_to_rome
						}
					}
				}
				piety = 1000
				set_character_flag = restored_catholic_pope_in_rome
			}
			if = {
				limit = {
					religion = fraticelli
					NOT = { has_character_flag = restored_fraticelli_pope_in_rome }
				}
				d_fraticelli = {
					holder_scope = {
						opinion = {
							who = ROOT
							modifier = opinion_restored_to_rome
						}
					}
				}
				piety = 1000
				set_character_flag = restored_fraticelli_pope_in_rome
			}
		}
		
		revoke_allowed = {
			always = no
		}
		
		ai_will_do = {
			factor = 1
			modifier = {
				factor = 0
				NOT = {
					any_demesne_province = {
						NOT = { province_id = 333 }
					}
				}
			}
			modifier = {
				factor = 0
				c_roma = {
					holder_scope = {
						OR = {
							ai = no
							any_liege = { ai = no }  # AI will never take Rome province away from a player
						}
					}
				}
			}
		}
	}

				}
			}
		}
