

targetted_decisions = {
	emf_bring_child_home = {
		is_high_prio = yes
		
		filter = dynasty
		ai_target_filter = self
		
		from_potential = {
			ai = no
		}
		potential = {
			is_child_of = FROM
			liege = { not = { character = FROM } }
			not = {
				prisoner = yes
				is_ruler = yes
				is_adult = yes
				is_married = yes
			}
		}
		allow = {
			liege = {
				or = {
					ai = yes
					lower_tier_than = FROM
				}
			}
		}
		effect = {
			move_character = FROM
		}
		revoke_allowed = {
			always = no
		}
		ai_will_do = {
			factor = 0
		}
	}
	
	emf_court_banishment = {
		filter = court
		ai_target_filter = self
		
		from_potential = {
			ai = no
		}
		potential = {
			not = { character = FROM }
			host = { character = FROM }
		}
		allow = {
			FROM = {
				prestige = 50
				has_regent = no
			}
			ROOT = {
				prisoner = no
				is_councillor = no
				custom_tooltip = {
					text = emf_ctt_is_not_guardian
					hidden_tooltip = {
						not = {
							any_ward = { always = yes }
						}
					}
				}
				is_adult = yes
			}
		}
		effect = {
			if = {
				limit = { religion = FROM }
				custom_tooltip = { text = emf_court_banishment_tt_samerel }
			}
			if = {
				limit = {
					not = { religion = FROM }
				}
				custom_tooltip = { text = emf_court_banishment_tt_difrel }
			}
			opinion = {
				modifier = opinion_banished_from_court
				who = FROM
				months = 60
			}
			hidden_tooltip = {
				if = { # Same religion as liege
					limit = { religion = FROM }
					FROM = {
						if = { # Go to dynasty members first
							limit = {
								any_realm_lord = {
									not = { character = FROM }
									not = { dynasty = none }
									dynasty = ROOT
								}
							}
							random_realm_lord = {
								limit = {
									not = { character = FROM }
									not = { dynasty = none }
									dynasty = ROOT
								}
								ROOT = { move_character = PREV }
							}
							break = yes
						}
						random_realm_lord = { # If no dynasty match, pick a random
							limit = {
								not = { character = FROM }
								is_playable = yes
							}
							ROOT = { move_character = PREV }
						}
					}
					break = yes
				}
				if = {  # If different religion, go to realm with same faith if possible
					limit = {
						any_independent_ruler = { religion = ROOT }
					}
					random_independent_ruler = {
						limit = { religion = ROOT }
						ROOT = { move_character = PREV }
					}
					break = yes
				}
				random_realm_lord = { # Otherwise just get the heck out
					limit = {
						not = { character = FROM }
						is_playable = yes
					}
					ROOT = { move_character = PREV }
				}
			}
		}
		revoke_allowed = {
			always = no
		}
		ai_will_do = {
			factor = 0
		}
	}
}
