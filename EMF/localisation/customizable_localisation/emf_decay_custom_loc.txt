# -*- ck2.custom_loc -*-


defined_text = { 
	name = EMF_GetImperialDecayStatus
	text = {
		localisation_key = String_GoodImperialDecay
		trigger = {
			emf_uses_imperial_decay = yes
			primary_title = {
				NOT = { check_variable = { which = "imperial_decay" value = 20 } }
			}
		}
	}
	text = {
		localisation_key = String_OKImperialDecay
		trigger = {
			emf_uses_imperial_decay = yes
			primary_title = {
				check_variable = { which = "imperial_decay" value = 20 }
				NOT = { check_variable = { which = "imperial_decay" value = 40 } }
			}
		}
	}
	text = {
		localisation_key = String_BadImperialDecay
		trigger = {
			emf_uses_imperial_decay = yes
			primary_title = {
				check_variable = { which = "imperial_decay" value = 40 }
				NOT = { check_variable = { which = "imperial_decay" value = 60 } }
			}
		}
	}
	text = {
		localisation_key = String_WorstImperialDecay
		trigger = {
			emf_uses_imperial_decay = yes
			primary_title = {
				check_variable = { which = "imperial_decay" value = 60 }
			}
		}
	}
	fallback_text = { localisation_key = String_ASCII_SPACE }
}
