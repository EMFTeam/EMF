# Birth of the Arsiyah
narrative_event = {
	id = emf_holy.0
	title = emf_holy.0.title
	desc = emf_holy.0.desc
	picture = GFX_evt_moors
	
	major = yes
	hide_from = yes
	
	only_playable = yes
	religion_group = muslim
	has_dlc = "Sons of Abraham"
	has_global_flag = muslim_jihads_unlocked
	
	trigger = {
		emf_holy_order_arsiyah_religion = yes
		NOT = { is_title_active = d_arsiyah }
	}
	
	mean_time_to_happen = {
		months = 6
	}
	
	immediate = {
		activate_title = { title = d_arsiyah status = yes }
		create_character = {
			random_traits = no
			dynasty = random
			religion = ROOT
			culture = khazar
			female = no
			age = 25
			attributes = {
				martial = 10
				diplomacy = 9
				stewardship = 9
				intrigue = 7
				learning = 9
			}
			health = 6
			fertility = 0.5
			trait = brilliant_strategist
			trait = hajjaj
			trait = hafiz
			trait = patient
			trait = zealous
			trait = scholar
			trait = temperate
		}
		new_character = {
			wealth = 500
			d_arsiyah = {
				grant_title = PREV
			}
			emf_new_character_effect = yes
		}
	}
	
	option = {
		name = emf_holy.0.a
		trigger = {
			religion = ROOT
		}
	}
	option = {
		name = emf_holy.0.b
		trigger = {
			NOT = { religion = ROOT }
		}
	}
}

# The Grand Master requests the right to build a castle in your demesne
character_event = {
	id = emf_holy.5
	desc = OK
	picture = GFX_evt_moors
	border = GFX_event_normal_frame_religion
	
	hide_window = yes
	
	only_playable = yes
	religion_group = muslim
	has_dlc = "Sons of Abraham"
	
	trigger = {
		emf_holy_order_arsiyah_religion = yes
		d_arsiyah = {
			holder_scope = {
				wealth = 300
				NOT = {
					realm_size = 10
				}
				NOT = {
					num_of_count_titles = 1
				}
			}
		}
		
		any_demesne_province = {
			has_empty_holding = yes
			port = yes
		}
		
		NOT = { has_character_modifier = expelled_d_arsiyah }
		NOT = {
			any_liege = {
				has_character_modifier = expelled_d_arsiyah
			}
		}
	}
	
	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = 0.5
			d_arsiyah = {
				holder_scope = {
					NOT = { demesne_size = 1 }
				}
			}
		}
		modifier = {
			factor = 0.5
			d_arsiyah = {
				holder_scope = {
					NOT = { demesne_size = 2 }
				}
			}
		}
		modifier = {
			factor = 2.0
			d_arsiyah = {
				holder_scope = {
					realm_size = 5
				}
			}
		}
	}
	
	immediate = {
		random_demesne_province = {
			limit = {
				has_empty_holding = yes
				port = yes
			}
			province_event = {
				id = emf_holy.6
			}
		}
	}
	
	option = {
		name = OK
	}
}
