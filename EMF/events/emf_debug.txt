		FROM = {
			save_event_target_as = emf_owner

				limit = { location = { has_owner = yes } }

				break = yes

			if = {
				limit = { capital_scope = { has_owner = yes } }

				limit = { primary_title = { capital_scope = { has_owner = yes } } }
