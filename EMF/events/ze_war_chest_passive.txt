namespace = ZE

# Vassal offer donation in return for favor
character_event = {
	id = ZE.22000
	desc = EVTDESC_ZE_22000
	picture = GFX_evt_bribe_accepted

	is_triggered_only = yes


	# Thank you!
	option = {
		name = EVTOPTA_ZE_22000

		ai_chance = {
			factor = 50

			modifier = {
				factor = 5
				NOT = { wealth = 0 }
			}

			modifier = {
				factor = 2
				trait = greedy
			}

			modifier = {
				factor = 2
				trait = trusting
			}

			modifier = {
				factor = 2
				trait = slothful
			}

			modifier = {
				factor = 2
				NOT = { wealth = 100 }
				independent = yes
			}
		}

		trigger = {
			FROM = {
				NAND = {
					is_friend = ROOT
					ai = yes
				}
			}
		}

		hidden_tooltip = {
			FROM = { character_event = { id = ZE.22100 days = 3 } }
		}
		wealth = 200
		tooltip = {
			FROM = {
				add_favor = ROOT
			}
		}
	}

	# No thanks
	option = {
		name = EVTOPTB_ZE_22000

		ai_chance = {
			factor = 50

			modifier = {
				factor = 5
				trait = proud
			}

			modifier = {
				factor = 2
				FROM = { is_councillor = yes }
			}

			modifier = {
				factor = 2
				trait = paranoid
			}

			modifier = {
				factor = 2
				trait = content
			}

			modifier = {
				factor = 2
				wealth = 300
				independent = yes
			}
		}

		hidden_tooltip = {
			FROM = { character_event = { id = ZE.22200 days = 3 } }
		}

		tooltip = {
			opinion = {
				who = FROM
				modifier = opinion_insulted
				years = 3
			}
		}
	}

	option = { # Thank you, friend!
		name = EVTOPTC_ZE_22000

		ai_chance = {
			factor = 100
		}

		trigger = {
			FROM = {
				is_friend = ROOT
				ai = yes
			}
		}

		wealth = 200
	}
}
