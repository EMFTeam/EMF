		OR = {

			AND = {
				martial = 12

					amount = 2

					diplomacy = 12

# Check to see if the raiding adventurer is ready to settle (once every three years)
character_event = {
	id = HL.110
	
	is_triggered_only = yes
	hide_window = yes
	
	has_character_flag = raiding_adventurer
	
	trigger = {
		primary_title = {
			temporary = yes
		}
	}
	
	immediate = {
		log = "HL.110: checking if raiding adventurer [Root.GetBestName] (id:[Root.GetID]) is ready to settle"
		if = {
			limit = { # If this is true, prepare to settle.
				OR = {
					wealth = 150
					had_character_flag = { flag = raiding_adventurer days = 1825 }
				}
				any_playable_ruler = {
					is_landed = yes
					has_opinion_modifier = { who = ROOT modifier = opinion_looted }
					top_liege = { NOT = { realm_levy_diff = { who = ROOT value = 3000 } } }
					any_realm_province = {
						OR = {
							culture = ROOT
							emf_province_same_region_as_home_trigger = yes
							emf_province_same_sea_zone_as_home_trigger = yes
						}
					}
				}
			}
			random_list = {
				50 = {
					modifier = {
						trait = cruel
						factor = 2
					}
					modifier = {
						trait = ambitious
						factor = 2
					}
					log = "---> [Root.GetBestName] prepares to settle."
					character_event = { id = HL.111 }
					break = yes
				}
				20 = {
					modifier = {
						factor = 0
						NOT = { trait = lunatic }
						NOT = { trait = possessed }
						event_target:target_province = {
							owner = {
								top_liege = {
									realm_levy_diff = { who = ROOT value = 3000 }
								}
							}
						}
					}
					modifier = {
						NOR = {
							has_character_flag = flag_denied_title
							trait = ambitious
							trait = lunatic
							trait = possessed
							event_target:target_province = {
								owner = {
									NOT = { culture = ROOT }
									NOT = { religion = ROOT }
								}
							}
						}
						factor = 0
					}
					modifier = {
						factor = 0
						NOT = { has_character_flag = flag_denied_title }
						event_target:target_province = {
							owner = {
								OR = {
									is_close_relative = ROOT
									reverse_opinion = { who = ROOT value = 25 }
									any_liege = { is_close_relative = ROOT }
									any_liege = { reverse_opinion = { who = ROOT value = 25 } }
								}
							}
						}
					}
					modifier = {
						trait = ambitious
						factor = 2
					}
					modifier = {
						trait = ambitious
						trait = arbitrary
						factor = 3
					}
					modifier = {
						trait = envious
						factor = 2
					}
					modifier = {
						event_target:target_province = {
							owner = {
								NOT = { culture = ROOT }
								NOT = { religion = ROOT }
							}
						}
						factor = 3
					}
					log = "---> [Root.GetBestName] returns home in a state of war."
					character_event = { id = HL.113 }
					break = yes
				}
				50 = {
					modifier = {
						had_character_flag = { flag = raiding_adventurer days = 2555 }
						factor = 0.5
					}
					modifier = {
						had_character_flag = { flag = raiding_adventurer days = 3285 }
						factor = 0.5
					}
					modifier = {
						had_character_flag = { flag = raiding_adventurer days = 4015 }
						factor = 0.5
					}
					modifier = {
						had_character_flag = { flag = raiding_adventurer days = 10950 }
						factor = 0
					}
					log = "---> [Root.GetBestName] continues raiding."
					repeat_event = { id = HL.110 days = 365 } # continue raiding
					break = yes
				}
			}
		}
		# Check for stagnant raider
		if = {
			limit = {
				had_character_flag = { flag = raiding_adventurer days = 10950 }	
			}
			log = "---> [Root.GetBestName] is stagnant and will attempt to find a duchy adventure target (NOTE: this is likely to fail)"
			character_event = { id = HL.111 }
			break = yes
		}
		
		# Else continue raiding.
		log = "---> [Root.GetBestName] continues raiding."
		repeat_event = { id = HL.110 days = 365 }
	}
}

# Raider is prepared to settle
character_event = {
	id = HL.111
	is_triggered_only = yes
	hide_window = yes
	
	immediate = {
		any_army = {
			set_can_toggle_looting = yes
			set_looting = no
		}
		clr_character_flag = raiding_adventurer
		set_character_flag = duchy_adventurer
		
		random_playable_ruler = {
			limit = {
				is_landed = yes
				has_opinion_modifier = { who = ROOT modifier = opinion_looted }
				top_liege = { NOT = { realm_levy_diff = { who = ROOT value = 3000 } } }
				any_realm_province = {
					OR = {
						culture_group = ROOT
						emf_province_same_region_as_home_trigger = yes
						emf_province_same_sea_zone_as_home_trigger = yes
					}
				}
			}
			top_liege = { save_event_target_as = target_liege }
			if = {
				limit = {
					ROOT = { is_seafarer = yes }
				}
				random_realm_province = {
					limit = { port = yes }
					ROOT = {
						unsafe_war = {
							target = event_target:target_liege
							casus_belli = duchy_adventure
							thirdparty_title = PREV
							tier = DUKE
						}
					}
					if = {
						limit = { ROOT = { war = yes } }
						log = "INFO: HL.111: raiding adventurer [Root.GetTitledFirstName] (ID:[Root.GetID]) has selected the [This.Duchy.GetFullName] as his duchy adventure target"
						owner = {
							character_event = { id = HL.112 }
							if = {
								limit = { independent = no }
								top_liege = { character_event = { id = HL.112 } }
							}
						}
					}
				}
			}
			if = {
				limit = {
					ROOT = { war = no }
				}
				random_realm_province = {
					limit = {
						OR = {
							port = yes
							any_neighbor_province = { NOT = { owner = { same_realm = PREVPREVPREV } } }
						}
					}
					ROOT = {
						unsafe_war = {
							target = event_target:target_liege
							casus_belli = duchy_adventure
							thirdparty_title = PREV
							tier = DUKE
						}
					}
					if = {
						limit = { ROOT = { war = yes } }
						log = "INFO: HL.111: raiding adventurer [Root.GetTitledFirstName] (ID:[Root.GetID]) has selected the [This.Duchy.GetFullName] as his duchy adventure target"
						owner = {
							character_event = { id = HL.112 }
							if = {
								limit = { independent = no }
								top_liege = { character_event = { id = HL.112 } }
							}
						}
					}
				}
			}
			if = {
				limit = {
					ROOT = { war = no }
				}			
				random_realm_province = {
					ROOT = {
						unsafe_war = {
							target = event_target:target_liege
							casus_belli = duchy_adventure
							thirdparty_title = PREV
							tier = DUKE
						}
					}
					if = {
						limit = { ROOT = { war = yes } }
						log = "INFO: HL.111: raiding adventurer [Root.GetTitledFirstName] (ID:[Root.GetID]) has selected the [This.Duchy.GetFullName] as his duchy adventure target"
						owner = {
							character_event = { id = HL.112 }
							if = {
								limit = { independent = no }
								top_liege = { character_event = { id = HL.112 } }
							}
						}
					}
				}
			}
		}
		if = {
			limit = { war = no }
			log = "SERIOUS: HL.111: raiding adventurer [Root.GetTitledFirstName] (ID:[Root.GetID]) failed to find a target after trying to end his adventure!"
			clr_character_flag = duchy_adventurer
			character_event = { id = HL.199 }
		}
	}
}

			limit = {
				is_close_relative = ROOT
				is_playable = yes
				is_landed = yes
				opinion = { who = ROOT value = 25 }
			}
			save_event_target_as = new_court_home
		}
		random_dynasty_member = {
			limit = {
				NOT = { event_target:new_court_home = { always = yes } }
				is_close_relative = ROOT
				is_playable = yes
				is_landed = yes
			}
			save_event_target_as = new_court_home
		}
		random_dynasty_member = {
			limit = {
				NOT = { event_target:new_court_home = { always = yes } }
				is_playable = yes
				is_landed = yes
			}
			save_event_target_as = new_court_home
		}
		random_playable_ruler = {
			limit = {
				NOT = { event_target:new_court_home = { always = yes } }
				culture = ROOT
				event_target:target_province = {
					owner = { character = PREVPREV }
				}
			}
			save_event_target_as = new_court_home
		}
		random_playable_ruler = {
			limit = {
				NOT = { event_target:new_court_home = { always = yes } }
				culture_group = ROOT
				event_target:target_province = {
					owner = { character = PREVPREV }
				}
			}
			save_event_target_as = new_court_home
		}
		random_playable_ruler = {
			limit = {
				NOT = { event_target:new_court_home = { always = yes } }
				culture = ROOT
			}
			save_event_target_as = new_court_home
		}
		random_playable_ruler = {
			limit = {
				NOT = { event_target:new_court_home = { always = yes } }
				culture_group = ROOT
			}
			save_event_target_as = new_court_home
		}
		random_playable_ruler = {
			limit = {
				NOT = { event_target:new_court_home = { always = yes } }
				event_target:target_province = {
					owner = { character = PREVPREV }
				}
			}
			save_event_target_as = new_court_home
		}

		if = {

		}

		any_courtier = {
			limit = {

			}
			move_character = event_target:new_court_home
		}

		}

					OR = {

							tier = DUKE

		}
