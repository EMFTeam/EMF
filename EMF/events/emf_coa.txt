# -*- ck2.events -*-


namespace = emf_coa

# emf_coa.0
#
# on_new_holder[_inheritance|usurpation] on_action listener for title transfer
# of titles with dynamic CoA rules.
character_event = {
	id = emf_coa.0
	desc = HIDE_EVENT
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		FROM = {
			OR = {
				title = k_england
				title = k_aragon
			}
		}
	}

	immediate = {
		
		if = {
			limit = { FROM = { title = k_england } }
			character_event = { id = emf_coa.2 }
			break = yes
		}
		
		if = {
			limit = { FROM = { title = k_aragon } }
			character_event = { id = emf_coa.3 }
			break = yes
		}
	}
	
	option = { name = OK }
}


# emf_coa.1 [Isis]
#
# Initialize all flags according to dynamic CoA rules. Run upon scenario
# startup and on_biyearly_pulse.
character_event = {
	id = emf_coa.1
	desc = AI_EVENT
	hide_window = yes
	is_triggered_only = yes

	only_independent = yes
	
	trigger = {
		has_landed_title = e_hip
	}
	
	immediate = {

		k_england = {
			owner = {
				character_event = { id = emf_coa.2 }
			}
		}
		k_aragon = {
			owner = {
				character_event = { id = emf_coa.3 }
			}
		}
	}
	
	option = { name = OK }
}



# emf_coa.2 [Holder]
#
# Adjust the flag of k_england based upon holder culture and dynasty
character_event = {
	id = emf_coa.2
	desc = HIDE_EVENT
	hide_window = yes
	is_triggered_only = yes

	immediate = {

		# Plantagenet Lions
		if = {
			limit = {
				OR = {
					dynasty = 752 # de Normandie
					dynasty = 106 # Plantagenet
					culture = norman
				}
			}
			k_emf_norman_england = {
				k_england = {
					set_coa = PREV
				}
			}
			break = yes
		}
		
		# Saxon Dragon
		if = {
			limit = {
				OR = {
					culture = saxon
					culture = anglonorse
				}
			}
			k_emf_saxon_england = {
				k_england = {
					set_coa = PREV
				}
			}
			break = yes
		}
		
		if = {
			limit = {
				religion_group = pagan_group
			}
			k_england = {
				reset_coa = THIS
			}
			break = yes
		}
		
		# Cross of St. George (all other cases)
		k_emf_english_england = {
			k_england = {
				set_coa = PREV
			}
		}
	}
	
	option = { name = OK }
}


# emf_coa.3 [Holder]
#
# Adjust the flag of k_aragon based upon holder dynasty
character_event = {
	id = emf_coa.3
	desc = HIDE_EVENT
	hide_window = yes
	is_triggered_only = yes

	immediate = {

		# de Barcelona dynasty controls Aragon
		if = {
			limit = {
				dynasty = 100204 # de Barcelona
			}
			k_emf_barcelona_aragon = {
				k_aragon = {
					set_coa = PREV
				}
			}
			break = yes
		}
		
		# Default (all other cases)
		k_aragon = {
			reset_coa = THIS
		}
	}
	
	option = { name = OK }
}
