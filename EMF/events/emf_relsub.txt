# -*- ck2.events -*-

namespace = emf_relsub

# emf_relsub.0 -- serve ultimatum to Religious Subjugation war loser (or vassal thereof)
letter_event = {
	id = emf_relsub.0

	is_triggered_only = yes

	desc = {
		text = emf_relsub.0.desc.overlord
		trigger = { character = event_target:emf_cb_loser }
	}

	desc = {
		text = emf_relsub.0.desc.underling
		trigger = {
			NOT = { character = event_target:emf_cb_loser }
			NOT = { religion = event_target:emf_cb_winner }
		}
	}

	desc = {
		text = emf_relsub.0.desc.underling_samerel
		trigger = {
			NOT = { character = event_target:emf_cb_loser }
			religion = event_target:emf_cb_winner
		}
	}

	immediate = {
		if = {
			limit = { character = event_target:emf_cb_loser }
			log = "-> emf_relsub.0: holy war ultimatum received by defending liege [Root.EMF_GetDebugName] from [From.EMF_GetDebugName], a [From.EMF_GetDebugTier]-tier [From.Religion.GetName] ([From.Religion.GetGroupName])"
		}
		if = {
			limit = { NOT = { character = event_target:emf_cb_loser } }
			log = "-> emf_relsub.0: holy war ultimatum received by underling: [Root.EMF_GetDebugName]"
		}
		religion_scope = { save_event_target_as = emf_old_religion }
		character_event = { id = emf_cb.5 } # Invalidate any civil wars against which we might be defending
		character_event = { id = emf_cb.6 } # Invalidate any civil wars which we might be leading
		if = {
			limit = { character = event_target:emf_cb_loser }
			event_target:emf_cb_winner = {
				emf_relsub_reset_statistics = yes
			}
			if = {
				limit = { tier = event_target:emf_cb_winner }
				# [Same-tier defenders] Then the de jure subrealm (non-recursive -- just direct vassals) must
				# also invalidate any of their civil wars, or this could get messy later. Technically, they
				# only need to invalidate the ones in which the [temporarily elevated] vassal ruler is
				# _leading_ a civil war (e.g., COUNT[TEMP. DUKE] vs. DUKE for a KING war loser) here,
				# so we'll do that only.
				any_vassal = {
					limit = { any_realm_title = { de_jure_liege_or_above = event_target:emf_cb_title } }
					character_event = { id = emf_cb.6 } # Invalidate civil wars which we might be leading
				}
			}
		}
		if = {
			limit = {
				any_demesne_title = {
					lower_tier_than = DUKE
					NOT = { ROOT = { family_palace = { title = PREVPREV } } }
					NOT = { de_jure_liege_or_above = event_target:emf_cb_title }
				}
			}
			log = "--> ruler has a landed demesne title outside the target title"
		}
		if = {
			limit = {
				any_realm_title = {
					tier = COUNT
					NOT = { de_jure_liege_or_above = event_target:emf_cb_title }
				}
			}
			log = "--> ruler has a realm county outside the target title"
		}
		if = {
			limit = { emf_relsub_can_submit = no }
			log = "--> ruler is not allowed to submit! reason(s):"
			if = {
				limit = {
					emf_can_convert_religion = no
					NOT = { religion = event_target:emf_cb_winner }
				}
				log = "---> ruler is a [This.Religion.GetName] but is not allowed to convert religion to [emf_cb_winner.Religion.GetName]"
			}
			if = {
				limit = { emf_relsub_can_submit_tier_check = no }
				log = "---> ruler's relative tier prohibits submission:"
				log = "----> attacker: [emf_cb_winner.EMF_GetDebugTier]"
				log = "----> ruler:    [This.EMF_GetDebugTier]"
			}
			if = {
				limit = {
					is_patrician = yes
					is_merchant_republic = no
				}
				log = "---> ruler is a patrician but not a merchant republic"
			}
			if = {
				limit = {
					any_demesne_title = {
						OR = {
							controls_religion = yes
							temporary = yes
							mercenary = yes
							holy_order = yes
							clan = yes
							adventurer = yes
							rebel = yes
							has_flag = emf_revolt
						}
					}
				}
				log = "---> ruler has a special title:"
				any_demesne_title = {
					limit = {
						OR = {
							controls_religion = yes
							temporary = yes
							mercenary = yes
							holy_order = yes
							clan = yes
							adventurer = yes
							rebel = yes
							has_flag = emf_revolt
						}
					}
					log = "----> [This.EMF_GetDebugTitleNameWithDeJureLieges]"
					if = {
						limit = { controls_religion = yes }
						log = "-----> controls religion"
					}
					if = {
						limit = { temporary = yes }
						log = "-----> temporary"
					}
					if = {
						limit = { mercenary = yes }
						log = "-----> mercenary"
					}
					if = {
						limit = { holy_order = yes }
						log = "-----> holy order"
					}
					if = {
						limit = { clan = yes }
						log = "-----> clan"
					}
					if = {
						limit = { adventurer = yes }
						log = "-----> adventurer"
					}
					if = {
						limit = { rebel = yes }
						log = "-----> rebel"
					}
					if = {
						limit = { has_flag = emf_revolt }
						log = "-----> emf_revolt"
					}
				}
			}
			any_demesne_title = {
				limit = {
					tier = event_target:emf_cb_winner
					OR = {
						lower_tier_than = DUKE
						emf_is_title_type_special = yes
					}
				}
				log = "---> ruler holds same-tier title that shouldn't be destroyed: [This.EMF_GetDebugTitleNameWithDeJureLieges]"
			}
		}
	}

	option = {
		name = {
			text = emf_relsub.0.opt.submit_and_convert
			trigger = { NOT = { religion = event_target:emf_cb_winner } }
		}
		name = {
			text = emf_relsub.0.opt.submit_samerel
			trigger = { religion = event_target:emf_cb_winner }
		}
		trigger = {
			emf_relsub_can_submit = yes
		}
		ai_chance = {
			factor = 500
			# Absorb the base score of the pretend-to-convert option if it cannot be taken:
			additive_modifier = {
				value = 300
				NAND = {
					NOT = { religion = event_target:emf_cb_winner }
					# Doesn't already have same secret religion as we'd be converting to (CB winner's public religion)
					NOT = { secret_religion = { target = event_target:emf_cb_winner target_type = public } }
				}
			}
			modifier = {
				factor = 0
				any_realm_title = {
					tier = COUNT
					NOT = { de_jure_liege_or_above = event_target:emf_cb_title }
				}
			}
			modifier = { factor = 1.333 trait = humble }
			modifier = { factor = 0.75 trait = proud }
			# vs. secret conversion + submission, not vs. refusal
			modifier = { factor = 1.5 trait = honest }
		}
		hidden_tooltip = { log = "--> chooses to submit & convert truly" }
		emf_relsub_basic_submission_decision = yes
		if = {
			limit = { NOT = { religion = event_target:emf_cb_winner } }
			# Mark converted kings & their convertors for the sainthood mechanic
			if = {
				limit = {
					higher_real_tier_than = DUKE
					religion_group = pagan_group
					event_target:emf_cb_winner = { religion_group = christian }
				}
				set_flag = king_converted
				event_target:emf_cb_winner = { set_flag = converted_king }
			}
			if = {
				limit = { has_secret_religion = yes }
				clear_secret_religion = yes
			}
			if = {
				limit = { higher_tier_than = BARON }
				religion_authority = { modifier = ruler_mass_converted_from years = 20 }
			}
			religion = event_target:emf_cb_winner
			if = {
				limit = { higher_tier_than = BARON }
				event_target:emf_cb_winner = {
					show_scope_change = no
					religion_authority = { modifier = ruler_mass_converted_to years = 20 }
				}
			}
			emf_relsub_convert_court = yes
			if = {
				limit = { NOT = { trait = zealous } }
				random = {
					chance = 33
					emf_notify_add_zealous = yes
				}
			}
		}
		emf_relsub_ask_vassals_to_convert = yes
	}

	option = {
		name = emf_relsub.0.opt.submit_and_pretend_to_convert
		trigger = {
			emf_relsub_can_submit = yes
			NOT = { religion = event_target:emf_cb_winner }
			# Doesn't already have same secret religion as we'd be converting to (CB winner's public religion)
			NOT = { secret_religion = { target = event_target:emf_cb_winner target_type = public } }
		}
		ai_chance = {
			factor = 300
			modifier = {
				factor = 0
				any_realm_title = {
					tier = COUNT
					NOT = { de_jure_liege_or_above = event_target:emf_cb_title }
				}
			}
			modifier = { factor = 1.2  trait = humble }
			modifier = { factor = 0.85 trait = proud }
			# not vs. refusal, but non-secret submission
			modifier = { factor = 1.5 trait = deceitful }
		}
		hidden_tooltip = { log = "--> chooses to submit & pretend to convert" }
		emf_relsub_basic_submission_decision = yes
		# Since the converted king is pretending, don't mark him, but mark the convertor for the sainthood mechanic
		if = {
			limit = {
				higher_real_tier_than = DUKE
				religion_group = pagan_group
				event_target:emf_cb_winner = { religion_group = christian }
			}
			event_target:emf_cb_winner = { set_flag = converted_king }
		}
		if = {
			limit = { higher_tier_than = BARON }
			religion_authority = { modifier = ruler_mass_converted_from years = 20 }
		}
		religion = event_target:emf_cb_winner
		if = {
			limit = { higher_tier_than = BARON }
			event_target:emf_cb_winner = {
				show_scope_change = no
				religion_authority = { modifier = ruler_mass_converted_to years = 20 }
			}
		}
		emf_relsub_pretend_to_convert_court = yes
		trigger_switch = {
			on_trigger = has_secret_religion
			yes = { custom_tooltip = { text = RELSUB_CONTINUE_OBSERVING_SECRET_FAITH } }
			no = { set_secret_religion = event_target:emf_old_religion }
		}
		emf_relsub_ask_vassals_to_convert = yes
	}

	option = {
		name = {
			text = emf_relsub.0.opt.refuse_diffrel
			trigger = { NOT = { religion = event_target:emf_cb_winner } }
		}
		name = {
			text = emf_relsub.0.opt.refuse_samerelcul
			trigger = {
				religion = event_target:emf_cb_winner
				culture = event_target:emf_cb_winner
			}
		}
		name = {
			text = emf_relsub.0.opt.refuse_diffcul
			trigger = {
				religion = event_target:emf_cb_winner
				NOT = { culture = event_target:emf_cb_winner }
			}
		}
		ai_chance = {
			factor = 500
			modifier = {
				factor = 0
				OR = {
					religion = event_target:emf_cb_winner
					true_religion = { target_type = public target = event_target:emf_cb_winner }
				}
				NOT = {
					any_realm_title = {
						lower_tier_than = DUKE
						NOT = { ROOT = { family_palace = { title = PREVPREV } } }
						NOT = { de_jure_liege_or_above = event_target:emf_cb_title }
					}
				}
			}
			## Adds the opinion from 'ROOT' to 'who' multiplied by factor to the score.
			additive_opinion_modifier = {
				who = event_target:emf_cb_winner
				factor = -0.64 # 64 to -64, from least opinion to most opinion
			}
			# Adds the percentage of available troops from 'ROOT' to 'who' multiplied by factor to the score.
			additive_power_diff_modifier = {
				who = event_target:emf_cb_winner
				factor = 1 # 0 (when having zero power) to 100 (when equally powerful) to ... more (when more powerful)
			}
			# Effects of zeal, rationality, and greed. Note that these properties are on a scale of [-100,100] though we
			# don't concern ourselves with the extrama (stop checking past +/-75). The total range of score that could
			# possibly be awarded due to these properties is [-128, +112]. Combined with the above two additive modifiers,
			# the smallest base score we could get would be 308 (500 - 64 + 0 - 128), while the largest, assuming that the
			# ruler is never more powerful than the attacker but could be equally powerful, is 500 + 64 + 100 + 112 = 776.
			#
			# [-48,+64] from ai_zeal when true religion of ruler is NOT same as attacker:
			# Zealous characters certainly don't want to convert religion, but the less zealous they are, the less they care
			# about the conversion relative to keeping their titles.
			additive_modifier = { value =  4 ai_zeal > 0   NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }
			additive_modifier = { value =  4 ai_zeal > 5   NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }

			additive_modifier = { value =  4 ai_zeal > 20  NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }
			additive_modifier = { value =  4 ai_zeal > 25  NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }
			additive_modifier = { value =  4 ai_zeal > 30  NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }
			additive_modifier = { value =  4 ai_zeal > 35  NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }
			additive_modifier = { value =  4 ai_zeal > 40  NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }
			additive_modifier = { value =  4 ai_zeal > 45  NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }
			additive_modifier = { value =  4 ai_zeal > 50  NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }
			additive_modifier = { value =  4 ai_zeal > 55  NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }

			additive_modifier = { value =  4 ai_zeal > 65  NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }
			additive_modifier = { value =  4 ai_zeal > 70  NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }
			additive_modifier = { value =  4 ai_zeal > 75  NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }
			additive_modifier = { value = -3 ai_zeal < 0   NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }
			additive_modifier = { value = -3 ai_zeal < -5  NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }
			additive_modifier = { value = -3 ai_zeal < -10 NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }
			additive_modifier = { value = -3 ai_zeal < -15 NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }
			additive_modifier = { value = -3 ai_zeal < -20 NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }
			additive_modifier = { value = -3 ai_zeal < -25 NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }
			additive_modifier = { value = -3 ai_zeal < -30 NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }
			additive_modifier = { value = -3 ai_zeal < -35 NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }
			additive_modifier = { value = -3 ai_zeal < -40 NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }
			additive_modifier = { value = -3 ai_zeal < -45 NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }
			additive_modifier = { value = -3 ai_zeal < -50 NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }
			additive_modifier = { value = -3 ai_zeal < -55 NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }

			additive_modifier = { value = -3 ai_zeal < -75 NOT = { true_religion = { target_type = public target = event_target:emf_cb_winner } } }
			# [-48, +32] from ai_rationality:

			additive_modifier = { value = -3 ai_rationality > 5 }
			additive_modifier = { value = -3 ai_rationality > 10 }
			additive_modifier = { value = -3 ai_rationality > 15 }
			additive_modifier = { value = -3 ai_rationality > 20 }
			additive_modifier = { value = -3 ai_rationality > 25 }
			additive_modifier = { value = -3 ai_rationality > 30 }
			additive_modifier = { value = -3 ai_rationality > 35 }
			additive_modifier = { value = -3 ai_rationality > 40 }
			additive_modifier = { value = -3 ai_rationality > 45 }
			additive_modifier = { value = -3 ai_rationality > 50 }
			additive_modifier = { value = -3 ai_rationality > 55 }
			additive_modifier = { value = -3 ai_rationality > 60 }
			additive_modifier = { value = -3 ai_rationality > 65 }
			additive_modifier = { value = -3 ai_rationality > 70 }
			additive_modifier = { value = -3 ai_rationality > 75 }
			additive_modifier = { value =  2 ai_rationality < 0 }
			additive_modifier = { value =  2 ai_rationality < -5 }
			additive_modifier = { value =  2 ai_rationality < -10 }
			additive_modifier = { value =  2 ai_rationality < -15 }
			additive_modifier = { value =  2 ai_rationality < -20 }
			additive_modifier = { value =  2 ai_rationality < -25 }

			additive_modifier = { value =  2 ai_rationality < -55 }
			additive_modifier = { value =  2 ai_rationality < -60 }
			additive_modifier = { value =  2 ai_rationality < -65 }
			additive_modifier = { value =  2 ai_rationality < -70 }
			additive_modifier = { value =  2 ai_rationality < -75 }
			# [-32,+16] from ai_greed
			# Greedy characters want to hold on to their titles/vassals; less materialistic characters don't care so much.
			additive_modifier = { value = -2 ai_greed > 0 }
			additive_modifier = { value = -2 ai_greed > 5 }
			additive_modifier = { value = -2 ai_greed > 10 }
			additive_modifier = { value = -2 ai_greed > 15 }
			additive_modifier = { value = -2 ai_greed > 20 }
			additive_modifier = { value = -2 ai_greed > 25 }

			additive_modifier = { value = -2 ai_greed > 45 }
			additive_modifier = { value = -2 ai_greed > 50 }
			additive_modifier = { value = -2 ai_greed > 55 }
			additive_modifier = { value = -2 ai_greed > 60 }
			additive_modifier = { value = -2 ai_greed > 65 }
			additive_modifier = { value = -2 ai_greed > 70 }
			additive_modifier = { value = -2 ai_greed > 75 }
			additive_modifier = { value =  1 ai_greed < 0 }
			additive_modifier = { value =  1 ai_greed < -5 }
			additive_modifier = { value =  1 ai_greed < -10 }
			additive_modifier = { value =  1 ai_greed < -15 }
			additive_modifier = { value =  1 ai_greed < -20 }
			additive_modifier = { value =  1 ai_greed < -25 }
			additive_modifier = { value =  1 ai_greed < -30 }
			additive_modifier = { value =  1 ai_greed < -35 }
			additive_modifier = { value =  1 ai_greed < -40 }
			additive_modifier = { value =  1 ai_greed < -45 }
			additive_modifier = { value =  1 ai_greed < -50 }
			additive_modifier = { value =  1 ai_greed < -55 }
			additive_modifier = { value =  1 ai_greed < -60 }
			additive_modifier = { value =  1 ai_greed < -65 }
			additive_modifier = { value =  1 ai_greed < -70 }
			additive_modifier = { value =  1 ai_greed < -75 }
			# Dukes are a bit less likely to refuse the ultimatum than higher tiers
			modifier = {
				factor = 0.75
				lower_real_tier_than = KING
			}
			# Counts and barons are significantly less likely to refuse the ultimatum
			modifier = {
				factor = 0.625
				lower_real_tier_than = DUKE
			}
			# Pagans are more willing to submit & convert than those of organized religions
			modifier = {
				factor = 0.5
				religion_group = pagan_group
				is_reformed_religion = no
				hard_to_convert = no # Unless they're hard-to-convert pagans
			}
			# If these are your only demesne titles, you've got a lot more impetus to not refuse the ultimatum
			modifier = {
				factor = 0.5
				NOT = {
					any_demesne_title = {

# emf_relsub.40 -- delayed event for attempting to trigger a religious uprising in a province conquered through holy war
#
# basically just a proxy for emf_revolt.0, the on_rebel_revolt handler when there isn't already a religious uprising
# underway and, alternately, a proxy for emf_revolt.5 (reinforcement of ongoing religious uprising war).
#
# this allows us to supply our own intentional "revolt risk pulse" instead of relying entirely upon the game's
# on_rebel_revolt triggering behavior effectively.
province_event = {
	id = emf_relsub.40

	is_triggered_only = yes
	hide_window = yes

	trigger = {
		emf_relsub_has_unrest = yes
		owner = {
			top_liege = {
				ROOT = { emf_hostile_religion_to_PREV = yes }
				NOT = { is_parent_religion = ROOT }
			}
		}
	}

	immediate = {
		owner = {
			top_liege = {
				if = {
					limit = {
						any_war = {
							defender = { character = PREV }
							attacker = { religion = ROOT }
							using_cb = emf_revolt_religious
						}
					}
					# Try to reinforce existing same-religion religious uprising
					ROOT = { province_event = { id = emf_revolt.5 } }
					break = yes
				}
				# If we reach this point, there was not yet an uprising of our religion underway, so forward
				# to beginning of religious uprising chain with ROOT as the origin province:
				ROOT = {
					# Set temp. flag that overrides any prioritization of other revolt types
					# over a religious revolt in the province:
					set_flag = emf_revolt_tmp_triggered
					province_event = { id = emf_revolt.0 }
				}
			}
		}
	}
}
