# The Pope singles out a Kingdom with Free Investiture (unless he has already targetted one)
character_event = {
	id = 67000
	desc = EVTDESC67000
	picture = GFX_evt_icon
	border = GFX_event_normal_frame_religion

	only_playable = yes
	religion_group = christian
	min_age = 16
	only_men = yes
	prisoner = no
	only_capable = yes
	war = no
	
	trigger = {
		controls_religion = yes
		year = 1000
		primary_title = {
			OR = {
				title = k_papal_state
				title = d_cathar
				title = d_bogomilist
				title = d_manichean
				title = d_fraticelli
			}
		}
		religion_authority = 0.35
		NOT = { has_character_flag = pope_currently_demanding }
		any_independent_ruler = {
			religion = ROOT
			rightful_religious_head = ROOT
			NOT = { has_character_flag = pope_wants_our_law_changed }
			NOT = { has_character_flag = pope_respects_our_law }
			primary_title = {
				OR = {
					tier = KING
					tier = EMPEROR
				}
				has_law = investiture_law_1 
			}
		}
	}

	mean_time_to_happen = {
		months = 100 #once every 10 years
		modifier = {
			factor = 0.25
			trait = zealous
		}
	}

	option = {
		name = EVTOPTA67000
		ai_chance = {	factor = 100 }
		set_character_flag = pope_currently_demanding
		random_independent_ruler = {
			limit = {
				religion = ROOT
				rightful_religious_head = ROOT
				NOT = { has_character_flag = pope_wants_our_law_changed }
				NOT = { has_character_flag = pope_respects_our_law }
				primary_title = {
					tier = EMPEROR
					has_law = investiture_law_1
				}
			}
			character_event = { id = 67001 days = 14 tooltip = EVTTOOLTIP67001 }
			set_character_flag = pope_wants_our_law_changed
			break = yes
		}
		random_independent_ruler = {
			limit = {
				religion = ROOT
				rightful_religious_head = ROOT
				NOT = { has_character_flag = pope_wants_our_law_changed }
				NOT = { has_character_flag = pope_respects_our_law }
				primary_title = {
					tier = KING
					has_law = investiture_law_1
				}
			}
			character_event = { id = 67001 days = 14 tooltip = EVTTOOLTIP67001 }
			set_character_flag = pope_wants_our_law_changed
		}
	}

	option = {
		name = EVTOPTB67000
		ai_chance = { factor = 0 }
	}
}


# King/Emperor gets the papal emissary asking them to change laws!
character_event = {
	id = 67001
	desc = EVTDESC67001
	picture = GFX_evt_pope
	border = GFX_event_normal_frame_religion

	is_triggered_only = yes

	option = {
		name = EVTOPTA67001 # Fold to demand

		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				trait = cynical
			}
			modifier = {
				factor = 8
				trait = zealous
			}
		}

		FROM = { clr_character_flag = pope_currently_demanding }

		# To avoid duplicating the RA and temp. rel. head opinion bonus from passing investiture_law_0, set this
		# special, temporary character flag which is cleared automatically by the law code. We would use
		# emf_no_law_penalties, but we want to be able to control the reward/malus for investiture law change
		# independently of its cooldown.
		set_character_flag = emf_forced_investiture_head_tmp
		primary_title = {
			tooltip = { revoke_law = investiture_law_1 }
			add_law = investiture_law_0
		}
		piety = 50
		prestige = -200
		religion_authority = {
			modifier = ruler_bowed_to_pope
			years = 10
		}
		clr_character_flag = pope_wants_our_law_changed
		FROM = {
			opinion = {
				modifier = pope_invest_approve
				who = ROOT
				years = 5
			}
		}
		hidden_tooltip = {
			FROM = { character_event = { id = 67002 } }
		}
	}
	option = {
		name = EVTOPTB67001 # Refuse
		ai_chance = {
			factor = 1
			modifier = {
				factor = 1.5
				trait = arbitrary
			}
			modifier = {
				factor = 1.5
				trait = cynical
			}
			modifier = {
				factor = 1.5
				trait = hedonist
			}
		}
		piety = -50
		if = {
			limit = { emf_can_religion_excommunicate = yes }
			excommunicate = yes
		}
		religion_authority = {
			modifier = ruler_defied_pope
			years = 10
		}
		set_character_flag = pope_we_refuse_law_changed
		FROM = {
			opinion = {
				modifier = pope_invest_dislike
				who = ROOT
				years = 5
			}
		}
		hidden_tooltip = {
			FROM = {  character_event = { id = 67003 } }
		}
	}
	option = {
		name = EVTOPTC67001 # Give gold to Pope
		trigger = {
			diplomacy = 12			
		}
		ai_chance = {
			factor = 5
			modifier = {
				factor = 3
				trait = cynical
			}
			modifier = {
				factor = 0
				trait = zealous
			}
			modifier = {
				factor = 0.1
				NOT = { wealth = 200 }
			}
		}
		wealth = -200
		FROM = { clr_character_flag = pope_currently_demanding }
		set_character_flag = pope_respects_our_law
		clr_character_flag = pope_wants_our_law_changed
		hidden_tooltip = {
			FROM = { character_event = { id = 67004  } }
		}
	}
	option = {
		name = EVTOPTD67001 # Impress with Learning
		tooltip_info = learning
		trigger = {
			learning = 12
		}
		ai_chance = {
			factor = 100
		}
		piety = 10
		FROM = { clr_character_flag = pope_currently_demanding }
		set_character_flag = pope_respects_our_law
		clr_character_flag = pope_wants_our_law_changed
		hidden_tooltip = {
			FROM = {  character_event = { id = 67005 } }
		}
	}
}
