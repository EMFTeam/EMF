# [on_birth] Child might be a reincarnation
character_event = {
	id = RoI.10000
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		has_dlc = "Rajas of India"
		NOT = {
			has_game_rule = {
				name = supernatural_events
				value = off
			}
		}
		OR = {
			religion_group = indian_group
			religion = druze
			religion = yazidi
			religion = manichean
		}
		OR = {
			AND = {
				is_father_real_father = yes
				father = {
					higher_tier_than = BARON
					is_reincarnated = no
					OR = {
						mother_even_if_dead = { is_alive = no }
						mother_even_if_dead = { mother_even_if_dead = { is_alive = no } }
						father_even_if_dead = { is_alive = no }
						father_even_if_dead = { father_even_if_dead = { is_alive = no } }
					}
					NOT = { any_sibling = { is_reincarnated = yes } }
				}
			}
			mother = {
				higher_tier_than = BARON
				is_reincarnated = no
				OR = {
					mother_even_if_dead = { is_alive = no }
					mother_even_if_dead = { mother_even_if_dead = { is_alive = no } }
					father_even_if_dead = { is_alive = no }
					father_even_if_dead = { father_even_if_dead = { is_alive = no } }
				}
				NOT = { any_sibling = { is_reincarnated = yes } }
			}
		}
		NOT = { any_sibling = { is_reincarnated = yes } }
	}
	
	immediate = {
		if = {
			limit = {
				is_father_real_father = yes
				father = {
					religion = ROOT
					higher_tier_than = BARON
				}
			}
			father = {
				father_even_if_dead = {
					save_event_target_as = paternal_grandfather
					father_even_if_dead = { save_event_target_as = paternal_great_grandfather }
					mother_even_if_dead = { save_event_target_as = paternal_great_grandmother2 }
				}
				mother_even_if_dead = {
					save_event_target_as = paternal_grandmother
					father_even_if_dead = { save_event_target_as = paternal_great_grandfather2 }
					mother_even_if_dead = { save_event_target_as = paternal_great_grandmother }
				}
			}
		}
		if = {
			limit = {
				mother = {
					religion = ROOT
					higher_tier_than = BARON
				}
			}
			mother = {
				father_even_if_dead = {
					save_event_target_as = maternal_grandfather
					father_even_if_dead = { save_event_target_as = maternal_great_grandfather }
					mother_even_if_dead = { save_event_target_as = maternal_great_grandmother2 }
				}
				mother_even_if_dead = {
					save_event_target_as = maternal_grandmother
					mother_even_if_dead = { save_event_target_as = maternal_great_grandmother }
					father_even_if_dead = { save_event_target_as = maternal_great_grandfather2 }
				}
			}
		}
		
		if = {
			limit = { is_female = no }
			random_list = {

			}
		}
	}

			limit = { 
				religion = ROOT
				any_province_holding = {
					NOT = { holding_type = nomad }
				}
			}
			province_event = { id = RoI.3521 }
		}
	}
}

# Province bounce event
province_event = {
	id = RoI.3521

	hide_window = yes
	
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = {
				any_neighbor_province = {
					religion_group = indian_group
					NOT = { religion = ROOT }
					religion_authority = 0.7
				}
			}
			random_neighbor_province = {
				limit = {
					religion_group = indian_group
					NOT = { religion = ROOT }
					religion_authority = 0.7
				}
				ROOT = {
					religion = PREV
				}
			}
			FROM = { character_event = { id = RoI.3525 } }
			break = yes
		}
		if = {
			limit = {
				religion = hindu
				NOT = {
					any_neighbor_province = {
						religion_group = indian_group
						NOT = { religion = ROOT }
						religion_authority = 0.7
					}
				}
			}
			random_list = {
				10 = { religion = buddhist }
				10 = { religion = jain }
			}
			FROM = { character_event = { id = RoI.3525 } }
			break = yes
		}
		if = {
			limit = {
				religion = buddhist
				NOT = {
					any_neighbor_province = {
						religion_group = indian_group
						NOT = { religion = ROOT }
						religion_authority = 0.7
					}
				}
			}
			random_list = {
				10 = { religion = hindu }
				10 = { religion = jain }
			}
			FROM = { character_event = { id = RoI.3525 } }
			break = yes
		}
		if = {
			limit = {
				religion = jain
				NOT = {
					any_neighbor_province = {
						religion_group = indian_group

				10 = { religion = buddhist }

			FROM = { character_event = { id = RoI.3525 } }
		}
	}
}
