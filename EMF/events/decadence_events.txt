### Raises Decadence

# Drunkard
character_event = {
	id = 91200
	desc = "EVTDESC91200"
	picture = "GFX_evt_drunk"

	min_age = 16
	capable_only = yes
	prisoner = no
	only_playable = yes
	religion_group = muslim

	trigger = {
		trait = drunkard
		NOT = { has_character_flag = decadence_drunkard }
	}

	mean_time_to_happen = {
		months = 200
		modifier = {
			factor = 2.0
			has_character_flag = decadence_change
			NOT = {
				had_character_flag = {
					flag = decadence_change
					days = 1825
				}
			}
		}
		modifier = {
			factor = 0.75
			tier = emperor
		}
		modifier = {
			factor = 0.9
			tier = king
		}
		modifier = {
			factor = 2.0
			tier = count
		}
	}

	option = {
		name = "EVTOPTA91200"
		piety = -50
		pf_tradition_minus2_effect = yes
		set_character_flag = decadence_drunkard
		set_character_flag = decadence_change
	}
}
