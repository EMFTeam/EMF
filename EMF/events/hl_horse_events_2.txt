# Other horse bit my horse
character_event = {
	id = HL.6510
	hide_window = yes
	is_triggered_only = yes

	only_playable = yes

	trigger = {
		has_character_modifier = nomad_warhorse
		is_incapable = no
		NOT = { is_inaccessible_trigger = yes }
		OR = {
		    any_vassal = {
			    is_adult = yes
			    prisoner = no
			    is_incapable = no
		    	ai = yes
		    }
		    any_courtier = {
			    is_adult = yes
			    prisoner = no
			    is_incapable = no
			    ai = yes
		    }
		}
	}

	weight_multiplier = {
		days = 1
	}

	immediate = {
		if = {
			limit = {
				any_vassal = {
					is_adult = yes
				    prisoner = no
				    is_incapable = no
			    	ai = yes
				}
			}
			any_vassal = {
				limit = {
					is_adult = yes
				    prisoner = no
				    is_incapable = no
			    	ai = yes
				}
				score_value = { #Prioritizes selecting an interesting courtier...
		            value = 10
		            interesting_courtier_selection_score = yes
		        }
				character_event = { id = HL.6511 }
			}
		}
		else = {
			any_courtier = {
				limit = {
					is_adult = yes
					prisoner = no
					is_incapable = no
					ai = yes
				}
				score_value = { #Prioritizes selecting an interesting courtier...
					value = 10
					interesting_courtier_selection_score = yes
				}
				character_event = { id = HL.6511 }
			}
		}
	}
}

# Flag and modifier clean-up
character_event = {
	id = HL.6999
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		remove_character_modifier = nomad_warhorse
		clr_character_flag = nomad_warhorse_name_lightning
		clr_character_flag = nomad_warhorse_name_dragon
		clr_character_flag = nomad_warhorse_name_sky
		clr_character_flag = nomad_warhorse_name_arrow
		clear_delayed_event = { id = HL.6500 }

		if = {
			limit = {
				NOT = { has_character_flag = nomad_warhorse_ignore_grief }
			}
			random = {
				chance = 10
				character_event = { id = HL.6990 days = 150 }
			}
		}
		clr_character_flag = nomad_warhorse_ignore_grief
	}
}
