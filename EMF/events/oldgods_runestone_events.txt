	id = TOG.562
	title = EVTNAME_TOG_550
	desc = {
		trigger = {
			mother_even_if_dead = { always = yes }
			father_even_if_dead = { always = yes }
		}
		text = EVTDESC_TOG_562
	}
	desc = {
		trigger = {
			NOT = {
				AND = {
					mother_even_if_dead = { always = yes }
					father_even_if_dead = { always = yes }
				}
			}
		}
		text = EVTDESC_TOG_562B

	border = GFX_event_narrative_frame_diplomacy

	hide_from = yes

	option = {

			name = raised_runestone
			duration = 730
		}

			chronicle = {
				entry = CHRONICLE_RUNESTONE_GLUTTONOUS

			}
		}

# Runestones can no longer be commissioned
character_event = {
	id = TOG.580
	desc = EVTDESC_TOG_580
	picture = GFX_evt_runestone_raising_oldgods
	border = GFX_event_normal_frame_diplomacy

	ai = no
	culture_group = north_germanic

	trigger = {
		year = 1150
		NOT = { has_character_flag = runestones_halted }
		NOT = {
			has_alternate_start_parameter = { key = religion_names value = random }
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = EVTOPTA_TOG_580
		custom_tooltip = { text = no_more_runestones }
		set_character_flag = runestones_halted
	}
}
