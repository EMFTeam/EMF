	hide_window = yes

	culture = mongol
	
	trigger = {
		dynasty = 11100

		OR = {
			has_character_flag = line_of_jochi
			has_character_flag = line_of_chagatai

		any_child = {
			limit = {

			}

				limit = { ROOT = { has_character_flag = line_of_chagatai } }
