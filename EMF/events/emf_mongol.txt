# Apply flags at game start
character_event = {
	id = emf_mongol.100
	
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		has_global_flag = temujin_born
		NOT = { year = 1242 }
	}
	
	immediate = {
		# Kill Temujin and his dynasty if they exist before the invasion event
		random_character = {
			limit = {
				has_character_flag = is_temujin
				is_ruler = no
			}
			log = "emf_mongol.100: unlanded Temujin found - removing him and his dynasty in favor of spawned version"
			any_dynasty_member_even_if_dead = {
				dynasty = father_bastard # neither 'none' nor '0' work here, so just give them a random one
				if = {
					limit = { is_alive = yes }
					death = { death_reason = death_missing }
				}
			}
			dynasty = father_bastard # neither 'none' nor '0' work here, so just give them a random one
			death = { death_reason = death_missing }
		}
		
		# Otherwise, the Mongol Empire already exists
		# So mark the sons of Temujin appropriately
		any_character = {
			limit = {
				culture = mongol
				is_female = no
			}
			if = {
				limit = { dynasty = 7379 }
				set_global_flag = jochi_dynasty_created
			}
			if = {
				limit = { dynasty = 7380 }
				set_global_flag = chagatai_dynasty_created
			}
			if = {
				limit = { dynasty = 1000054550 }
				set_global_flag = ogodei_dynasty_created
			}
			if = {
				limit = { dynasty = 101916 }
				set_global_flag = tolui_dynasty_created
			}
			if = {
				limit = {
					dynasty = 11100 # Genghis Khan's dynasty
					OR = {
						character = 125502
						father_even_if_dead = { character = 125502 }
						father_even_if_dead = { father_even_if_dead = { character = 125502 } }
					}
				}
				set_character_flag = line_of_jochi
			}
			if = {
				limit = {
					dynasty = 11100 # Genghis Khan's dynasty
					OR = {
						character = 93062
						father_even_if_dead = { character = 93062 }
						father_even_if_dead = { father_even_if_dead = { character = 93062 } }
					}
				}
				set_character_flag = line_of_chagatai
			}
			if = {
				limit = {
					dynasty = 11100 # Genghis Khan's dynasty
					OR = {
						character = 172005
						father_even_if_dead = { character = 172005 }
						father_even_if_dead = { father_even_if_dead = { character = 172005 } }
					}
				}
				set_character_flag = line_of_ogodei
			}
			if = {
				limit = {
					dynasty = 11100 # Genghis Khan's dynasty
					OR = {
						character = 125520
						father_even_if_dead = { character = 125520 }
						father_even_if_dead = { father_even_if_dead = { character = 125520 } }
					}
				}
				set_character_flag = line_of_tolui
			}
		}
	}
}

# Establish contiguous provinces in mongol realms
character_event = {
	id = emf_mongol.105
	
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		any_realm_province = { clr_province_flag = emf_mongol_connected }
		
		capital_scope = { # Seed the capital
			set_province_flag = emf_mongol_connected
		}
		
		# First figure out which provinces in the home region are connected
		while = {
			limit = {
				any_realm_province = {
					owner_under_ROOT = yes
					NOT = { has_province_flag = emf_mongol_connected }
					OR = {
						AND = {
							ROOT = { has_landed_title = e_golden_horde }
							emf_golden_horde_province = yes
						}
						AND = {
							ROOT = { has_landed_title = e_il-khanate }
							emf_il_khanate_province = yes
						}
						AND = {
							ROOT = { has_landed_title = e_chagatai }
							region = emf_region_mongol_east
						}
					}
					any_neighbor_province = {
						owner_under_ROOT = yes
						has_province_flag = emf_mongol_connected
					}
				}
			}
			any_realm_province = {
				limit = {
					owner_under_ROOT = yes
					NOT = { has_province_flag = emf_mongol_connected }
					OR = {
						AND = {
							ROOT = { has_landed_title = e_golden_horde }
							emf_golden_horde_province = yes
						}
						AND = {
							ROOT = { has_landed_title = e_il-khanate }
							emf_il_khanate_province = yes
						}
						AND = {
							ROOT = { has_landed_title = e_chagatai }
							region = emf_region_mongol_east
						}
					}
					any_neighbor_province = {
						owner_under_ROOT = yes
						has_province_flag = emf_mongol_connected
					}
				}
				set_province_flag = emf_mongol_connected
			}
		}
		
		# Then allow the flags to extend up to two provinces outside the home region
		any_realm_province = {
			limit = {
				owner_under_ROOT = yes
				NOT = { has_province_flag = emf_mongol_connected }
				any_neighbor_province = {
					owner_under_ROOT = yes
					has_province_flag = emf_mongol_connected
				}
			}
			set_province_flag = emf_mongol_connected
		}
		any_realm_province = {
			limit = {
				owner_under_ROOT = yes
				NOT = { has_province_flag = emf_mongol_connected }
				any_neighbor_province = {
					owner_under_ROOT = yes
					has_province_flag = emf_mongol_connected
				}
			}
			set_province_flag = emf_mongol_connected
		}
	}
}

	desc = EVTDESC_emf_mongol_200

	title = {
		text = EVTNAME_emf_mongol_200
		trigger = { ROOT = { NOT = { has_character_modifier = greatest_of_khans } } }
	}
	title = {
		text = EVTNAME_emf_mongol_200_genghis
		trigger = { ROOT = { has_character_modifier = greatest_of_khans } }
	}
	
	is_triggered_only = yes
	major = yes
	
	ai = yes
	culture = mongol
	only_playable = yes
	only_independent = yes
	
	trigger = {
		has_landed_title = e_mongol_empire
		num_of_count_titles_in_realm = 75
		
		# Has at least twenty provinces for Golden Horde
		any_realm_province = {
			count = 20
			emf_golden_horde_province = yes
		}
		
		# Has at enough holdings in Il-Khanate to create empire
		any_realm_title = {
			count = 120
			tier = BARON
			location = { emf_il_khanate_province = yes }
		}
		
		# Has at least five provinces for Chagatai
		any_realm_province = {
			count = 5
			region = emf_region_mongol_east
		}
		
		# Has at least three valid heirs
		any_dynasty_member = {
			count = 3
			emf_valid_mongol_heir_for_ROOT = yes
		}
	}

	immediate = {
		hidden_tooltip = {
			# First attempt to give titles to historical people
			# Starting with Jochi & descendants
			if = {
				limit = { has_character_flag = is_temujin }

					limit = {
						has_character_flag = line_of_jochi
						emf_valid_mongol_heir_for_ROOT = yes

							any_dynasty_member = {
								has_character_flag = line_of_jochi
								is_older_than = PREV
								emf_valid_mongol_heir_for_ROOT = yes
							}
						}
					}
					save_event_target_as = first_mongol_heir
				}
				random_dynasty_member = {
					limit = {
						has_character_flag = line_of_tolui
						emf_valid_mongol_heir_for_ROOT = yes
						NOT = {
							any_dynasty_member = {
								has_character_flag = line_of_tolui
								is_older_than = PREV
								emf_valid_mongol_heir_for_ROOT = yes
							}
						}
					}
					save_event_target_as = second_mongol_heir
				}
				random_dynasty_member = {
					limit = {
						has_character_flag = line_of_chagatai
						emf_valid_mongol_heir_for_ROOT = yes
						NOT = {
							any_dynasty_member = {
								has_character_flag = line_of_chagatai
								is_older_than = PREV
								emf_valid_mongol_heir_for_ROOT = yes
							}
						}
					}
					save_event_target_as = third_mongol_heir
				}
			}
			if = {
				limit = { NOT = { event_target:first_mongol_heir = { is_alive = yes } } }
				current_heir = {
					if = {
						limit = {
							NOR = {
								trait = craven
								trait = content
								trait = humble
								is_infertile_trigger = yes
								is_dumb_trigger = yes
								is_weak_trigger = yes
								has_severe_disability_trigger = yes
							}
							NOR = {
								character = event_target:second_mongol_heir
								character = event_target:third_mongol_heir
							}
						}
						save_event_target_as = first_mongol_heir
					}
				}
			}
			if = {
				limit = { NOT = { event_target:first_mongol_heir = { is_alive = yes } } }
				random_dynasty_member = {
					limit = {
						NOT = { event_target:first_mongol_heir = { is_alive = yes } }

	culture = mongol
	only_capable = yes
	min_age = 16
	war = no
	ai = yes

		emf_mongol_invasion_mission = yes

		culture = mongol
		NOT = { trait = incapable }

		has_regent = no

		custom_tooltip = {
			text = MONGOLS_HAVE_SPARE_VASSAL_LIMIT
			hidden_tooltip = {
				NOT = { over_vassal_limit = 0 }
			}
		}

		# Don't consider invasion until it's been awhile since last successful invasion ended
		custom_tooltip = {
			text = MONGOLS_HAVE_NO_INVASION_COOLDOWN
			hidden_tooltip = {
				OR = {
					NOT = { has_character_flag = emf_mongol_invasion_war_won }
					NOT = { # No cooldown during first 30yr
						had_global_flag = {
							flag = mongol_empire_arrival
							days = 10950
						}
					}
					AND = {
						NOT = { # Minimal cooldown during first 30-60yrs
							had_global_flag = {
								flag = mongol_empire_arrival
								days = 21900
							}
						}
						had_character_flag = {
							flag = emf_mongol_invasion_war_won
							days = 180
						}
					}
					had_character_flag = { # Larger cooldown thereafter
						flag = emf_mongol_invasion_war_won
						days = 730
					}
				}
			}
		}

		# Only if we still have our initial event troops
		# IMPORTANT: we do need to start with such earmarked troops, or else this will never fire, the mongols will
		# never be "defeated," and their all-CBs block will never be lifted (i.e., they'll just sit there forever)
		custom_tooltip = {
			text = MONGOLS_HAVE_INVASION_EVENT_TROOPS
			hidden_tooltip = {
				OR = {
					has_earmarked_regiments = horde_invasion
					has_earmarked_regiments = start_troops
				}
			}
		}

		# Minimum gold to fire a scripted invasion
		OR = {
			wealth = 50
			custom_tooltip = {
				text = MONGOLS_ARRIVED_LESS_THAN_40_YEARS_AGO
				hidden_tooltip = {
					NOT = {
						had_global_flag = {
							flag = mongol_empire_arrival
							days = 14600 # 40yr
						}
					}
				}
			}
		}
	}

		days = 0
	}

	immediate = {
		# If they're particularly low on cash, give them a little, but only support them
		# in this way for awhile after their arrival (duration enforced by event trigger).
		while = {
			limit = { NOT = { wealth = 50 } }
			wealth = 50
		}
		
		if = {
			limit = { is_landed = no }
			# Something screwed-up happened to our initial invasion (but it wasn't an actual loss), so retry

			if = {
				limit = { has_global_flag = SWMH }

				limit = { NOT = { has_global_flag = SWMH } }

			}
			character_event = { id = emf_mongol.1000 } # Declare war
			break = yes
		}

		# Ensure we have the nomadic lucky ruler (+mongols) building
		capital_holding = {
			if = {
				limit = {
					NOR = {
						has_building = no_lucky_1
						year = 1300
					}
				}
				add_building = no_lucky_1
			}
		}
		
		# Pick a random available target from the core conquest region...
		# Prefer those in the same de jure kingdom as our last launched war(s) for troop locality

			limit = {
				event_target:emf_mongol_invasion_last_king_title = { tier = THIS }

					region = emf_region_mongol_core
					emf_mongol_invasion_valid_target_province = yes
				}
			}
			random_neighbor_province = {
				limit = {
					kingdom = { title = event_target:emf_mongol_invasion_last_king_title }
					region = emf_region_mongol_core
					emf_mongol_invasion_valid_target_province = yes
				}
				kingdom = {
					save_event_target_as = emf_king_title
					break = yes
				}
			}
		}
		random_realm_province = {
			limit = {
				has_province_flag = emf_capital_connected
				any_neighbor_province = {
					region = emf_region_mongol_core
					emf_mongol_invasion_valid_target_province = yes
				}
			}

					region = emf_region_mongol_core
					emf_mongol_invasion_valid_target_province = yes

					save_event_target_as = emf_king_title
					break = yes
				}

		}

		random_realm_province = {
			limit = {
				event_target:emf_mongol_invasion_last_king_title = { tier = THIS }
				has_province_flag = emf_capital_connected
				any_neighbor_province = {
					kingdom = { title = event_target:emf_mongol_invasion_last_king_title }
					region = emf_region_mongol_extended
					emf_mongol_invasion_valid_target_province = yes

				}

					save_event_target_as = emf_king_title
					break = yes
				}
			}
		}
	}

			event_target:emf_king_title = { always = yes }
		}

		# at once (allows the Mongols to take advantage of locality of their event troops, to
		# some degree).
		any_playable_ruler = {
			limit = {
				emf_mongol_invasion_valid_target = yes
