# emf_nomad.1400 [400 - Zyriane]
province_event = {
	id = emf_nomad.1400

	is_triggered_only = yes
	hide_window = yes

	trigger = {
		owner = { is_nomadic = yes }
		NOT = { any_province_holding = { NOT = { holding_type = nomad } } }
		has_empty_holding = yes
	}

	immediate = {
		if = {
			limit = { culture = samoyed religion = finnish_pagan }
			break = yes # Not necessary to build a settlement
		}
		build_holding = { type = tribal }
		culture  = samoyed
		religion = finnish_pagan
		emf_nomad_antinomad_tribal_effect = yes
		log = "emf_antinomad(400, '[Root.GetName]') => tribal / samoyed / finnish_pagan"
	}
}
