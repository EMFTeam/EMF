### Marshal suggests candidate for commander position (if corrupt, may suggest moderately competent friends instead of the best)
character_event = {
	id = ZE.4000
	desc = EVTDESC_ZE_4000
	picture = GFX_evt_melee
	border = GFX_event_normal_frame_war

	is_triggered_only = yes

	ai = no
	only_capable = yes

	trigger = {
		has_dlc = "Zeus"
		is_nomadic = no
		job_marshal = {
			ai = yes
			NOT = { is_inaccessible_trigger = yes }
		}
		OR = {
			NOT = { ROOT = { has_law = succ_byzantine_elective } }  #Either not Imperial...
			NOT = { #...Or there are no valid vassal candidates that can be commanders.
				any_vassal = {
					higher_real_tier_than = BARON
					NOT = {
						primary_title = { temporary = yes }
					}
					OR = {
						AND = {
							OR = {
								male_can_hold_minor_title_trigger = yes
								female_can_hold_minor_commander_title_trigger = yes
							}
							OR = {
								NOT = { religion = hindu }
								trait = kshatriya
							}
							is_adult = yes
						}
						has_character_flag = special_marshal
					}

					prisoner = no
					is_incapable = no
					is_inaccessible_trigger = no
					NOT = { has_character_flag = guru }
					NOT = { has_character_modifier = refusing_to_lead }
					NOR = {
						has_job_title = job_chancellor
						has_job_title = job_marshal
						has_job_title = job_treasurer
						has_job_title = job_spymaster
						has_job_title = job_spiritual
						has_minor_title = title_commander
					}
				}
			}
		}
		OR = {
			AND = {
				tier = COUNT
				NOT = {
					any_courtier_or_vassal = {
						count = 2
						has_minor_title = title_commander
					}
				}
				any_courtier_or_vassal = {
					ai = yes
					martial = 9
					NOT = { has_minor_title = title_commander }
					NOT = { has_job_title = job_marshal }
					can_hold_title = title_commander
				}
			}
			AND = {
				tier = DUKE
				NOT = {
					any_courtier_or_vassal = {
						count = 4
						has_minor_title = title_commander
					}
				}
				any_courtier_or_vassal = {
					ai = yes
					martial = 9
					NOT = { has_minor_title = title_commander }
					NOT = { has_job_title = job_marshal }
					can_hold_title = title_commander
				}
			}
			AND = {
				tier = KING
				NOT = {
					any_courtier_or_vassal = {
						count = 6
						has_minor_title = title_commander
					}
				}
				any_courtier_or_vassal = {
					ai = yes
					martial = 9
					NOT = { has_minor_title = title_commander }
					NOT = { has_job_title = job_marshal }
					can_hold_title = title_commander
				}
			}
			AND = {
				tier = EMPEROR
				NOT = {
					any_courtier_or_vassal = {
						count = 8
						has_minor_title = title_commander
					}
				}
				any_courtier_or_vassal = {
					ai = yes
					martial = 9
					NOT = { has_minor_title = title_commander }
					NOT = { has_job_title = job_marshal }
					can_hold_title = title_commander
				}
			}
		}
	}

	weight_multiplier = {
		days = 1

		modifier = {
			factor = 2
			job_marshal = {
				OR = {
					trait = arbitrary
					trait = deceitful
				}
				any_friend = { ai = yes }
			}
		}
	}

	immediate = {
		if = {
			limit = {
				job_marshal = {
					OR = {
						trait = arbitrary
						trait = deceitful
					}
					any_friend = { ai = yes }
				}
				any_courtier_or_vassal = {
					ai = yes
					ROOT = { job_marshal = { is_friend = PREVPREV } }
					NOT = { has_minor_title = title_commander }
					NOT = { has_job_title = job_marshal }
					can_hold_title = title_commander
					martial = 9
				}
			}
			job_marshal = { set_character_flag = marshal_interested_commander_suggestion }
			random_courtier_or_vassal = {
				limit = {
					ai = yes
					ROOT = { job_marshal = { is_friend = PREVPREV } }
					NOT = { has_minor_title = title_commander }
					NOT = { has_job_title = job_marshal }
					can_hold_title = title_commander
					martial = 9
				}
				save_event_target_as = recommended_for_marshal_ze_4000
				break = yes
			}
		}
		if = {
			limit = {
				job_marshal = {
					OR = {
						trait = arbitrary
						trait = deceitful
					}
				}
				any_courtier_or_vassal = {
					ai = yes
					ROOT = { job_marshal = { dynasty = PREVPREV } }
					NOT = { has_minor_title = title_commander }
					NOT = { has_job_title = job_marshal }
					can_hold_title = title_commander
					martial = 9
				}
			}
			job_marshal = { set_character_flag = marshal_interested_commander_suggestion }
			random_courtier_or_vassal = {
				limit = {
					ai = yes
					ROOT = { job_marshal = { dynasty = PREVPREV } }
					NOT = { has_minor_title = title_commander }
					NOT = { has_job_title = job_marshal }
					can_hold_title = title_commander
					martial = 9
				}
				save_event_target_as = recommended_for_marshal_ze_4000
				break = yes
			}
		}
		random_courtier_or_vassal = {
			limit = {
				ai = yes
				NOT = { has_minor_title = title_commander }
				NOT = { has_job_title = job_marshal }
				can_hold_title = title_commander
				martial = 9
			}
			preferred_limit = {
				martial = 19
			}
			preferred_limit = {
				martial = 15
			}
			preferred_limit = {
				martial = 12
			}
			save_event_target_as = recommended_for_marshal_ze_4000
		}
	}

	option = {
		name = EVTOPTA_ZE_4000 # Yes
		event_target:recommended_for_marshal_ze_4000 = {
			give_minor_title = title_commander
		}
		job_marshal = { clr_character_flag = marshal_interested_commander_suggestion }
	}
	option = {
		name = EVTOPTB_ZE_4000 # No
		job_marshal = {
			prestige = -20
			if = {
				limit = { has_character_flag = marshal_interested_commander_suggestion }
				opinion = {
					who = ROOT
					modifier = opinion_very_disappointed
					years = 3
				}
				clr_character_flag = marshal_interested_commander_suggestion
			}
		}
	}
}
