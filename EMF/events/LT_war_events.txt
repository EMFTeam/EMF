# Henry I dies, this is an **on_death** fired event
character_event = {
	id = LT.21100

	is_triggered_only = yes
	hide_window = yes

	trigger ={
		character = 202992
		has_landed_title = k_england
		NOR = {
			primary_title = {
				succ_law_title = {
					has_law = agnatic_succession
				}
			}
			any_child = {
				is_heir = yes
				is_female = no
			}
		}
		c_202991 = { is_alive = yes }
		c_205851 = {
			is_alive = yes
			is_incapable = no
			prisoner = no
		}
	}

	immediate = {
		save_event_target_as = dead_king_henry
		c_202991 = {
			add_claim = k_england
			save_event_target_as = matilda
		}
		c_205851 = {
			hidden_tooltip = { narrative_event = { id = LT.21110 days = 1 } }
		}
	}
}

# Stephen must decide either to support Matilda or usurp the Throne
narrative_event = {
	id = LT.21110
	title = EVTTITLE_LT_21110
	desc = EVTDESC_LT_21110
	picture = GFX_evt_rival_rulers
	border = GFX_event_narrative_frame_war

	is_triggered_only = yes

	trigger = {
		character = 205851
	}

	option = {
		name = EVTOPTA_LT_21110 #Support Matilda
		ai_chance = { factor = 20 }
		trigger = {
			event_target:matilda = {
				is_alive = yes
				is_incapable = no
				prisoner = no
			}
		}
		k_england = {
			set_title_flag = stephen_supported_matilda
		}

		event_target:matilda = {
			narrative_event = { id = LT.21120 days = 1 }
		}
	}

	option = {
		name = EVTOPTB_LT_21110 #Usurp the throne
		ai_chance = { factor = 80 }
		d_normandy = {
			if = {
				limit = { holder = event_target:matilda }
				usurp_title_plus_barony_if_unlanded_and_vassals = ROOT
			}
		}
		d_york = {
			if = {
				limit = { holder = event_target:matilda }
				usurp_title_plus_barony_if_unlanded_and_vassals = ROOT
			}
		}
		d_lancaster = {
			if = {
				limit = { holder = event_target:matilda }
				usurp_title_plus_barony_if_unlanded_and_vassals = ROOT
			}
		}
		vassalize_or_take_under_title = {
			title = k_england
			enemy = event_target:matilda
			#same_religion = yes # Only vassalize rulers of my religion
			#is_religious = yes
			type = host
		}
		k_england = {
 			usurp_title_plus_barony_if_unlanded_and_vassals = ROOT
		}
		c_middlesex = {
			ROOT = { capital = PREV }
		}
		c_cornwall = {
			if = {
				limit = { holder = event_target:matilda }
				usurp_title_plus_barony_if_unlanded_and_vassals = ROOT
			}
		}
		set_name = "Stephen"
		culture = "norman"

		hidden_effect = {
			event_target:matilda = {
				narrative_event = { id = LT.21130 days = 2 }
				if = {
					limit = {
						any_spouse = {
							OR = {
								is_landed = yes
								is_patrician = yes
							}
							is_incapable = no
						}
					}
					random_spouse = {
						limit = {
							OR = {
								is_landed = yes
								is_patrician = yes
							}
							is_incapable = no
						}
						preferred_limit = {
							is_main_spouse = yes
						}
						save_event_target_as = matilda_spouse
						narrative_event = { id = LT.21131 days = 2 }
					}
				}
				else = {
					spouse = { save_event_target_as = matilda_spouse }
				}
			}

			k_scotland = {
				holder_scope = {
					save_event_target_as = king_of_scotland
					#narrative_event = { id = LT.21132 days = 10 }
				}
			}

			any_player = {
				#narrative_event = { id = LT.21133 days = 10 }
			}
		}
	}
}
