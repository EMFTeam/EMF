character_event = {
	id = emf_feudalizer.10
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		any_demesne_title = {
			tier = BARON
			holding_type = tribal
		}
	}
	
	option = {
		name = OK
		
		# Convert tribal settlements, build holdings in tribal provinces
		character_event = { id = emf_feudalizer.11 }
		
		# Depending upon year, auto-build some basic bootstrap buildings in cities and temples
		character_event = { id = emf_feudalizer.12 }
		
		character_event = { id = emf_feudalizer.13 } # Grant all cities to new burghers [recursive]
		character_event = { id = emf_feudalizer.14 } # Grant all temples to new priests [recursive]
		
		# Queue-up proper levy refills for all newly-built settlements
		
		any_title_under = {
			limit = { tier = BARON }
			set_title_flag = emf_refill_levy
		}
		
		character_event = { id = emf_core.40 days = 3 } # Deferred levy refill for flagged subrealm
		
		# Set new government type for tribal
		emf_switch_to_feudal_gov_effect = yes
		
		# Initialize normal crown laws if they hold a crown title
		set_character_flag = emf_no_law_penalties
		
		any_demesne_title = {
			limit = { higher_tier_than = DUKE }
			if = {
				limit = {
					is_law_potential = crown_authority_0
					NOR = {
						has_law = crown_authority_0
						has_law = crown_authority_1
						has_law = crown_authority_2
						has_law = crown_authority_3
						has_law = crown_authority_4
					}
				}
				add_law = crown_authority_0
			}
			if = {
				limit = {
					is_law_potential = investiture_law_1
					NOR = {
						has_law = investiture_law_0
						has_law = investiture_law_1
					}
					NAND = {
						is_law_potential = investiture_law_0
						year = 1000
					}
				}
				add_law = investiture_law_1
			}
			if = {
				limit = {
					is_law_potential = investiture_law_0
					year = 1000
					NOR = {
						has_law = investiture_law_0
						has_law = investiture_law_1
					}
				}
				add_law = investiture_law_0
			}
		}
		
		clr_character_flag = emf_no_law_penalties
	}
}



character_event = {
	id = emf_feudalizer.11
	
	is_triggered_only = yes
	hide_window = yes
	
	option = {
		name = OK
		
		any_demesne_title = {
			limit = {
				tier = BARON
				holding_type = tribal
			}
			convert_to = CASTLE
			refill_holding_levy = yes
			if = {
				limit = {
					dejure_liege_title = {
						holder = ROOT
						location = {
							num_of_empty_holdings = 2
						}
						NOT = {
							any_direct_de_jure_vassal_title = {
								holding_type = city
							}
						}
					}
				}
				location = {
					build_holding = {
						type = city
						holder = ROOT
					}
				}
			}
			if = {
				limit = {
					dejure_liege_title = {
						holder = ROOT
						location = {
							num_of_empty_holdings = 2
						}
						NOT = {
							any_direct_de_jure_vassal_title = {
								holding_type = temple
							}
						}
					}
				}
				location = {
					build_holding = {
						type = temple
						holder = ROOT
					}
				}
			}
		}
	}
}

character_event = {
	id = emf_feudalizer.13
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		any_demesne_title = {
			tier = BARON
			holding_type = city
		}
	}
	
	option = {
		name = OK
		random_demesne_title = {
			limit = {
				tier = BARON
				holding_type = city
			}
			ROOT = {
				create_random_steward = {
					random_traits = yes
					dynasty = none
					female = no
				}
				new_character = {
					wealth = 300
					grant_title_no_opinion = PREVPREV # The barony
					set_government_type = republic_government
					set_defacto_liege = ROOT
				}
			}
		}
		character_event = { id = emf_feudalizer.13 }
	}
}


character_event = {
	id = emf_feudalizer.14
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		NOT = { religion_group = muslim }
		NOT = { religion = bogomilist }
		any_demesne_title = {
			tier = BARON
			holding_type = temple
		}
	}
	
	option = {
		name = OK
		random_demesne_title = {
			limit = {
				tier = BARON
				holding_type = temple
			}
			ROOT = {
				random_list = {
					50 = { # Male priest
						create_random_priest = {
							random_traits = yes
							dynasty = none
							female = no
						}
					}
					50 = { # Female priest
						modifier = {
							factor = 0
							NOT = { religion_allows_female_temple_holders = yes }
						}
						create_random_priest = {
							random_traits = yes
							dynasty = none
							female = yes
						}
					}
				}
				new_character = {
					wealth = 300
					grant_title_no_opinion = PREVPREV # The barony
					set_government_type = theocracy_government
					set_defacto_liege = ROOT
				}
			}
		}
		character_event = { id = emf_feudalizer.14 }
	}
}
