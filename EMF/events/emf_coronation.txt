# Initialization event
character_event = {
	id = emf_coronation.1100
	
	is_triggered_only = yes
	hide_window = yes
	
	immediate = {
		# Add Crowned/Crowned by Pope/Weak Basileus traits
		any_playable_ruler = {
			limit = {
				higher_real_tier_than = DUKE
				is_adult = yes
				is_feudal = yes
				has_horde_culture = no
				is_nomadic = no
				primary_title = { is_vice_royalty = no }
				OR = {
					emf_imperial_government_trigger = no
					real_tier = EMPEROR
				}
				NOR = {
					trait = crowned
					trait = uncrowned
					trait = crowned_by_pope
					trait = strong_basileus
					trait = weak_basileus
				}
			}
			if = {
				limit = { emf_uses_byzantine_coronation = yes }
				add_trait = weak_basileus
			}
			if = {
				limit = { emf_uses_hre_coronation = yes }
				add_trait = crowned_by_pope
			}
			if = {
				limit = {
					emf_uses_byzantine_coronation = no
					emf_uses_hre_coronation = no
				}
				add_trait = crowned
			}
		}
		
		# Add Uncrowned to everyone else
		any_playable_ruler = {
			limit = {
				higher_real_tier_than = DUKE
				is_feudal = yes
				has_horde_culture = no
				is_nomadic = no
				primary_title = { is_vice_royalty = no }
				OR = {
					emf_imperial_government_trigger = no
					real_tier = EMPEROR
				}
				NOR = {
					trait = crowned
					trait = uncrowned
					trait = crowned_by_pope
					trait = strong_basileus
					trait = weak_basileus
				}
			}
			add_trait = uncrowned
		}
		
		# Initialize Imperial Decay
		any_playable_ruler = {
			limit = { emf_uses_imperial_decay_trigger = yes }
			primary_title = { set_variable = { which = "imperial_decay_change" value = 0 } }
			character_event = { id = emf_imperial.1852 }
		}
	}
}

# Add Uncrowned trait to new kings and emperors
character_event = {
	id = emf_coronation.1101
	
	is_triggered_only = yes
	hide_window = yes
	
	trigger = {
		FROM = {
			higher_tier_than = DUKE
			temporary = no
		}
		NOT = { has_landed_title = e_hip }
	}
	
	immediate = {
		character_event = { id = emf_coronation.1103 days = 1 } # check for Uncrowned
		
		# Remove old ruler's coronation traits, if necessary
		FROMFROM = {
			if = {
				limit = {
					is_alive = yes
					OR = {
						trait = crowned
						trait = uncrowned
						trait = crowned_by_pope
						trait = strong_basileus
						trait = weak_basileus
					}	
				}
				if = {
					limit = {
						OR = {
							is_ruler = no
							lower_real_tier_than = KING
							AND = {
								real_tier = KING
								emf_imperial_government_trigger = yes
							}
						}
					}
					remove_trait = crowned
					remove_trait = uncrowned
					remove_trait = crowned_by_pope
					remove_trait = strong_basileus
					remove_trait = weak_basileus
					break = yes
				}
				if = {
					limit = {
						trait = crowned_by_pope
						emf_uses_hre_coronation = no
					}
					add_trait = crowned
				}
				if = {
					limit = {
						OR = {
							trait = strong_basileus
							trait = weak_basileus
						}
						emf_uses_byzantine_coronation = no
					}
					add_trait = crowned
				}
			}
		}
	}
}

# Ruler begins planning the coronation
character_event = {
	id = emf_coronation.1200
	desc = EVTDESC_emf_coronation_1200
	picture = GFX_evt_coronation
	
	is_triggered_only = yes
	hide_from = yes
	
	option = {
		name = EVTOPTA_emf_coronation_1200
		custom_tooltip = { text = send_invitations }
		hidden_tooltip = {
			# Clear out old opinion modifiers, just in case
			any_opinion_modifier_target = {
				limit = { reverse_has_opinion_modifier = { who = ROOT modifier = attending_coronation } }
				reverse_remove_opinion = { who = ROOT modifier = attending_coronation }
				remove_opinion = { who = ROOT modifier = attending_coronation }
			}
			any_opinion_modifier_target = {
				limit = { reverse_has_opinion_modifier = { who = ROOT modifier = declined_coronation } }
				reverse_remove_opinion = { who = ROOT modifier = declined_coronation }
			}
			
			# Send out invitations
			any_character = {
				limit = {
					is_adult = yes
					prisoner = no
					is_within_diplo_range = ROOT
					OR = {
						is_vassal_or_below = ROOT
						is_liege_or_above = ROOT
						is_close_relative = ROOT
						dynasty = ROOT
						is_allied_with = ROOT
						has_non_aggression_pact_with = ROOT
						AND = {
							same_realm = ROOT
							higher_tier_than = BARON
						}
						AND = {
							religion = ROOT
							independent = yes
							any_realm_province = {
								ROOT = {
									any_realm_province = {
										NOT = { distance = { where = PREVPREV value = 125 } }
									}
								}
							}
						}
					}
					NOR = {
						character = ROOT
						trait = incapable
						trait = excommunicated
						has_truce = ROOT
					}
				}
				letter_event = { id = emf_coronation.1201 }
			}
			
			# Look for priest to perform ceremony
			character_event = { id = emf_coronation.1205 days = 5 }
		}
	}
}
