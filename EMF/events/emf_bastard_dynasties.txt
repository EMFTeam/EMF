# -*- ck2.events -*-

namespace = emf_bastard_dynasties

# emf_bastard_dynasties.0 -- Younger House of Welf [on_birth]
character_event = {
	id = emf_bastard_dynasties.0

	is_triggered_only = yes # on_birth
	hide_window = yes

	trigger = {
		father_even_if_dead = { character = 1530 }
	}
	
	immediate = {
		father_even_if_dead = {
			character_event = { id = emf_bastard_dynasties.1 days = 1 }
		}
	}
}

# emf_bastard_dynasties.1 -- Younger House of Welf [day after on_birth, fires on possibly dead father]
character_event = {
	id = emf_bastard_dynasties.0

	is_triggered_only = yes
	hide_window = yes

	trigger = {
		character = 1530
	}

	immediate = {
		any_child_even_if_dead = {
			any_child_even_if_dead = {
				any_child_even_if_dead = {
					limit = { dynasty = ROOT }
					dynasty = 650
				}
				if = {
					limit = { dynasty = ROOT }
					dynasty = 650
				}
			}
			if = {
				limit = { dynasty = ROOT }
				dynasty = 650
			}
		}
		dynasty = 650
	}
}
