##########################################
# ADULTERY DISCOVERED
##########################################
# Fired for an unfaithful spouse who was discovered cheating
# Do not fire for both adulterers!
# Adulterer: event_target:target_adulterer
###########################################

character_event = {
	id = 20
	
	hide_window = yes
	
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = {
				OR = {
					is_married = event_target:target_adulterer
					is_consort = event_target:target_adulterer
					event_target:target_adulterer = { is_consort = PREV }
					character = event_target:target_adulterer
				}
			}
			break = yes
		}
		
		set_character_flag = adultery_discovered
		
		if = {
			limit = { has_nickname = no }
			if = {
				limit = {
					is_opposite_sex = event_target:target_adulterer
					OR = {
						AND = {
							is_married = yes
							trigger_if = {
								limit = { is_female = no }
								is_married_matrilineally = yes
							}
							trigger_else = {
								is_married_matrilineally = no
							}
						}
						is_consort = yes
					}
				}
				random = {
					chance = 50
					give_nickname = nick_the_unfaithful
				}
			}
			else = {
				random = {
					chance = 5
					give_nickname = nick_the_unchaste
				}
			}
		}
		
		if = {
			limit = {
				OR = {
					is_ascetic_trigger = yes
					AND = {
						is_ruler = yes
						is_theocracy = yes
						religion_scope = { NOT = { has_flag = emf_priests_can_marry } }
					}
				}
			}
			add_character_modifier = {
				name = adulterous_priest
				years = 20
			}
			tiered_high_piety_negative_effect = yes
			tiered_high_piety_negative_effect = yes
		}
		
		if = {
			limit = {
				age_diff = {
					who = event_target:target_adulterer
					years = 25
				}
				is_older_than = event_target:target_adulterer
				NOT = { has_character_modifier = cradle_robber }
			}
			add_character_modifier = {
				name = cradle_robber
				years = 5
			}
		}
		
		if = {
			limit = {
				is_pregnant = yes
				father_of_unborn_known = no
				father_of_unborn = {
					character = event_target:target_adulterer
				}
			}
			father_of_unborn_known = yes
		}
		
		if = {
			limit = {
				is_close_relative = event_target:target_adulterer
				emf_divine_blood_religion = no
			}
			if = {
				limit = { is_female = yes }
				add_character_modifier = {
					name = incestuous_adulteress
					years = 20
				}
			}
			else = {
				add_character_modifier = {
					name = incestuous_adulterer
					years = 20
				}
			}
			tiered_high_piety_negative_effect = yes
			tiered_high_piety_negative_effect = yes
			tiered_prestige_negative_effect = yes
		}
		else_if = {
			limit = { # Extramarital, same-sex relationships are generally not considered adultery at this point in time
				OR = {
					is_opposite_sex = event_target:target_adulterer
					is_married = yes
					is_consort = yes
				}
			}
			if = {
				limit = { is_female = yes }
				add_character_modifier = {
					name = adulteress
					years = 10
				}
			}
			else = {
				add_character_modifier = {
					name = adulterer
					years = 10
				}
			}
			tiered_high_piety_negative_effect = yes
			if = {
				limit = {
					OR = {
						is_married = yes
						is_consort = yes
					}
				}
				tiered_high_piety_negative_effect = yes
			}
		}
		
		if = {
			limit = {
				same_sex = event_target:target_adulterer
				trigger_if = {
					limit = { is_female = yes }
					emf_female_homosexual_indifferent_religion = no
				}
				trigger_else = {
					emf_male_homosexual_indifferent_religion = no
				}
			}
			tiered_high_piety_negative_effect = yes
			tiered_prestige_negative_effect = yes
			if = {
				limit = { is_female = no }
				tiered_high_piety_negative_effect = yes
				tiered_prestige_negative_effect = yes
				if = {
					limit = { religion_group = pagan_group }
					add_character_modifier = {
						name = known_sodomite_pagan
						years = 20
					}
				}
				else = {
					add_character_modifier = {
						name = known_sodomite
						years = 20
					}
				}
			}
		}

			limit = {
				trigger_if = {
					limit = { is_female = yes }

				trigger_else = {
					# Male "conquering" high-tier female in a non-female-dominant society
					religion_scope = { NOT = { has_religion_feature = religion_matriarchal } }
				}
				OR = {
					is_opposite_sex = event_target:target_adulterer

						limit = { is_female = yes }

					}
					trigger_else = {
						emf_male_homosexual_indifferent_religion = yes

					is_close_relative = event_target:target_adulterer

					is_ascetic_trigger = yes
					AND = {
						is_ruler = yes

						religion_scope = { NOT = { has_flag = emf_priests_can_marry } }
					}
				}

					# Make sure target_adulterer is seen as a lower real tier than ROOT (otherwise it's not really a conquest)

						AND = {
							NOT = { lower_real_tier_than = ROOT }

									any_spouse = { NOT = { lower_real_tier_than = PREVPREV } }
									any_consort = {
										is_senior_consort_party = yes
										NOT = { lower_real_tier_than = PREVPREV }
									}
									father_even_if_dead = { NOT = { lower_real_tier_than = PREVPREV } }
									mother_even_if_dead = { NOT = { lower_real_tier_than = PREVPREV } }
								}
							}
						}
						any_spouse = {
							NOT = { lower_real_tier_than = ROOT }
							ROOT = {
								NAND = {
									any_spouse = { NOT = { lower_real_tier_than = PREVPREV } }
									any_consort = {
										is_senior_consort_party = yes
										NOT = { lower_real_tier_than = PREVPREV }
									}
									father_even_if_dead = { NOT = { lower_real_tier_than = PREVPREV } }
									mother_even_if_dead = { NOT = { lower_real_tier_than = PREVPREV } }
								}
							}
						}
						any_consort = {
							is_senior_consort_party = yes
							NOT = { lower_real_tier_than = ROOT }
							ROOT = {
								NAND = {
									any_spouse = { NOT = { lower_real_tier_than = PREVPREV } }
									any_consort = {
										is_senior_consort_party = yes
										NOT = { lower_real_tier_than = PREVPREV }
									}
									father_even_if_dead = { NOT = { lower_real_tier_than = PREVPREV } }
									mother_even_if_dead = { NOT = { lower_real_tier_than = PREVPREV } }
								}
							}
						}
						father_even_if_dead = {
							NOT = { lower_real_tier_than = ROOT }
							ROOT = {
								NAND = {
									any_spouse = { NOT = { lower_real_tier_than = PREVPREV } }
									any_consort = {
										is_senior_consort_party = yes
										NOT = { lower_real_tier_than = PREVPREV }
									}
									father_even_if_dead = { NOT = { lower_real_tier_than = PREVPREV } }
									mother_even_if_dead = { NOT = { lower_real_tier_than = PREVPREV } }
								}
							}
						}

							NOT = { lower_real_tier_than = ROOT }

								NAND = {
									any_spouse = { NOT = { lower_real_tier_than = PREVPREV } }
									any_consort = {
										is_senior_consort_party = yes
										NOT = { lower_real_tier_than = PREVPREV }
									}
									father_even_if_dead = { NOT = { lower_real_tier_than = PREVPREV } }
									mother_even_if_dead = { NOT = { lower_real_tier_than = PREVPREV } }
								}
							}
						}
					}
				}
			}
			# Scale Prestige reward based off target_adulterer's (real) tier
			event_target:target_adulterer = { save_event_target_as = target_tier_character }
			tiered_prestige_reward_effect = yes
			tiered_prestige_reward_effect = yes
			clear_event_target = target_tier_character
		}
		
		# Spouse is more forgiving if it's a same-sex relationship unless they're religiously against it
		if = {
			limit = {
				OR = {
					is_opposite_sex = event_target:target_adulterer
					AND = {
						is_senior_consort_party = no
						any_consort = {
							OR = {
								trait = zealous
								ai_zeal = 25
							}
							trigger_if = {
								limit = { ROOT = { is_female = yes } }
								emf_female_homosexual_indifferent_religion = no
							}
							trigger_else = {
								emf_male_homosexual_indifferent_religion = no
							}
						}
					}
					any_spouse = {
						OR = {
							trait = zealous
							ai_zeal = 25
						}
						trigger_if = {
							limit = { ROOT = { is_female = yes } }
							emf_female_homosexual_indifferent_religion = no
						}
						trigger_else = {
							emf_male_homosexual_indifferent_religion = no
						}
					}
				}
			}
			if = {
				limit = { is_senior_consort_party = no }
				any_consort = {
					limit = {
						NAND = {
							is_lover = ROOT
							is_lover = event_target:target_adulterer
						}

							ROOT = { is_opposite_sex = event_target:target_adulterer }
							AND = {
								OR = {
									trait = zealous

								}
								trigger_if = {
									limit = { ROOT = { is_female = yes } }

								}

								}

						}
					}

						limit = { ROOT = { is_female = yes } }

							limit = { religion_scope = { has_religion_feature = religion_matriarchal } }

								who = ROOT
								modifier = opinion_unfaithful_wife_nocrime
								years = 10
							}

								is_married = event_target:target_adulterer
