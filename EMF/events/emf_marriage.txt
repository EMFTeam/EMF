# Add secret sterility trait once relevant

	id = emf_marriage.1
	
	hide_window = yes
	is_triggered_only = yes
	
	min_age = 14
	has_character_flag = emf_is_sterile
	
	trigger = {
		OR = {
			is_married = yes
			is_consort = yes
			has_lover = yes
		}
		NOR = {
			is_pregnant = yes

			trait = sterile_known
		}

	immediate = {
		clr_character_flag = emf_is_sterile
		set_character_flag = emf_was_sterile
		add_trait = sterile_secret
	}
}
