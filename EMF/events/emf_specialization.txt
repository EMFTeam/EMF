namespace = emf_specialization

# [0] CTE - Learning the sword from your educator
character_event = {
	id = emf_specialization.0
	desc = "EVTDESC550100"
	picture = GFX_evt_child_sword
	
	min_age = 5
	max_age = 16
	prisoner = no
	only_capable = yes

	trigger = {
		is_adult = no
		educator = {
			OR = {
				trait = misguided_warrior
				trait = tough_soldier
				trait = skilled_tactician
				trait = brilliant_strategist
				trait = poor_warrior
				trait = trained_warrior
				trait = skilled_warrior
				trait = master_warrior
				trait = duelist
				has_job_title = job_marshal
				has_minor_title = title_commander
			}
		}
		NOT = { trait = master_warrior }
	}

	mean_time_to_happen = {
		months = 30
		
		# Advancement slows down based on current level
		modifier = {
			factor = 2
			trait = poor_warrior
		}
		modifier = {
			factor = 4
			trait = trained_warrior
		}
		modifier = {
			factor = 6
			trait = skilled_warrior
		}
		
		# Mentor Effects
		modifier = {
			factor = 0.85
			educator = {
				OR = {
					trait = poor_warrior
					trait = trained_warrior
					trait = skilled_warrior
					trait = master_warrior
				}
			}
		}

		# Mentor has martial education
		modifier = {
			factor = 0.85
			educator = {
				OR = {
					has_job_title = job_marshal
					has_minor_title = title_commander
					trait = misguided_warrior
					trait = tough_soldier
					trait = skilled_tactician
					trait = brilliant_strategist
				}
			}
		}
		
		# Mentor is skilled in combat
		modifier = {
			factor = 0.85
			educator = { trait = duelist }
		}

		# Personality Effects
		#Good
		modifier = {
			factor = 0.85
			OR = {
				trait = diligent
				trait = brave
			}
		}
		#Bad
		modifier = {
			factor = 1.35
			OR = {
				trait = slothful
				trait = craven
			}
		}		
		
		# Intelligence Effects
		#Good
		modifier = {
			factor = 0.85
			trait = genius
		}
		modifier = {
			factor = 0.9
			OR = {
				trait = quick
				trait = shrewd
			}
		}
		#Bad
		modifier = {
			factor = 1.2
			OR = {
				trait = slow
				trait = dull
			}
		}
		modifier = {
			factor = 1.35
			trait = imbecile
		}
		
		# Physical Effects
		#Good
		modifier = {
			factor = 0.85
			is_strong_trigger = yes
		}
		#Bad
		modifier = {
			factor = 1.35
			is_weak_trigger = yes
		}
		
		#Sexism
		modifier = {
			factor = 1.4
			is_female = yes
		}
	}

	immediate = {
		if = {
			limit = { educator = { ai = no } }
			educator = { save_event_target_as = emf_notify_receiver }
		}
	}

	option = {
		name = "EVTOPTA5550100" # Stick 'em with the pointy end!
		if = {
			limit = { trait = skilled_warrior }
			tooltip = { remove_trait = skilled_warrior }
			emf_notify_add_master_warrior_effect = yes
		}
		if = {
			limit = { trait = trained_warrior }
			tooltip = { remove_trait = trained_warrior }
			emf_notify_add_skilled_warrior_effect = yes
		}
		if = {
			limit = { trait = poor_warrior }
			tooltip = { remove_trait = poor_warrior }
			emf_notify_add_trained_warrior_effect = yes
		}
		if = {
			limit = {
				NOR = {
					trait = poor_warrior
					trait = trained_warrior
					trait = skilled_warrior
					trait = master_warrior
				}
			}
			emf_notify_add_poor_warrior_effect = yes
		}
	}
}

# [2] CTE - Getting old.
character_event = {
	id = emf_specialization.2
	desc = "EVTDESC5550112"
	picture = GFX_evt_illness
	
	only_playable = yes
	min_age = 45

	trigger = {
		OR = {
			trait = poor_warrior
			trait = trained_warrior
			trait = skilled_warrior
			trait = master_warrior
		}
		NOT = { has_character_flag = cte_demoted }
	}

	mean_time_to_happen = {
		months = 180

		# Personality Effects
		#Good
		modifier = {
			factor = 1.25
			OR = {
				trait = diligent
				trait = brave
				trait = duelist
			}
		}
		#Bad
		modifier = {
			factor = 0.75
			trait = slothful
		}
		
		# Physical Effects
		#Good
		modifier = {
			factor = 1.5
			is_strong_trigger = yes
		}
		#Bad
		modifier = {
			factor = 0.2
			is_weak_trigger = yes
		}
	}

	option = {
		name = "CURSES"
		set_character_flag = cte_demoted
		
		if = {
			limit = { trait = poor_warrior }
			remove_trait = poor_warrior
		}
		if = {
			limit = { trait = trained_warrior }
			remove_trait = trained_warrior
			add_trait = poor_warrior
		}
		if = {
			limit = { trait = skilled_warrior }
			remove_trait = skilled_warrior
			add_trait = trained_warrior
		}
		if = {
			limit = { trait = master_warrior }
			remove_trait = master_warrior
			add_trait = skilled_warrior
		}
	}
}

# [5] diplo trait advancement due to guardian
character_event = {
	id = emf_specialization.5
	desc = PB_specialization.0.desc
	picture = GFX_evt_child_sword
	
	min_age = 6
	max_age = 16
	prisoner = no
	only_capable = yes
	
	trigger = {
		is_adult = no
		educator = {
			OR = {
				trait = naive_appeaser
				trait = underhanded_rogue
				trait = charismatic_negotiator
				trait = grey_eminence
				trait = novice_diplomat
				trait = trained_diplomat
				trait = skilled_diplomat
				trait = master_diplomat
				trait = gregarious
				has_job_title = job_chancellor
			}
		}
		NOT = { trait = master_diplomat }
	}

	immediate = {
		if = {
			limit = { educator = { ai = no } }
			educator = { save_event_target_as = emf_notify_receiver }
		}
	}
	
	mean_time_to_happen = {
		months = 30
		
		# Advancement slows down based on current level
		modifier = {
			factor = 2
			trait = novice_diplomat
		}
		modifier = {
			factor = 4
			trait = trained_diplomat
		}
		modifier = {
			factor = 6
			trait = skilled_diplomat
		}
		
		# Mentor Effects
		modifier = {
			factor = 0.85
			educator = {
				OR = {
					trait = novice_diplomat
					trait = trained_diplomat
					trait = skilled_diplomat
					trait = master_diplomat
				}
			}
		}
		modifier = {
			factor = 0.85
			educator = {
				OR = {
					trait = naive_appeaser
					trait = underhanded_rogue
					trait = charismatic_negotiator
					trait = grey_eminence
				}
			}
		}
		modifier = {
			factor = 0.85
			educator = { has_job_title = job_chancellor }
		}
		modifier = {
			factor = 0.85
			educator = {
				has_job_title = job_chancellor
				diplomacy = 12
				liege = { higher_real_tier_than = DUKE }
			}
		}
		modifier = {
			factor = 0.85
			educator = { trait = gregarious }
		}

		# Personality Effects
		modifier = {
			factor = 0.85
			trait = diligent
		}
		modifier = {
			factor = 0.85
			trait = gregarious
		}
		modifier = {
			factor = 1.35
			OR = {
				trait = slothful
				trait = shy
			}
		}		
		
		# Intelligence Effects
		modifier = {
			factor = 0.85
			trait = genius
		}
		modifier = {
			factor = 0.9
			OR = {
				trait = quick
				trait = shrewd
			}
		}
		modifier = {
			factor = 1.2
			OR = {
				trait = slow
				trait = dull
			}
		}
		modifier = {
			factor = 1.35
			trait = imbecile
		}
		#Male aristocrats are supposed to know how to fight and rule, not write and talk -- totally diff, yo
		modifier = {
			factor = 1.5
			is_primary_heir = yes
			is_female = no
		}
		#Sexism
		modifier = {
			factor = 1.25
			is_female = no
		}
	}

	option = {
		name = "OK"
		if = {
			limit = { trait = skilled_diplomat }
			tooltip = { remove_trait = skilled_diplomat }
			emf_notify_add_master_diplomat_effect = yes
		}
		if = {
			limit = { trait = trained_diplomat }
			tooltip = { remove_trait = trained_diplomat }
			emf_notify_add_skilled_diplomat_effect = yes 
		}
		if = {
			limit = { trait = novice_diplomat }
			tooltip = { remove_trait = novice_diplomat }
			emf_notify_add_trained_diplomat_effect = yes
		}
		if = {
			limit = {
				NOR = {
					trait = novice_diplomat
					trait = skilled_diplomat
					trait = trained_diplomat
					trait = master_diplomat
				}
			}
			emf_notify_add_novice_diplomat_effect = yes
		}
	}
}
