# Provincia Syria Palaestina
narrative_event = {
	id = LoR.1
	title = EVTNAMELoR.1
	desc = EVTDESCLoR.1
	
	picture = GFX_evt_victory_arch_byzantine
	
	has_dlc = "Legacy of Rome"
	
	is_triggered_only = yes
	major = yes
	
	major_trigger = {
		OR = {
			character = ROOT
			is_vassal_or_below = ROOT
			is_liege_or_above = ROOT
		}
	}
	
	trigger = {
		is_alternate_start = no
		top_liege = { emf_has_roman_empire = yes }
		NOT = { has_global_flag = syria_palaestina }
		FROM = { # Filter before expensive completely_controls checks
			trigger_if = {
				limit = { lower_tier_than = DUKE }
				OR = {
					de_jure_liege_or_above = d_oultrejourdain
					de_jure_liege_or_above = d_aleppo
					de_jure_liege_or_above = d_palmyra
					de_jure_liege_or_above = d_damascus
					de_jure_liege_or_above = d_jerusalem
					de_jure_liege_or_above = d_ascalon
					de_jure_liege_or_above = d_galilee
					de_jure_liege_or_above = d_edessa
					de_jure_liege_or_above = d_antioch
					de_jure_liege_or_above = d_tripoli
				}
			}
			trigger_else_if = {
				limit = { higher_tier_than = DUKE }
				OR = {
					de_jure_vassal_or_below = d_oultrejourdain
					de_jure_vassal_or_below = d_aleppo
					de_jure_vassal_or_below = d_palmyra
					de_jure_vassal_or_below = d_damascus
					de_jure_vassal_or_below = d_jerusalem
					de_jure_vassal_or_below = d_ascalon
					de_jure_vassal_or_below = d_galilee
					de_jure_vassal_or_below = d_edessa
					de_jure_vassal_or_below = d_antioch
					de_jure_vassal_or_below = d_tripoli
				}
			}
			trigger_else = {
				OR = {
					title = d_oultrejourdain
					title = d_aleppo
					title = d_palmyra
					title = d_damascus
					title = d_jerusalem
					title = d_ascalon
					title = d_galilee
					title = d_edessa
					title = d_antioch
					title = d_tripoli
				}
			}
		}
		top_liege = {
			### WARNING: completely_controls is a very demanding trigger.
			completely_controls = d_oultrejourdain
			completely_controls = d_aleppo
			completely_controls = d_palmyra
			completely_controls = d_damascus
			completely_controls = d_jerusalem
			completely_controls = d_ascalon
			completely_controls = d_galilee
			completely_controls = d_edessa
			completely_controls = d_antioch
			completely_controls = d_tripoli
		}
	}
	
	immediate = {
		set_global_flag = syria_palaestina
	}
	
	option = {
		name = EVTOPTALoR.1
		trigger = { top_liege = { emf_has_roman_empire = yes } }
		if = {
			limit = { emf_has_roman_empire = yes }
			prestige = 100
		}
	}
	
	option = {
		name = I_SEE
		trigger = { NOT = { top_liege = { emf_has_roman_empire = yes } } }
	}
}
