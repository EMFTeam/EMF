character_event = { # Pillaging Repeating Event
	id = HL.4800
	hide_window = yes

	is_triggered_only = yes

	ai = no

	trigger = {
		pacifist = no
		any_realm_province = {
			any_province_holding = {
				owner = { character = ROOT }
				has_siege = no
				has_holding_modifier = recently_burnt_the_land
				NOT = { has_holding_modifier = stop_burning_land }
				# Make sure province's culture cannot flip from pillaging
				OR = {
					has_any_building = yes
					location = {
						OR = {
							culture = ROOT
							any_province_holding = {
								NOR = {
									title = PREVPREV
									holding_type = nomad
								}
							}
						}
					}
				}
			}
		}
	}

	immediate = {
		character_event = { id = HL.4800 days = 180 }

		add_character_modifier = {
			modifier = pillage_cooldown
			hidden = yes
			months = 6
		}

		any_realm_province = {
			any_province_holding = {
				limit = {
					owner = { character = ROOT }
					has_siege = no
					has_holding_modifier = recently_burnt_the_land
					NOT = { has_holding_modifier = stop_burning_land }
					# Make sure province's culture cannot flip from pillaging
					OR = {
						has_any_building = yes
						location = {
							OR = {
								culture = ROOT
								any_province_holding = {
									NOR = {
										title = PREVPREV
										holding_type = nomad
									}
								}
							}
						}
					}
				}
				remove_holding_modifier = recently_burnt_the_land
				add_holding_modifier = {
					modifier = recently_burnt_the_land
					months = 7
				}
				location = {
					remove_province_modifier = rampaging_nomads
					remove_province_modifier = burnt_the_land
					add_province_modifier = {
						modifier = rampaging_nomads
						years = 10
					}
					add_province_modifier = {
						modifier = burnt_the_land
						years = 10
					}
				}

				#Provide benefits for pillaging
				trigger_switch = {
					on_trigger = holding_type
					city = {
						if = {
							limit = { has_any_building = no }
							FROM = {
								wealth = 20
								economy_techpoints = 5
							}
						}
						else_if = {
							limit = { num_of_buildings = 6 }
							FROM = {
								wealth = 80
								economy_techpoints = 20
							}
						}
						else_if = {
							limit = { num_of_buildings = 4 }
							FROM = {
								wealth = 80
								economy_techpoints = 15
							}
						}
						if = {
							limit = { num_of_buildings = 2 }
							FROM = {
								wealth = 60
								economy_techpoints = 10
							}
						}
						if = {
							limit = { has_any_building = yes }
							FROM = {
								wealth = 40
								economy_techpoints = 5
							}
						}
					}
					temple = {
						if = {
							limit = { has_any_building = no }
							FROM = {
								wealth = 10
								culture_techpoints = 5
							}
						}
						else_if = {
							limit = { num_of_buildings = 6 }
							FROM = {
								wealth = 50
								culture_techpoints = 20
							}
						}
						else_if = {
							limit = { num_of_buildings = 4 }
							FROM = {
								wealth = 40
								culture_techpoints = 15
							}
						}
						else_if = {
							limit = { num_of_buildings = 2 }
							FROM = {
								wealth = 30
								culture_techpoints = 10
							}
						}
						else = {
							FROM = {
								wealth = 20
								culture_techpoints = 5
							}
						}
					}
					castle = {
						if = {
							limit = { has_any_building = no }
							FROM = {
								wealth = 10
								military_techpoints = 5
							}
						}
						else_if = {
							limit = { num_of_buildings = 6 }
							FROM = {
								wealth = 50
								military_techpoints = 20
							}
						}
						else_if = {
							limit = { num_of_buildings = 4 }
							FROM = {
								wealth = 40
								military_techpoints = 15
							}
						}
						else_if = {
							limit = { num_of_buildings = 2 }
							FROM = {
								wealth = 30
								military_techpoints = 10
							}
						}
						else = {
							FROM = {
								wealth = 20
								military_techpoints = 5
							}
						}
					}
					tribal = {
						if = {
							limit = { has_any_building = no }
							FROM = {
								wealth = 5
								population = 50
							}
						}
						else_if = {
							limit = { num_of_buildings = 5 }
							FROM = {
								wealth = 25
								population = 150
							}
						}
						else_if = {
							limit = { num_of_buildings = 4 }
							FROM = {
								wealth = 20
								population = 150
							}
						}
						else_if = {
							limit = { num_of_buildings = 3 }
							FROM = {
								wealth = 15
								population = 150
							}
						}
						else = {
							FROM = {
								wealth = 10
								population = 75
							}
						}
					}
				}

				#Destroy settlement if no buildings
				if = {
					limit = { has_any_building = no }
					destroy_settlement = THIS
					hidden_tooltip = {
						location = {
							province_event = { id = HL.4810 days = 1 }
							save_event_target_as = target_location
							ROOT = { character_event = { id = HL.4814 days = 2 random = 2 } }
						}
					}
				}
				#Destroy buildings
				else = {
					custom_tooltip = {
						text = "PILLAGE_DESTROY_BUILDINGS"
						hidden_tooltip = {
							destroy_random_building = THIS
							destroy_random_building = THIS
						}
					}
				}
			}
		}
	}
}

province_event = { # Clear prosperity/depopulated modifiers when province is empty
	id = HL.4810
	hide_window = yes

	is_triggered_only = yes

	immediate = {
		owner = {
			any_character = {
				limit = {
					NOT = { culture = PREV }
					is_incapable = no
					is_adult = yes
					prisoner = no
					culture = ROOT
					location = {
						NOT = {
							distance = {
								who = PREVPREV
								distance = 200
							}
						}
					}
				}
				opinion = { who = PREV modifier = opinion_outraged years = 10 }
			}
		}
		if = {
			limit = {
				NOT = {
					any_province_holding = {
						OR = {
							holding_type = city
							holding_type = temple
							holding_type = castle
							holding_type = tribal
						}
					}
				}
			}
			remove_province_modifier = prosperity_modifier_1
			remove_province_modifier = prosperity_modifier_2
			remove_province_modifier = prosperity_modifier_3
			remove_province_modifier = depopulated_1
			remove_province_modifier = depopulated_2
			remove_province_modifier = depopulated_3
		}
	}
}

character_event = {
	id = HL.4999
	hide_window = yes

	is_triggered_only = yes

	trigger = {
		is_nomadic = no
		FROMFROM = { is_nomadic = yes }
		#NAND = {
		#	FROMFROM = { culture = ROOT }
		#	FROMFROM = { is_nomadic = yes }
		#	ROOT = { is_nomadic = yes }
		#}
		FROM = {
			tier = COUNT
			location = {
				NOR = {
					has_province_modifier = incredibly_poor
					any_province_holding = { holding_type = tribal }
					any_province_holding = { holding_type = castle }
					#any_province_holding = { holding_type = temple }
					any_province_holding = { holding_type = city }
					has_settlement_construction = yes
				}
			}
		}
	}

	immediate = {
		if = { # Raiding Adventurers of the same culture become Clan Chiefs
			limit = {
				ROOT = {
					culture = FROMFROM
					OR = {
						AND = {
							is_ruler = yes
							is_landed = no
							is_patrician = no
							mercenary = no
							holy_order = no
							is_offmap_governor = no #offmap governors do not become nomads
						}
						has_character_modifier = raiding_adventurer_nomad_timer
					}
				}
			}
			FROM = {
				location = {
					build_holding = {
					   type = tribal
					   holder = ROOT
					}
				}
			}
			ROOT = {
				set_government_type = nomadic_government
				add_character_modifier = {
					modifier = raiding_adventurer_nomad_timer
					days = 10
					hidden = yes
				}
			}
			FROM = {
				location = {
					any_province_holding = {
						limit = { holding_type = tribal }
						destroy_settlement = THIS
					}
				}
			}
		}
		else_if = { # Raiding Adventurers of the wrong culture become Tribal Vassals
			limit = {
				ROOT = {
					NOT = { culture = FROMFROM }
					OR = {
						AND = {
							is_ruler = yes
							is_landed = no
							is_patrician = no
							mercenary = no
							holy_order = no
							is_offmap_governor = no #offmap governors do not become nomads
						}
						has_character_modifier = raiding_adventurer_tribal_timer
					}
				}
			}
			FROM = {
				location = {
					build_holding = {
					   type = tribal
					   holder = ROOT
					}
					if = {
						limit = { ROOT = { culture_group = altaic } }
						religion = ROOT
						culture = ROOT
					}
					else = {
						religion = FROMFROM
						culture = FROMFROM
					}
				}
			}
			ROOT = {
				set_government_type = tribal_government
				add_character_modifier = {
					modifier = raiding_adventurer_tribal_timer
					days = 10
					hidden = yes
				}
			}
		}
		else = {
			FROM = {
				location = {
					build_holding = {
					   type = tribal
					   holder = ROOT
					}
					religion = FROMFROM
					culture = FROMFROM
					add_province_modifier = {
						name = nomad_agitation
						years = 100
					}
				}
			}
		}
	}
