			any_province_holding = {
				owner = { character = ROOT }

				NOT = { has_holding_modifier = stop_burning_land }

province_event = { # Clear prosperity/depopulated modifiers when province is empty
	id = HL.4810
	hide_window = yes
	
	is_triggered_only = yes
	
	immediate = {
		owner = {
			any_character = {
				limit = {
					NOT = { culture = PREV }
					NOT = { trait = incapable }
					is_adult = yes
					prisoner = no
					culture = ROOT
					location = {
						NOT = {
							distance = {
								who = PREVPREV
								distance = 200
							}
						}
					}
				}
				opinion = { who = PREV modifier = opinion_outraged years = 10 }
			}
		}
		if = {
			limit = {
				NOT = {
					any_province_holding = {
						OR = {
							holding_type = city
							holding_type = temple
							holding_type = castle
							holding_type = tribal
						}
					}
				}
			}
			remove_province_modifier = prosperity_modifier_1
			remove_province_modifier = prosperity_modifier_2
			remove_province_modifier = prosperity_modifier_3
			remove_province_modifier = depopulated_1
			remove_province_modifier = depopulated_2
			remove_province_modifier = depopulated_3
		}
	}
}

			}
			break = yes
		}
		if = { # Raiding Adventurers of the wrong culture become Tribal Vassals
			limit = {
				ROOT = {
					NOT = { culture = FROMFROM }
					OR = {
						AND = {
							is_ruler = yes
							is_landed = no	
						}
						has_character_modifier = raiding_adventurer_tribal_timer
					}
				}
				FROMFROM = { is_nomadic = yes }
			}
			FROM = {
				location = {
					build_holding = {
					   type = tribal
					   holder = ROOT
					}
					if = {
						limit = {
							ROOT = { NOT = { culture_group = altaic } }
						}
						religion = FROMFROM
						culture = FROMFROM
					}
					if = {
						limit = {
							ROOT = { culture_group = altaic }
						}
						religion = ROOT
						culture = ROOT
					}
				}
			}
			ROOT = { 
				set_government_type = tribal_government
				add_character_modifier = {
					modifier = raiding_adventurer_tribal_timer
					days = 10
					hidden = yes
				}
			}
			break = yes
		}
		FROM = {
			location = {
				build_holding = {
				   type = tribal
				   holder = ROOT
				}
				religion = FROMFROM
				culture = FROMFROM
				add_province_modifier = {
					name = nomad_agitation
					years = 100
				}
			}
		}
	}
}
