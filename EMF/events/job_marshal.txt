# 20223: Hidden ping event to check the status of the raiders
character_event = {
	id = 20223

	hide_window = yes

	is_triggered_only = yes

	immediate = {
		if = {
			limit = { has_earmarked_regiments_not_raiding = tribal_organize_raid }
			if = {
				limit = { has_character_flag = tribal_organize_raid_troops_restless }
				clr_character_flag = tribal_organize_raid_troops_restless
				character_event = { id = 20224 }
			}
			else = {
				set_character_flag = tribal_organize_raid_troops_restless
			}
		}
		else = {
			clr_character_flag = tribal_organize_raid_troops_restless
			repeat_event = { id = 20223 days = 30 }
		}
	}
}
