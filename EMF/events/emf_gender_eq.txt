# emf_gender_eq.6 [Isis]
# Convert all possible titles to Absolute Cognatic succession. Also called on
# annual pulse to ensure that the world stays Absolute Cognatic. By all logic,
# that should be unnecessary but apparently it is.
character_event = {
	id = emf_gender_eq.6
	hide_window = yes
	is_triggered_only = yes

	only_independent = yes
	only_women = yes
	culture = hip_culture

	trigger = {
		has_landed_title = e_hip
		has_global_flag = emf_gender_eq_law
	}

	immediate = {
		# Status of Women (Zeus-only)
		if = {
			limit = { has_dlc = "Zeus" }
			any_title = {
				limit = {
					has_holder = yes
					is_primary_holder_title = yes
					NOT = { has_law = status_of_women_4 }
				}
				holder_scope = { set_character_flag = emf_no_law_penalties }
				add_law_w_cooldown = status_of_women_4
				holder_scope = { clr_character_flag = emf_no_law_penalties }
			}
		}
		# Absolute Cognatic Succession (not DLC-locked)
		any_title = {
			limit = {
				has_holder = yes
				succ_law_title = {
					NOT = { has_law = true_cognatic_succession }
					is_law_potential = true_cognatic_succession
				}
				NOT = { holder = ROOT }
			}
			succ_law_title = { set_title_flag = emf_not_feminist_enough }
		}
		any_title = {
			limit = {
				has_holder = yes
				has_title_flag = emf_not_feminist_enough
			}
			holder_scope = { set_character_flag = emf_no_law_penalties }
			add_law = true_cognatic_succession
			holder_scope = { clr_character_flag = emf_no_law_penalties }
			clr_title_flag = emf_not_feminist_enough
		}
	}
}
