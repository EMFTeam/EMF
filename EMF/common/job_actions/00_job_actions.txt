action_fabricate_claims = {
	attribute = diplomacy
	
	potential = {
		FROM = { is_nomadic = no }
	}
	
	trigger = {
		is_within_diplo_range = FROM
		OR = {
			NOT = { region = emf_region_sahara }
			FROM = {
				OR = {
					ai = no
					capital_scope = { region = emf_region_sahara }
				}
			}
			owner = {
				OR = {
					ai = no
					capital_scope = { NOT = { region = emf_region_sahara } }
				}
			}
		}
		# If AI, must be independent or vassal duke+ to fab claims outside top realm...
		OR = { 
			owner = { same_realm = FROM }
			FROM = { higher_tier_than = COUNT }
			FROM = { independent = yes }
			FROM = { ai = no }
		}
		owner = {
			NOR = {
				character = FROM # Can't fab claims on self-owned counties
				FROM = {
					job_chancellor = {
						character = PREVPREV # Can't fab claims on chancellor-owned counties
					}
				}
			}
		}
		OR = {
			county = {
				NOT = { claimed_by = FROM }
				holder_scope = {
					NOR = {
						character = FROM
						is_vassal_or_below = FROM # Can't fab claim on any_liege's counties
					}
				}
			}
			AND = {
				FROM = { ai = no }
				duchy = {
					NOT = { claimed_by = FROM }
					holder_scope = {
						NOR = {
							character = FROM
							is_vassal_or_below = FROM # Can't fab claim on any_liege's duchies
						}
					}
				}
			}
		}
		# Must be in same top realm OR subrealm neighboring it OR subrealm neighboring it via a single adjacent sea zone
		OR = {
			owner = { same_realm = FROM }
			any_neighbor_province = {
				OR = {
					owner_under_FROM = yes
					AND = {
						is_land = no
						any_neighbor_province = {
							owner_under_FROM = yes
						}
					}
				}
			}
		}
	}
	
	events = { 913 914 }
}

# Court Chaplain
action_inquisition = {
	attribute = learning
	
	potential = {
	}
	
	trigger = {
		num_of_settlements = 1
		NAND = {
			FROM = { job_spiritual = { has_character_modifier = spiritual_refused_pagans } }  # See Event ZE.5040
			owner = { religion_group = pagan_group }
		}
		OR = {
			owner = {
				FROM = { NOT = { religion_group = pagan_group } }
				independent = yes
				NOT = { religion = FROM }
				religion_group = pagan_group
				OR = {
					NOT = { religion = aztec }
					year = 1350
				}
				is_reformed_religion = no
				capital_scope = {
					province = ROOT
				}
				controls_religion = no
				FROM = {
					higher_tier_than = COUNT
					OR = {
						independent = yes
						higher_tier_than = DUKE
					}
				}
				NOT = {
					has_opinion_modifier = { who = FROM modifier = opinion_sent_preacher }
				}
				is_within_diplo_range = FROM
			}
			any_province_lord = {
				OR = {
					character = FROM
					is_liege_or_above = FROM
				}
			}
		}
	}

	events = { 900 901 902 903 }
}
