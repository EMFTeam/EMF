# -*- ck2.scripted_effects -*-

emf_make_court_family_neapolitan = {
	any_courtier = {
		limit = {
			is_liege_or_above = PREV
			is_ruler = no
			prisoner = no
			emf_is_neapolitan_source_culture = yes
			OR = {
				is_married = PREV
				is_close_relative = PREV
				AND = {
					dynasty = PREV
					NOT = { dynasty = none }
				}
			}
		}
		culture = neapolitan
	}
}

emf_make_court_neapolitan = {
	emf_make_court_family_neapolitan = yes
}

emf_make_province_neapolitan = {
	culture = neapolitan
	set_province_flag = melting_pot
	set_province_flag = emf_melting_pot_neapolitan
	any_province_lord = {
		limit = {
			tier = BARON
			emf_is_neapolitan_source_culture = yes
		}
		culture = neapolitan
		hidden_tooltip = { emf_make_court_family_neapolitan = yes }
	}
}
