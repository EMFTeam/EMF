emf_new_character = {
	hidden_effect = {
		if = {
			limit = {
				OR = {
					religion = hellenic_pagan
					religion = hellenic_pagan_reformed
					has_religion_feature = religion_astrology
				}
			}
			add_western_zodiac_trait_effect = yes
		}
		if = {
			limit = {
				has_religion_feature = religion_no_leader
				religion_group = pagan_group
			}
			random_list = {  
				5  = { add_trait = pagan_branch_1 }
				25 = { add_trait = pagan_branch_2 }
				25 = { add_trait = pagan_branch_3 }
				25 = { add_trait = pagan_branch_4 }
			}
		}
		if = {
			limit = {
				trait = brilliant_strategist
				leader_traits < 1
			}
			emf_add_random_generic_leader_trait = yes
		}
		character_event = { id = emf_startup.10 }
		character_event = { id = emf_startup.11 }
	}
}

	hidden_effect = {

			}
			character_event = { id = RoI.30121 }
		}
	}
	emf_new_character = yes
}

# distribute THIS barony title to a new, appropriate character and vassalize him/her to THIS's county
emf_distribute_barony = {
	hidden_effect = {
		dejure_liege_title = { owner = { save_event_target_as = emf_county_holder_tmp } }
		event_target:emf_county_holder_tmp = {
			if = {
				limit = { PREV = { holding_type = city } }
				if = {
					limit = { has_religion_feature = religion_matriarchal }
					create_character = {
						random_traits = yes
						female = yes
						dynasty = none
						age = 16
						attributes = { stewardship = 7 }
					}
				}
				else_if = {
					limit = {
						OR = {
							gender_equality_trigger = yes
							has_religion_feature = religion_equal
							has_religion_feature = religion_feature_bon
						}
					}
					create_character = {
						random_traits = yes
						female = 50 # 50% chance of female
						dynasty = none
						age = 16
						attributes = { stewardship = 7 }
					}
				}
				else = {
					create_character = {
						random_traits = yes
						female = no
						dynasty = none
						age = 16
						attributes = { stewardship = 7 }
