emf_character_become_heretic_effect = {
	# Prefer heresy of a fellow dynasty member
	if = {
		limit = { any_dynasty_member = { is_heresy_of = PREV } }
		random_dynasty_member = {
			limit = { is_heresy_of = PREV }
			reverse_religion = PREV
		}
		break = yes
	}
	
	location = { save_event_target_as = heretic_province }
	save_event_target_as = converter
	emf_become_heretic_effect = yes
}

emf_province_become_heretic_effect = {
	save_event_target_as = heretic_province
	save_event_target_as = converter
	emf_become_heretic_effect = yes
}

emf_become_heretic_effect = {
	# Prefer heresy of their current location
	if = {
		limit = {
			event_target:heretic_province = {
				event_target:converter = { is_parent_religion = PREV }
			}
		}
		event_target:heretic_province = {
			event_target:converter = { religion = PREV }
		}
		break = yes
	}
	# Prefer a neighboring heresy
	if = {
		limit = {
			event_target:heretic_province = {
				any_neighbor_province = {
					event_target:converter = { is_parent_religion = PREV }
				}
			}
		}
		event_target:heretic_province = {
			random_neighbor_province = {
				limit = {
					event_target:converter = { is_parent_religion = PREV }
				}
				event_target:converter = { religion = PREV }
			}
		}
		break = yes
	}
	# Prefer same-religion group of their current location
	if = {
		limit = {
			event_target:heretic_province = {
				event_target:converter = {
					PREV = {
						NOT = { religion = PREV }
						religion_group = PREV
					}
				}
			}
		}
		event_target:heretic_province = {
			event_target:converter = { religion = PREV }
		}
		break = yes
	}
	# Prefer a same-religion group of a same-culture group neighboring province if Pagan or Muslim
	if = {
		limit = {
			OR = {
				religion_group = pagan_group
				religion_group = muslim
			}
			event_target:heretic_province = {
				any_neighbor_province = {
					event_target:converter = {
						PREV = {
							culture_group = PREV
							religion_group = PREV
							NOT = { religion = PREV }
						}
					}
				}
			}
		}
		event_target:heretic_province = {
			random_neighbor_province = {
				limit = {
					event_target:converter = {
						PREV = {
							culture_group = PREV
							religion_group = PREV
							NOT = { religion = PREV }
						}
					}
				}
				event_target:converter = { religion = PREV }
			}
		}
		break = yes
	}
	# Prefer an existing heresy in the realm
	if = {
		limit = {
			event_target:heretic_province = {
				owner = {
					top_liege = {
						any_realm_province = {
							event_target:converter = { is_parent_religion = PREV }
							NOT = { distance = { where = PREVPREVPREV value = 200 } }
						}
					}
				}
			}
		}
		event_target:heretic_province = {
			owner = {
				top_liege = {
					random_realm_province = {
						limit = {
							event_target:converter = { is_parent_religion = PREV }
							NOT = { distance = { where = PREVPREVPREV value = 200 } }
						}
						event_target:converter = { religion = PREV }
					}
				}
			}
		}
		break = yes
	}
	# ...otherwise, just pick a religion to convert to
	if = {
		limit = {
			OR = {
				religion = catholic
				religion = cathar
				religion = fraticelli
				religion = waldensian
				religion = lollard
				religion = adoptionist
				religion = arian
			}
		}
		emf_catholic_conversion_effect = yes
	}
	if = {
		limit = {
			religion_group = christian
			NOR = {
				religion = catholic
				religion = cathar
				religion = fraticelli
				religion = waldensian
				religion = lollard
				religion = adoptionist
				religion = arian
			}
		}
		emf_eastern_conversion_effect = yes
	}
	if = {
		limit = { religion_group = muslim }
		emf_muslim_conversion_effect = yes
	}
	if = {
		limit = { religion_group = zoroastrian_group }
		emf_zoroastrian_conversion_effect = yes
	}
	if = {
		limit = { religion_group = jewish_group }
		emf_jewish_conversion_effect = yes
	}
	become_heretic = yes
}
