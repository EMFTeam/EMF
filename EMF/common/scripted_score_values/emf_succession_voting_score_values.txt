# Used in succession_voting rules to avoid having to sync changes across multiple rules

# Basic modifiers used by pretty much every succession voting rule
emf_succession_voting_elector_selection_default_head = {
	additive_modifier = {
		value = -1000
		FROMFROM = {
			OR = {
				has_holder = no
				NOR = { #Either title is higher than Duke-tier, or must be primary.
					higher_tier_than = DUKE
					holder_scope = { primary_title = { title = PREVPREV } }
				}
			}
		}
	}
	additive_modifier = { # Current ruler is always an elector.
		value = 500
		FROMFROM = { holder = ROOT }
	}
}

# Modifiers to be used for most normal elective rules (Feudal Elective, Tanistry, Elective Gavelkind, etc.)
emf_succession_voting_elector_selection_basic_elective = {
	additive_modifier = {
		value = -100
		NAND = {
			# New change: independents can no longer vote in (Feudal) Elective, though they can potentially still be candidates and win
			is_liege_or_above = FROMFROM
			# Any rulers of one or two lower tiers than the main title are viable Electors.
			trigger_if = {
				limit = { FROMFROM = { real_tier = EMPEROR } }
				higher_real_tier_than = COUNT # Dukes and Kings
			}
			trigger_else_if = {
				limit = { FROMFROM = { real_tier = KING } }
				higher_real_tier_than = BARON # Counts and Dukes
			}
			trigger_else = {
				is_ruler = yes # Barons and Counts (or just Barons for Counties)
			}
		}
	}
	additive_modifier = {
		value = -100
		# If title is not titular, vassals need to either be de jure vassals or have no other title affecting them of the same tier (i.e. no titular titles or other de jure liege titles held by the same liege)
		NOR = {
			FROMFROM = {
				OR = {
					is_titular = yes
					holder_scope = {
						primary_title = { title = PREVPREV }
						NOT = {
							any_demesne_title = {
								real_tier = ROOT_FROMFROM
								OR = {
									is_titular = yes
									de_jure_vassal_or_below = ROOT
								}
							}
						}
					}
				}
			}
			any_demesne_title = {
				lower_real_tier_than = ROOT_FROMFROM
				trigger_if = {
					limit = { FROMFROM = { real_tier = EMPEROR } }
					higher_real_tier_than = COUNT # Duchies and Kingdoms
				}
				trigger_else_if = {
					limit = { FROMFROM = { real_tier = KING } }
					higher_real_tier_than = BARON # Counties and Duchies
				}
				trigger_else = {
					always = yes
				}
				any_dejure_liege = { title = ROOT_FROMFROM }
			}
		}
	}
	additive_modifier = {
		value = 10
		is_feudal = yes
	}
	additive_modifier = { # Theocracies favored over other government forms, feudalism left at base score.
		value = 14
		is_theocracy = yes
		#Tied to religion.
		OR = {
			is_heretic = no
			FROMFROM = {
				holder_scope = {
					OR = {
						religion = ROOT
						NOT = { religion_group = ROOT }
					}
				}
			}
		}
		FROMFROM = {
			holder_scope = {
				OR = {
					religion = ROOT
					is_heretic = yes
					emf_unreformed_pagan = yes
					religion_scope = {
						trigger_if = {
							limit = { has_religion_features = yes }
							has_religion_feature = religion_cosmopolitan
						}
						trigger_else = {
							religion = hellenic_pagan_reformed
						}
					}
					AND = {
						has_random_religions = no
						religion = hellenic_pagan
					}
				}
			}
		}
	}
	additive_modifier = {
		value = 10
		OR = {
			is_tribal = yes
			is_nomadic = yes
			is_republic = yes
		}
		# Non-feudals and non-theocracies can only vote if they are not baronies and are either direct vassals or independent
		OR = {
			higher_real_tier_than = BARON
			FROMFROM = { holder_scope = { is_feudal = no } }
		}
		FROMFROM = { holder_scope = { is_liege_of = ROOT } }
	}
}

# Modifiers used fairly commonly to prioritize by culture and religion when there would be too many electors
emf_succession_voting_elector_selection_default_prioritization = {
	additive_modifier = { # Prioritization based on culture
		value = -2
		FROMFROM = { holder_scope = { NOT = { culture = ROOT } } }
	}
	additive_modifier = { # Prioritization based on culture group
		value = -2
		FROMFROM = { holder_scope = { NOT = { culture_group = ROOT } } }
	}
	additive_modifier = { # Prioritization based on religion
		value = -2
		FROMFROM = {
			holder_scope = {
				is_heretic = no
				NOR = {
					religion = ROOT
					religion_scope = {
						trigger_if = {
							limit = { has_religion_features = yes }
							has_religion_feature = religion_cosmopolitan
						}
						trigger_else = {
							religion = hellenic_pagan_reformed
						}
					}
					AND = {
						has_random_religions = no
						religion = hellenic_pagan
					}
				}
			}
		}
	}
	additive_modifier = { # Prioritization based on religion group
		value = -2
		FROMFROM = {
			holder_scope = {
				NOR = {
					religion_group = ROOT
					emf_syncretized_religion_with_ROOT = yes
					religion_scope = {
						trigger_if = {
							limit = { has_religion_features = yes }
							has_religion_feature = religion_cosmopolitan
						}
						trigger_else = {
							religion = hellenic_pagan_reformed
						}
					}
					AND = {
						has_random_religions = no
						religion = hellenic_pagan
					}
				}
			}
		}
	}
}
