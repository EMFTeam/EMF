	}

	# AI Pick chance
	chance = {

		modifier = {
			factor = 2

					is_ruler = yes
					ai_rationality > 30
				}
				AND = {
					is_ruler = no
					liege = { ai_rationality > 30 }
				}
			}
		}
		modifier = {
			factor = 0.5
			is_female = yes
			OR = {
				AND = {
					is_ruler = yes

				}
				AND = {
					is_ruler = no
					liege = { emf_feminist_religion = no }
				}
			}
		}
		modifier = {
			factor = 10
			OR = {
				AND = {
					is_female = no
					liege = {
						primary_title = {
							succ_law_title = {
								NOR = {
									has_law = enatic_succession
									has_law = enatic_cognatic_succession
								}
							}
						}
					}
				}
				AND = {
					is_female = yes
					liege = {
						primary_title = {
							succ_law_title = {
								NOR = {
									has_law = agnatic_succession
									has_law = cognatic_succession
								}
							}
						}
					}
				}
			}
			OR = {
				AND = {
					is_ruler = yes
					OR = {
						is_nomadic = yes
						is_tribal = yes
					}
				}
				AND = {
					is_ruler = no

							succ_law_title = {
								has_law = succ_byzantine_elective
								NOR = {
									has_law = enatic_succession
									has_law = enatic_cognatic_succession
								}
							}
						}
					}
				}
				AND = {
					is_female = yes
					any_liege = {
						primary_title = {
							succ_law_title = {
								has_law = succ_byzantine_elective
