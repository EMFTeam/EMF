	potential = {

	}
	allow = {
		is_adult = yes
		is_incapable = no
	}
	chance = {
		factor = 20
		modifier = {
			factor = 0.1
			trait = content
		}
		modifier = {
			factor = 1.5
			demesne_size = 2
		}
		modifier = {
			factor = 1.5
			demesne_size = 3
		}
		modifier = {
			factor = 1.5

		}

			factor = 10

		}
		modifier = {
			factor = 2
			trait = ambitious
		}
	}
	creation_effect = {
		if = {
			limit = { ai = yes }
			add_character_modifier = {

			}
		}
	}

		higher_real_tier_than = COUNT
	}
	abort = {
		OR = {
			is_playable = no

			}
		}
	}

		if = {
			limit = { ai = yes }

		}
	}

			if = {
				limit = { ai = yes }

			}
		}
		prestige = 50
	}
