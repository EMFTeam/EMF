obj_become_chancellor = {
	type = character
	
	expectation_of_liege = yes
	
	potential = {
		independent = no
		liege = {
			is_playable = yes
			religion_group = ROOT
			NOT = { is_married = ROOT }
			any_demesne_title = { is_titular = no }
		}
	}

	allow = {
		diplomacy = 8
		emf_is_voter = no
		can_be_chancellor_trigger = yes
		liege = {
			OR = {
				primary_title = { NOT = { has_law = protected_appointment_1 } }
				job_chancellor = { is_powerful_vassal = no }
			}
		}
	}
	
	chance = {
		factor = 10
		
		modifier = {
			factor = 0
			NOR = {
				NOT = { has_dlc = "Zeus" }
				trait = ambitious
				is_powerful_vassal = yes
				relative_income_to_liege = 0.5
				NOT = { liege = { job_chancellor = { always = yes } } }
				liege = {
					job_chancellor = {
						ROOT = { attribute_diff = { character = PREV attribute = diplomacy value = 2 } }
					}
				}
			}
		}
		modifier = {
			factor = 0
			liege = {
				job_chancellor = {
					attribute_diff = {
						character = ROOT
						attribute = diplomacy
						value = 5
					}
				}
			}
		}
		modifier = {
			factor = 0.01
			trait = content
		}
		modifier = {
			factor = 0.1
			trait = shy
		}
		modifier = {
			factor = 0.25
			trait = slothful
		}
		modifier = {
			factor = 1.25
			from_ruler_dynasty = yes
		}
		modifier = {
			factor = 1.5
			trait = poet
		}
		modifier = {
			factor = 1.5
			trait = falconer
		}
		modifier = {
			factor = 1.5
			trait = gregarious
		}
		modifier = {
			factor = 2.0
			trait = ambitious
		}
		modifier = {
			factor = 5
			OR = {
				trait = grey_eminence
				trait = socializer
			}
		}
		modifier = {
			factor = 0.5
			NOR = {
				trait = naive_appeaser
				trait = underhanded_rogue
				trait = charismatic_negotiator
				trait = grey_eminence
			}
		}
		modifier = {
			factor = 1.25
			diplomacy = 12
		}
		modifier = {
			factor = 1.25
			diplomacy = 14
		}
		modifier = {
			factor = 1.25
			diplomacy = 16
		}
		modifier = {
			factor = 1.5
			diplomacy = 18
		}
		modifier = {
			factor = 1.5
			diplomacy = 20
		}
		modifier = {
			factor = 5
			liege = { job_chancellor = { NOT = { diplomacy = ROOT } } }
		}
		modifier = { 
			factor = 10
			has_dlc = "Zeus"
			is_ruler = yes
			OR = {
				relative_power_to_liege = 0.25
				is_powerful_vassal = yes

		}
		modifier = { 
			factor = 5

		if = {
			limit = { ai = yes }

				name = emf_objective_timer
				duration = 1825
				hidden = yes
			}
		}
	}
	success = {
		has_job_title = job_chancellor
	}
	abort = {
		OR = {
			independent = yes

				emf_is_voter = yes
				NOT = { has_job_title = job_chancellor }

				primary_title = { has_law = protected_appointment_1 }

					NOT = { character = ROOT }
				}
			}

				ai = yes

			}
		}
	}
	abort_effect = {
		if = {
			limit = { ai = yes }
			remove_character_modifier = emf_objective_timer
		}
	}
	effect = {
		hidden_tooltip = {
			if = {
				limit = { ai = yes }
				remove_character_modifier = emf_objective_timer
			}
		}
		if = {
			limit = {
				NOT = { has_character_flag = obj_become_chancellor }
			}
			change_diplomacy = 1
		}
		opinion = {
			who = liege
			modifier = opinion_ambition_fulfilled
			months = 60
		}
		set_character_flag = obj_become_chancellor
	}
	is_job_ambition=yes
}

		}
