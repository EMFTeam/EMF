# -*- ck2.objectives -*-

# adapted from CK2+ (original source unknown)

obj_five_children = {
	type = character

	fertility = 0.1
	
	potential = {
		OR = {
			is_playable = yes
			spouse = { is_playable = yes }
		}
		is_adult = yes
		is_married = yes
		num_of_children = 1
		NOT = { num_of_children = 5 }
		NOT = { has_character_flag = obj_have_child_3 }
	}
	allow = {
		prisoner = no
		OR = {
			NOT = { age = 45 }
			is_female = no
		}
		NOR = {
			age = 65
			is_ascetic_trigger = yes
			trait = celibate
			trait = eunuch
			trait = incapable
		}
	}
	chance = {
		factor = 50

		modifier = {
			factor = 0.5
			is_female = no
		}
		modifier = {
			factor = 2.0
			trait = lustful
		}
		modifier = {
			factor = 1.5
			num_of_children = 2
		}
		modifier = {
			factor = 1.5
			num_of_children = 3
		}
		modifier = {
			factor = 1.5
			num_of_children = 4
		}
	}
	creation_effect = {
		if = {
			limit = { ai = yes }
			add_character_modifier = {
				name = emf_objective_timer
				duration = 1825
				hidden = yes
			}
		}
	}
	success = {
		num_of_children = 5
	}
	abort = {
		OR = {
			prisoner = yes
			age = 65
			AND = {
				age = 45
				is_female = yes
			}
			is_married = no
			trait = celibate
			trait = eunuch
			is_ascetic_trigger = yes
			trait = incapable
			AND = {
				ai = yes
				NOT = { has_character_modifier = emf_objective_timer }
			}
		}
	}
	abort_effect = {
		if = {
			limit = { ai = yes }
			remove_character_modifier = emf_objective_timer
		}
	}
	effect = {
		hidden_tooltip = {
			if = {
				limit = { ai = yes }
				remove_character_modifier = emf_objective_timer
			}
		}
		# FIXME: a Court Faction mood increase
		if = {
			limit = { religion_group = muslim }
			prestige = 50
		}
		if = {
			limit = { NOT = { religion_group = muslim } }
			prestige = 100
		}
		set_character_flag = obj_have_child_3
	}
}
