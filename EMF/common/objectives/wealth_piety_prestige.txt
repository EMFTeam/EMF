# Wealth
obj_amass_wealth = {
	type = character

	potential = {
		NOT = { has_character_flag = amassed_wealth }
		NOT = { has_character_flag = amassed_fortune }
		NOT = { has_character_flag = amassed_hoard }
	}
	allow = {
		is_adult = yes
		is_playable = yes
		OR = {
			is_landed = yes
			is_patrician = yes
		}
		NOT = { wealth = 200 }
		NOT = { trait = charitable }
		is_incapable = no
		NOT = { has_character_flag = amassed_wealth }
		NOT = { has_character_flag = amassed_fortune }
		NOT = { has_character_flag = amassed_hoard }
	}
	chance = {
		factor = 100

		modifier = {
			factor = 1.25
			trait = patient
		}
		modifier = {
			factor = 1.25
			trait = fortune_builder
		}
		modifier = {
			factor = 1.25
			trait = midas_touched
		}
		modifier = {
			factor = 10.0
			trait = greedy
		}
		modifier = {
			factor = 0.02
			trait = content
		}
		modifier = {
			factor = 0.1
			NOT = { yearly_income = 10 }
		}
		modifier = {
			factor = 0.5
			NOT = { yearly_income = 15 }
		}
		modifier = {
			factor = 0.5
			NOT = { yearly_income = 25 }
		}
	}
	creation_effect = {
		if = {
			limit = { ai = yes }
			add_character_modifier = {
				name = emf_objective_timer
				duration = 1825
				hidden = yes
			}
		}
	}
	success = {
		wealth = 500
		NOT = { has_character_modifier = small_loan }
		NOT = { has_character_modifier = medium_loan }
		NOT = { has_character_modifier = large_loan }
		NOT = { has_character_modifier = huge_loan }
	}
	abort = {
		OR = {
			trait = charitable
			is_incapable = yes
			AND = {
				ai = yes
				NOT = { has_character_modifier = emf_objective_timer }
			}
		}
	}
	abort_effect = {
		if = {
			limit = { ai = yes }
			remove_character_modifier = emf_objective_timer
		}
	}
	effect = {
		change_stewardship = 1
		hidden_tooltip = {
			set_character_flag = amassed_wealth
			if = {
				limit = { ai = yes }
				remove_character_modifier = emf_objective_timer
			}
		}
	}
}
