# -*- ck2.objectives -*-

# Once EMF is independent of PB, this file should be renamed to be consistent with EMF naming standards

# Convert A Province
# From Project Balance
obj_convert_a_province = {
	type = character
	
	creation_effect = {
		change_stewardship = -2
	}
	allow = {
		is_playable = yes
		is_adult = yes
		prisoner = no
		NOT = { trait = incapable }
		NOT = { trait = cynical }
		piety = 0
		OR = {
			AND = {
				ai = no		
				any_realm_province = {
					NOT = { religion = ROOT }
				}
			}
			any_demesne_province = {
				NOT = { religion = ROOT }
			}
		}
	}
	chance = {
		factor = 100
		
		modifier = {
			factor = 1.25
			trait = patient
		}
		modifier = {
			factor = 1.25
			trait = diligent
		}
		modifier = {
			factor = 0.01
			trait = content
		}
		modifier = {
			factor = 10
			trait = zealous
		}
		modifier = {
			factor = 2.0
			trait = crusader
		}
		modifier = {
			factor = 0.5
			NOT = { learning = 5 }
		}
		modifier = {
			factor = 1.5
			learning = 10
		}
	}
	creation_effect = {
		if = {
			limit = { ai = yes }
			add_character_modifier = {
				name = emf_objective_timer
				duration = 1825
				hidden = yes
			}
		}
	}
	success = {
		custom_tooltip = {
			text = emf_ctt_obj_convert_province_success
			hidden_tooltip = {
				has_character_flag = converted_a_province
			}
		}
	}
	abort = {
		OR = {
			prisoner = yes
			trait = incapable
			trait = cynical
			NOR = {
				AND = {
					ai = no		
					any_realm_province = {
						NOT = { religion = ROOT }
					}
				}
				AND = {
					ai = yes
					any_demesne_province = {
						NOT = { religion = ROOT }
					}
				}
			}
			AND = {
				ai = yes
				NOT = { has_character_modifier = emf_objective_timer }
			}
		}
	}
	abort_effect = {
		if = {
			limit = { ai = yes }
			remove_character_modifier = emf_objective_timer
		}
		change_stewardship = 2
	}
	effect = {
		clr_character_flag = converted_a_province
		piety = 25
		change_stewardship = 2
		if = {
			limit = {
				NOT = { has_character_flag = converted_province_ambition_done }
			}
			change_learning = 1
		}
		set_character_flag = converted_province_ambition_done
		letter_event = { id = emf_ambitions.2 days = 1 } # Reward for converting a province
		if = {
			limit = { ai = yes }
			remove_character_modifier = emf_objective_timer
		}
	}
}
