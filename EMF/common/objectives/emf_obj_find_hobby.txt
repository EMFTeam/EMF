# -*- ck2.objectives -*-

# Find a Hobby, relies on lifestyle_event.txt
obj_find_hobby = {
	type = character

	potential = {
		NOT = { has_dlc = "Way of Life" } # Not currently compatible with Way of Life
		is_adult = yes
	}
	allow = {
		prisoner = no
		age = 20
		NOT = { lifestyle_traits = 1 }
	}
	chance = {
		factor = 25

		modifier = {
			factor = 0.5
			trait = slothful
		}
		modifier = {
			factor = 0.75
			trait = content
		}
		modifier = {
			factor = 1.25
			trait = envious
		}
		modifier = {
			factor = 1.25
			trait = brave
		}
		modifier = {
			factor = 1.5
			trait = shy
		}
		modifier = {
			factor = 1.5
			trait = gregarious
		}
	}
	creation_effect = {
		if = {
			limit = { ai = yes }
			add_character_modifier = {
				name = emf_objective_timer
				duration = 1825
				hidden = yes
			}
		}
	}
	success = {
		lifestyle_traits = 1
	}
	abort = {
		OR = {
			is_incapable = yes
			prisoner = yes
			AND = {
				ai = yes
				NOT = { has_character_modifier = emf_objective_timer }
			}
		}
	}
	abort_effect = {
		if = {
			limit = { ai = yes }
			remove_character_modifier = emf_objective_timer
		}
	}
	effect = {
		prestige = 10
		if = {
			limit = { ai = yes }
			remove_character_modifier = emf_objective_timer
		}
	}
}
