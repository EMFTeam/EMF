						claimed_by = FROM

					}

				NOT = { is_liege_or_above = FROM }

								character = PREVPREV

						}
					}
				}

				NOT = { is_liege_or_above = FROM }
				NOT = { reverse_opinion = { who = FROM value = -50 } }
				FROM = { emf_can_execute_ROOT = yes }
			}
