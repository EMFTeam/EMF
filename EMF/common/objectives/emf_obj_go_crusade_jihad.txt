obj_go_crusade = {
	type = character

	potential = {
		ai = no
		OR = {
			religion = catholic
			religion = fraticelli
		}
		NOT = { has_flag = obj_go_crusade }
		NOT = { trait = crusader }
		can_be_marshal_trigger = yes
	}
	allow = {
		prisoner = no
		is_adult = yes
		religion_head = { has_called_crusade = yes }
		is_incapable = no
	}
	chance = {
		factor = 0
	}
	success = {
		trait = crusader
	}
	abort = {
		NAND = {
			ai = no
			OR = {
				religion = catholic
				religion = fraticelli
			}
			can_be_marshal_trigger = yes
			prisoner = no
			is_adult = yes
			religion_head = { has_called_crusade = yes }
			is_incapable = no
		}
	}
	abort_effect = {
	}
	effect = {
		pf_tradition_plus2_effect = yes
		piety = 100
		set_flag = obj_go_crusade
	}
}
