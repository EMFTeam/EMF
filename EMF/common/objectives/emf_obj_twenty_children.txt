# adapted from CK2+ (original source unknown)

obj_twenty_children = {
	type = character

	fertility = 0.1
	
	potential = {
		or = {
			is_playable = yes
			spouse = { is_playable = yes }
		}
		is_adult = yes
		is_married = yes
		num_of_children = 15
		not = { num_of_children = 20 }
		not = { has_character_flag = obj_have_child_6 }
	}

	allow = {
		prisoner = no
		not = { age = 65 }
		or = {
			is_female = no
			not = { age = 45 }
		}
		NOR = {
			trait = monk
			trait = nun
			trait = celibate
			trait = eunuch
			trait = incapable
		}
	}
	chance = {
		factor = 50

		modifier = {
			factor = 0.5
			is_female = no
		}
		modifier = {
			factor = 2.0
			trait = lustful
		}
		modifier = {
			factor = 1.5
			num_of_children = 16
		}
		modifier = {
			factor = 1.5
			num_of_children = 17
		}
		modifier = {
			factor = 1.5
			num_of_children = 18
		}
	}
	creation_effect = {
		if = {
			limit = { ai = yes }
			add_character_modifier = {
				name = emf_objective_timer
				duration = 1825
				hidden = yes
			}
		}
	}
	success = {
		num_of_children = 20
	}
	abort = {
		OR = {
			prisoner = yes
			age = 65
			and = {
				is_female = yes
				age = 45
			}
			is_married = no
			trait = monk
			trait = nun
			trait = celibate
			trait = eunuch
			trait = incapable
			AND = {
				ai = yes
				NOT = { has_character_modifier = emf_objective_timer }
			}
		}
	}
	abort_effect = {
		if = {
			limit = { ai = yes }
			remove_character_modifier = emf_objective_timer
		}
	}
	effect = {
		hidden_tooltip = {
			if = {
				limit = { ai = yes }
				remove_character_modifier = emf_objective_timer
			}
		}
		# FIXME: a Court Faction mood increase +2
		if = {
			limit = { religion_group = muslim }
			prestige = 50
		}
		if = {
			limit = { NOT = { religion_group = muslim } }
			prestige = 100
		}
		set_character_flag = obj_have_child_6
	}
}
