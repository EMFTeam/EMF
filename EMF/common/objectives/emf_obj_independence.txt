# -*- ck2.objectives -*-

obj_independence = {
	type = character

	potential = {
		ai = no
		independent = no
	}
	allow = {
		is_adult = yes
		is_landed = yes
		prisoner = no
		is_incapable = no
		liege = {
			independent = yes
			OR = {
				is_feudal = yes
				is_tribal = yes
				is_nomadic = yes
			}
		}
	}
	chance = {
		factor = 0
	}
	success = {
		independent = yes
	}
	abort = {
		OR = {
			ai = yes
			prisoner = yes
			is_incapable = yes
		}
	}
	abort_effect = {
	}
	effect = {
		if = {
			limit = { has_character_flag = obj_independence }
			prestige = 50
		}
		if = {
			limit = { NOT = { has_character_flag = obj_independence } }
			prestige = 200
			set_character_flag = obj_independence
		}
	}
}
