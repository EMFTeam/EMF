# -*- ck2.objectives -*-

obj_civilize_province = {
	type = character
	
	potential = {
		ai = no
		any_realm_province = {
			NOR = {
				culture = ROOT
				owner = { culture = PREV }
			}
		}
		NOT = { has_character_flag = emf_civilized_province }
	}
	allow = {
		prisoner = no
		is_adult = yes
		NOT = { trait = incapable }
	}
	chance = {
		factor = 0
	}
	creation_effect = {
		custom_tooltip = {
			text = emf_ctt_obj_civilize_province_creation_effect
			hidden_tooltip = {
				change_stewardship = -1
			}
		}
	}
	success = {
		custom_tooltip = {
			text = emf_ctt_obj_civilize_province_success
			hidden_tooltip = {
				has_character_flag = emf_civilized_province
			}
		}
	}
	abort = {
		OR = {
			trait = incapable
			prisoner = yes
			ai = yes
			NOR = {
				has_character_flag = emf_civilized_province
				any_realm_province = {
					NOR = {
						culture = ROOT
						owner = { culture = PREV }
					}
				}
			}
		}
	}
	abort_effect = {
		custom_tooltip = {
			text = emf_ctt_obj_civilize_province_abort_effect
			hidden_tooltip = {
				change_stewardship = 1
			}
		}
	}
	effect = {
		custom_tooltip = {
			text = emf_ctt_obj_civilize_province_abort_effect
			hidden_tooltip = {
				change_stewardship = 1
			}
		}
		if = {
			limit = { NOT = { has_character_flag = emf_obj_civilize_province } }			
			custom_tooltip = {
				text = emf_ctt_obj_civilize_province_first_effect
				hidden_tooltip = {
					change_stewardship = 1
				}
			}
			set_character_flag = emf_obj_civilize_province
		}
		prestige = 100
		clr_character_flag = emf_civilized_province
	}
}
