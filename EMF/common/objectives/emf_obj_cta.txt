# A version of these were integrated into the New Duel Engine & several other places (authored by ziji).
# This is a newer version that's EMF-specific due to the scripted_triggers used.

# no combat trait -> poor_warrior
obj_cta_0 = {
	type = character
	
	potential = {
		is_adult = yes
		OR = {
			is_ruler = yes
			has_job_title = job_marshal
			has_minor_title = title_commander
		}
		nor = {
			age = 50
			trait = blinded
		}
		NOR = {
			trait = poor_warrior
			trait = trained_warrior
			trait = skilled_warrior
			trait = master_warrior
		}
		not = { has_character_flag = cta_done }
		NAND = {
			controls_religion = yes
			religion_group = christian
		}
	}
	allow = {
		emf_council_marshal_trigger = yes
	}
	chance = {
		factor = 50
		
		modifier = {
			factor = 0
			not = { martial = 8 }
		}
		modifier = {
			factor = 20
			has_job_title = job_marshal
		}
		modifier = {
			factor = 15
			has_minor_title = title_commander
			NOT = { has_job_title = job_marshal }
		}
		modifier = {
			factor = 0.75
			age = 40
		}
		modifier = {
			factor = 0.25
			age = 45
		}
		modifier = {
			factor = 2
			trait = diligent
		}
		modifier = {
			factor = 1.5
			trait = ambitious
			martial = 10
		}
		modifier = {
			factor = 2.5
			trait = brave
		}
		modifier = {
			factor = 2.5
			trait = crusader
		}
		modifier = {
			factor = 2.5
			trait = mujahid
		}
		modifier = {
			factor = 1.75
			trait = scarred # already has proven battle experience, likely candidate to be in future battles
		}
		modifier = {
			factor = 1.75
			trait = agile
		}
		modifier = {
			factor = 2
			trait = strong
		}
		modifier = {
			factor = 10
			trait = duelist
		}
		modifier = {
			factor = 2
			trait = hunter
		}
		modifier = {
			factor = 0.25
			trait = weak
		}
		modifier = {
			factor = 0.25
			trait = clubfooted
		}
		modifier = {
			factor = 0.75
			trait = content
		}
		modifier = {
			factor = 0.25
			trait = craven
		}
		modifier = {
			factor = 0.05
			trait = slothful
		}
		modifier = {
			factor = 4
			trait = brilliant_strategist
		}
		modifier = {
			factor = 3
			trait = skilled_tactician
		}
		modifier = {
			factor = 2
			trait = tough_soldier
		}
		modifier = {
			factor = 1.5
			martial = 12
		}
		modifier = {
			factor = 1.5
			martial = 15
		}
		modifier = {
			factor = 1.5
			martial = 18
		}
		modifier = {
			factor = 1.5
			martial = 21
		}
		modifier = {
			factor = 1.5
			martial = 24
		}
		modifier = { # Be less likely to drop / not re-take the ambition if already gained 1/3 XP
			factor = 15
			has_character_flag = ctxp0
		}
		modifier = { # Be less likely to drop / not re-take the ambition if already gained 2/3 XP 
			factor = 30 # ctxp0 is exclusive to ctxp1, so these modifiers won't stack.
			has_character_flag = ctxp1
		}
	}
	success = {
		OR = { # CTA only advances 1 level, but external code or console might jump the trait higher
			trait = poor_warrior
			trait = trained_warrior
			trait = skilled_warrior
			trait = master_warrior
		}
	}
	abort = {
		OR = {
			NOR = {
				is_ruler = yes
				has_job_title = job_marshal
				has_minor_title = title_commander
			}
			age = 50
			trait = blinded
			AND = {
				controls_religion = yes
				religion_group = christian
			}
			emf_council_marshal_trigger = no
		}
	}
	effect = {
		if = {
			limit = { liege = { real_tier = emperor } }
			prestige = 50
		}
		if = {
			limit = { liege = { real_tier = king } }
			prestige = 40
		}
		if = {
			limit = { liege = { real_tier = duke } }
			prestige = 30
		}
		if = {
			limit = { liege = { lower_real_tier_than = duke } }
			prestige = 20
		}
	}
}
