			temporary = no
			is_primary_type_title = no
		}
	}
	allow = {
		prisoner = no
		NOR = {
			is_incapable = yes
			is_inaccessible_trigger = yes
			any_child = {
				is_alive = yes
				OR = {
					has_character_flag = introduced_heir
					has_character_modifier = groomed_to_perfection

		any_child = {
			prisoner = no
			host = { character = ROOT }
			is_alive = yes

				holder = ROOT

					AND = {
						tier = COUNT

						tier = BARON
						holding_type = family_palace
