# Council ambition

obj_become_council_member = {
	type = character

	expectation_of_liege = yes
	
	potential = {
		independent = no
	}

	allow = {
		liege = { independent = yes }
		NOT = {
			liege = { spouse = { character = ROOT } }
		}
		has_dlc = "Zeus"
		prisoner = no
		is_adult = yes
		is_councillor = no
		OR = {
			can_hold_title = job_chancellor
			can_hold_title = job_marshal
			can_hold_title = job_treasurer
			can_hold_title = job_spymaster
			can_hold_title = job_spiritual
		}
		NOR = { trait = incapable trait = in_hiding }
	}
	
	chance = {
		factor = 100
		
		modifier = {
			factor = 0.5
			trait = content
		}
		modifier = {
			factor = 3.0
			trait = ambitious
		}
		modifier = {
			factor = 0.5
			trait = shy
		}
		modifier = {
			factor = 1.5
			trait = gregarious
		}
		modifier = {
			factor = 1.5
			trait = proud
		}
		modifier = {
			factor = 0.5
			trait = humble
		}
		modifier = {
			factor = 1.25
			from_ruler_dynasty = yes
		}
		modifier = {
			factor = 1.25
			OR = {
				diplomacy = 10
				stewardship = 10
				intrigue = 10
				martial = 10
				learning = 10
			}
		}
		modifier = {
			factor = 1.5
			OR = {
				diplomacy = 16
				stewardship = 16
				intrigue = 16
				martial = 16
				learning = 16
			}
		}
		modifier = { 
			factor = 10
			among_most_powerful_vassals = 5 #TODO replace this with more appropriate trigger
		}
		modifier = { 
			factor = 5
			relative_income_to_liege = 0.5
		}
	}
	success = {
		is_councillor = yes
	}
	abort = {
		OR = {
			liege = { spouse = { character = ROOT } }
			trait = in_hiding
			independent = yes
			trait = incapable
			NOR = { 
				can_hold_title = job_chancellor
				can_hold_title = job_marshal
				can_hold_title = job_treasurer
				can_hold_title = job_spymaster
				can_hold_title = job_spiritual
			}
		}
	}
	abort_effect = {
	}
	effect = {
		opinion = {
			who = liege
			modifier = opinion_ambition_fulfilled
			months = 24
		}
	}
	
	is_job_ambition=yes
}
