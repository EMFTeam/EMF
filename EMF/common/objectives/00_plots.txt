# Gain one of the liege's landed titles
plot_gain_title = {
	type = liege_titles
	military_plot = yes

	# Plotter scope
	potential = {
		always = no # Disabled for being too similar to the factions

		prisoner = no
		is_adult = yes
		is_ruler = yes
		independent = no
		OR = {
			is_landed = yes
			is_patrician = yes
		}
		OR = {
			primary_title = { higher_tier_than = BARON }
			is_patrician = yes
		}
		is_incapable = no
		NOT = { trait = imbecile }
		OR = {
			emf_laws_crownlaw_title_not_high_ca_plus = yes
			# Does not have a crownlaw title
			emf_laws_no_crown_law_title = yes
			# Plot against top liege or king
			OR = {
				top_liege = {
					character = ROOT
				}
				liege = {
					character = ROOT
					primary_title = {
						tier = KING
					}
				}
			}
		}
	}

	# Target scope
	allow = {
		holder_scope = {
			NOR = {
				is_married = FROM
				FROM = { is_consort = PREV }
			}
			NOT = { reverse_has_truce = FROM  }
		}
		OR = {
			FROM = {
				OR = {
					is_feudal = yes
					is_tribal = yes
					is_republic = yes
					is_theocracy = yes
				}
			}
			lower_tier_than = KING
		}
		OR = {
			claimed_by = FROM
			AND = {
				is_primary_holder_title = yes
				FROM = {
					any_pretender_title = {
						title = ROOT
					}
				}
			}
			AND = {
				is_primary_holder_title = no
				is_primary_holder_title_tier = yes
				is_titular = no
				holder_scope = {
					overlord_of = FROM
				}
				FROM = {
					primary_title = {
						kingdom = {
							title = PREVPREVPREV
						}
					}
				}
			}
		}
	}

	chance = {
		factor = 50

		modifier = {
			factor = 0.0
			FROM = {
				ai = yes
				war = yes
			}
		}

				FROM = { trait = envious }

			}
		}
		modifier = {
			factor = 0
			FROM = { opinion = { who = LIEGE value = 25 } }
			NOT = {
				OR = {
					FROM = { trait = envious }
					FROM = { trait = deceitful }

				}
			}
		}

			NOT = {
				OR = {
					FROM = { trait = envious }

		}
		modifier = {

				FROM = { trait = deceitful }

# Get a character killed
plot_kill_character = {
	type = realm_characters_plus # All characters in the same realm, plus rivals and relatives of self, parents, children and spouse, and quest target
	intrigue_plot = yes
	murder_plot = yes
	society_plot = {
		ROOT = {
			society_member_of = the_assassins
			society_rank >= 2
		}
		OR = {
			FROM = {
				NOR = {
					religion = ismaili
					secret_religion = ismaili
				}
			}
			ROOT = {
				has_quest = quest_the_assassins_assassination
				quest_target = { character = FROM }
			}
		}
	}

	# Plotter scope
	potential = {
		prisoner = no
		is_adult = yes
		is_incapable = no
		emf_isolated_character = no

		OR = {
			ai = no # Human
			NOR = {
				trait = honest
				trait = kind
				trait = just
			}
		}

		OR = {
			ai = no # Human

			# Has rivals
			num_of_rivals = 1

			# A spymaster
			has_job_title = job_spymaster

			# The target of an adventurer
			has_any_opinion_modifier = opinion_targeted_by_adventurer

			# Pretender to a title
			any_pretender_title = {
				always = yes
			}

			# Heir to a title
			any_heir_title = {
				always = yes
			}

			# Character's child is heir or pretender to a title that I don't hold
			any_child = {
				OR = {
					any_pretender_title = {
						NOT = { holder = ROOT }
						NOT = { current_heir = { character = ROOT } }
					}
					any_heir_title = {
						NOT = { holder = ROOT }
					}
				}
			}

			# Spouse is pretender to a title
			any_spouse = {
				any_pretender_title = {
					NOT = { holder = ROOT }
					NOT = { current_heir = { character = ROOT } }
				}
				any_heir_title = {
					NOT = { holder = ROOT }
				}
			}

			# spouse has a lover...
			AND = {
				NOT = { trait = homosexual }
				any_spouse = {
					has_lover = yes
				}
			}

			# Is nuts...
			trait = lunatic
			trait = possessed
		}
	}

	target_potential = {
		NOT = { is_married = FROM } # This case is covered by another plot
		emf_isolated_character = no
	}

	# Target allow trigger for when players target a specific character in the GUI
	player_allow = {
		NOT = { is_child_of = FROM } # Not my own children
		is_within_diplo_range = FROM
		NOT = { has_blood_oath_with = FROM }
		assassination_interaction_trigger = yes
	}

	# Target character scope
	allow = {
		NOT = { is_child_of = FROM } # Not my own children
