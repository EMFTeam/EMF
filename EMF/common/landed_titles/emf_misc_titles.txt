

e_bulgaria = {
	color={ 20 70 20 }
	color2={ 255 255 255 }
	
	capital = 499 # Tyrnovo
	short_name = yes
	culture = bulgarian
	
	# Creation/Usurp Trigger
	allow = {
		or = {
			culture = bulgarian
			culture = serbian
		}
		realm_size = 120
		war = no
		ruled_years = 10
		is_adult = yes
	}
	gain_effect = {
		if = {
			limit = { e_bulgaria = { is_titular = yes } }
			k_bulgaria = { de_jure_liege = e_bulgaria }
			k_serbia = { de_jure_liege = e_bulgaria }
		}
	}
}
