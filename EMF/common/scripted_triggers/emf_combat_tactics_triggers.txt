# Harass
emf_harass_tactic_troop_requirements = {
	OR = {
		light_cavalry = 0.01
		camel_cavalry = 0.01
		light_infantry = 0.01
	}
	OR = {
		light_infantry = 0.35
		light_cavalry = 0.35
		camel_cavalry = 0.35
		NOR = {
			archers = 0.5
			horse_archers = 0.5
			pikemen = 0.5
			heavy_infantry = 0.5
		}
	}
	OR = {
		light_troops = {
			who = light_infantry
			value = 0.35
		}
		light_troops = {
			who = light_cavalry
			value = 0.35
		}
		light_troops = {
			who = camel_cavalry
			value = 0.35
		}
		NOR = {
			light_troops = { # Prefer Volley
				who = archers
				value = 0.35
			}
			light_troops = { # Prefer Swarm
				who = horse_archers
				value = 0.35
			}
		}
	}
}

# Charge
emf_melee_charge_tactic_troop_requirements = {
	OR = {
		knights = 0.01
		light_cavalry = 0.01
		camel_cavalry = 0.01
		horse_archers = 0.01
		war_elephants = 0.01
	}
	NAND = {
		OR = {
			light_infantry = 0.60
			heavy_infantry = 0.30
			pikemen = 0.30
			archers = 0.30
		}
		NOR = {
			knights = 0.16
			war_elephants = 0.03
			AND = {
				horse_archers = 0.30
				OR = {
					knights = 0.10
					NOT = { light_cavalry = 0.10 }
					NOT = { camel_cavalry = 0.10 }
				}
			}
			AND = {
				light_cavalry = 0.30
				OR = {
					knights = 0.10
					NOT = { camel_cavalry = 0.10 }
					NOT = { horse_archers = 0.10 }
				}
			}
			camel_cavalry = 0.30
		}
	}
}
