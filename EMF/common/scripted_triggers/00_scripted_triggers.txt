#Used with character interaction to replace merc leader. DO NOT REMOVE
can_be_mercenary_leader_potential_trigger = {
	prisoner = no
	is_ruler = no
	
	NOR = { 
		trait = incapable
		trait = in_hiding
		trait = monk
	}
	same_realm = FROM
	OR = {
		is_female = no
		AND = {
			emf_feminist_religion_trigger = yes
			FROM = { emf_feminist_religion_trigger = yes }
		}
		FROM = { primary_title = { succ_law_title = { has_law = true_cognatic_succession } } }
	}
}

#Used with character interaction to swap council job. DO NOT REMOVE
change_council_job_potential_trigger = {
	liege = { character = FROM }
	NOT = {
		AND = {
			has_job_title = job_spiritual
			liege = { religion_group = muslim }
			religion = FROM
		}
	}
	OR = {
		AND = {
			OR = {
				has_minor_title = title_councilmember_king
				has_minor_title = title_councilmember_emperor
			}
			OR = {
				can_be_chancellor_trigger = yes
				can_be_marshal_trigger = yes
				can_be_treasurer_trigger = yes
				can_be_spymaster_trigger = yes
				AND = {
					can_be_spiritual_trigger = yes
					ROOT = { NOT = { religion_group = muslim } }
				}
			}
		}
		AND = {
			OR = {
				has_job_title = job_chancellor
				has_job_title = job_marshal
				has_job_title = job_treasurer
				has_job_title = job_spymaster
				has_job_title = job_spiritual
			}
			liege = {
				OR = {
					job_chancellor = {
						NOT = { character = ROOT }
						can_swap_job_title = ROOT
					}
					job_marshal = {
						NOT = { character = ROOT }
						can_swap_job_title = ROOT
					}
					job_treasurer = {
						NOT = { character = ROOT }
						can_swap_job_title = ROOT
					}
					job_spymaster = {
						NOT = { character = ROOT }
						can_swap_job_title = ROOT
					}
					job_spiritual = {
						NOT = { character = ROOT }
						can_swap_job_title = ROOT
					}
				}
			}
		}
	}
}

can_be_chancellor_trigger = {
	is_adult = yes
	prisoner = no
	NOT = { trait = incapable }
	NAND  = {
		trait = in_hiding
		ai = no
	}
	OR = {
		is_female = no
		trait = horse
		AND = {
			emf_feminist_religion_trigger = yes
			liege = { religion = ROOT }
		}
		AND = {
			culture = basque
			liege = {
				culture = basque
				primary_title = {
					succ_law_title = { NOT = { has_law = agnatic_succession } }
				}
			}
		}
		liege = {
			primary_title = {
				succ_law_title = { has_law = true_cognatic_succession }
			}
		}
		AND = {
			liege = {
				primary_title = {
					OR = {
						has_law = status_of_women_2
						has_law = status_of_women_3
					}
				}
			}
			OR = {
				liege = {
					OR = {
						dynasty = ROOT
						is_close_relative = ROOT
						any_spouse = { character = ROOT }
						any_consort = { character = ROOT }
					}
				}
				is_landed = yes
				trait = nun
			}
		}
		has_law = status_of_women_4
		AND = {
			has_global_flag = emf_gender_eq_council
			NOT = { liege = { religion_group = muslim } }
		}
		has_character_flag = special_chancellor
	}
}

can_be_spiritual_trigger = {
	is_adult = yes
	prisoner = no
	NOT = { trait = incapable }
	NAND  = {
		trait = in_hiding
		ai = no
	}
	liege = { religion = ROOT }
	OR = {
		is_female = no
		emf_feminist_religion_trigger = yes
		religion_allows_female_temple_holders = yes
		AND = {
			has_global_flag = emf_gender_eq_council
			NOT = { liege = { religion_group = muslim } }
		}
		has_character_flag = special_spiritual
	}
	OR = {
		NOT = { religion = hindu }
		trait = brahmin
	}
	OR = {
		is_theocracy = yes
		AND = {
			liege = { is_tribal = yes }
			is_ruler = no
		}
		liege = { religion_group = muslim }
		AND = {
			clan = yes
			liege = {
				is_nomadic = yes
			}
		}
		AND = {
			has_autocephaly = no
			is_ruler = no
		}
		AND = {
			has_autocephaly = yes
			is_ruler = no
			NOT = { liege = { any_vassal = { is_theocracy = yes } } }
		}
		trait = monk
		trait = nun
	}
	OR = {
		NOT = { religion_group = muslim }
		NOT = { liege = { religion_group = muslim } }
		trait = detached_priest
		trait = martial_cleric
		trait = scholarly_theologian
		trait = mastermind_theologian
	}
}
