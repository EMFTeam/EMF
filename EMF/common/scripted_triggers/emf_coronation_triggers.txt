emf_uses_byzantine_coronation = {
	primary_title = {
		emf_is_roman_empire = yes
	}
	OR = {
		religion = orthodox
		religion = bogomilist
		religion = monothelite
		religion = iconoclast
		religion = hellenic_pagan
		religion = hellenic_pagan_reformed
	}
	is_feudal = yes
	controls_religion = no
	religion_head = { is_alive = yes }
}

emf_uses_hre_coronation = {
	primary_title = {
		OR = {
			emf_is_hre = yes
			title = e_roman_empire
		}
	}
	OR = {
		religion = catholic
		religion = cathar
		religion = manichean
		religion = fraticelli
		AND = {
			religion = bogomilist
			NOT = { primary_title = { title = e_roman_empire } }
		}
	}
	is_heretic = no
	is_feudal = yes
	rightful_religious_head_scope = {
		controls_religion = yes
		NOT = { character = ROOT }
	}
}

emf_common_coronation_requirements = {
	higher_real_tier_than = DUKE
	is_feudal = yes
	is_landed = yes
	holy_order = no
	OR = {
		real_tier = EMPEROR
		emf_imperial_government = no
	}
	is_vice_royalty = no
}

emf_uses_furry_coronation = {
	has_dlc = "Holy Fury"
	OR = { 
		religion = catholic
		religion = fraticelli
	}
	emf_common_coronation_requirements = yes
}

emf_uses_native_coronation = {
	NAND = {
		has_dlc = "Holy Fury"
		OR = { 
			religion = catholic
			religion = fraticelli
		}
	}
	emf_common_coronation_requirements = yes
}

# does THIS need to get coronated somehow? use when checking, e.g., for uncrowned state as a weakness
emf_needs_coronation = {
	emf_common_coronation_requirements = yes
	emf_is_uncrowned = yes
}

emf_is_crowned = {
	custom_tooltip = {
		text = EMF_IS_CROWNED_CTT
		hidden_trigger = {
			OR = {
				trait = emf_crowned
				trait = emf_crowned_by_pope
				trait = strong_basileus
				trait = weak_basileus
				trait = crowned_by_priest
				trait = crowned_by_bishop
				trait = crowned_by_pope
				trait = crowned_by_myself
			}
		}
	}
}

emf_is_uncrowned = {
	custom_tooltip = {
		text = EMF_IS_UNCROWNED_CTT
		hidden_trigger = {
			emf_is_crowned = no
		}
	}
}
