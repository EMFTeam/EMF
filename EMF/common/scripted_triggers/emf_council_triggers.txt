
emf_council_chancellor_trigger = {
	is_adult = yes
	prisoner = no
	NOT = { trait = incapable }
	NOT = { trait = in_hiding }
	OR = {
		is_female = no
		trait = horse
		AND = {
			emf_core_feminist_religion_trigger = yes
			liege = { religion = PREV }
		}
		AND = {
			culture = basque
			liege = {
				culture = basque
				primary_title = {
					succ_law_title = { NOT = { has_law = agnatic_succession } }
				}
			}
		}
		liege = {
			primary_title = {
				succ_law_title = { has_law = true_cognatic_succession }
			}
		}
		AND = {
			liege = {
				primary_title = {
					OR = {
						has_law = status_of_women_2
						has_law = status_of_women_3
					}
				}
			}
			OR = {
				liege = {
					OR = {
						dynasty = PREV
						is_close_relative = PREV
						any_spouse = { character = PREVPREV }
						any_consort = { character = PREVPREV }
					}
				}
				is_landed = yes
				trait = nun
				
			}
		}
		has_law = status_of_women_4
		AND = {
			has_global_flag = emf_gender_eq_council
			NOT = { liege = { religion_group = muslim } }
		}
		has_character_flag = special_chancellor
	}
}

emf_council_spiritual_trigger = {
	is_adult = yes
	prisoner = no
	NOT = { trait = incapable }
	NOT = { trait = in_hiding }
	liege = { religion = ROOT }
	OR = {
		is_female = no
		emf_core_feminist_religion_trigger = yes
		religion_allows_female_temple_holders = yes
		AND = {
			has_global_flag = emf_gender_eq_council
			NOT = { liege = { religion_group = muslim } }
		}
		has_character_flag = special_spiritual
	}
	OR = {
		NOT = { religion = hindu }
		trait = brahmin
	}
	OR = {
		is_theocracy = yes
		AND = {
			liege = { is_tribal = yes }
			is_ruler = no
		}
		liege = { religion_group = muslim }
		AND = {
			clan = yes
			liege = {
				is_nomadic = yes
			}
		}
		AND = {
			has_autocephaly = no
			is_ruler = no
		}
		AND = {
			has_autocephaly = yes
			is_ruler = no
			NOT = { liege = { any_vassal = { is_theocracy = yes } } }
		}
		trait = monk
		trait = nun
	}
	OR = {
		NOT = { religion_group = muslim }
		NOT = { liege = { religion_group = muslim } }
		trait = detached_priest
		trait = martial_cleric
		trait = scholarly_theologian
		trait = mastermind_theologian
	}
}
