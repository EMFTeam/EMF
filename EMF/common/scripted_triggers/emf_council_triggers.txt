
emf_council_chancellor_trigger = {
	is_adult = yes
	prisoner = no
	NOT = { trait = incapable }
	NOT = { trait = in_hiding }
	OR = {
		is_female = no
		AND = {
			OR = {
				religion = cathar
				religion = messalian
				religion = bogomilist
				religion = mazdaki
			}
			liege = { religion = PREV }
		}
		AND = {
			culture = basque
			liege = {
				culture = PREV
				succ_law_title = {
					NOT = { has_law = agnatic_succession }
				}
			}
		}
		liege = {
			succ_law_title = {
				has_law = true_cognatic_succession
			}
		}
		AND = {
			has_global_flag = emf_gender_eq_council
			NOT = { liege = { religion_group = muslim } }
		}
		has_character_flag = special_chancellor
	}
}

emf_council_spiritual_trigger = {
	is_adult = yes
	prisoner = no
	NOT = { trait = incapable }
	NOT = { trait = in_hiding }
	liege = { religion = ROOT }
	OR = {
		is_female = no
		religion = cathar
		religion = bogomilist
		religion = messalian
		religion = mazdaki
		AND = {
			religion_group = pagan_group
			NOT = { religion = aztec }
			NOT = { religion = aztec_reformed }
			NOT = { religion = tengri_pagan }
			NOT = { religion = tengri_pagan_reformed }
		}
		AND = {
			has_global_flag = emf_gender_eq_council
			NOT = { liege = { religion_group = muslim } }
		}
		has_character_flag = special_spiritual
	}
	OR = {
		NOT = { religion = hindu }
		trait = brahmin
	}
	OR = {
		is_theocracy = yes
		trait = monk
		trait = nun
		AND = {
			liege = { is_tribal = yes }
			is_ruler = no
		}
		liege = { religion_group = muslim }
		AND = {
			is_ruler = no
			has_autocephaly = no
		}
	}
	OR = {
		NOT = { religion_group = muslim }
		NOT = { liege = { religion_group = muslim } }
		trait = detached_priest
		trait = martial_cleric
		trait = scholarly_theologian
		trait = mastermind_theologian
	}
}
