
emf_empires_mongol_target_prov_trigger = {
	
	owner = { not = { same_realm = ROOT } } # Border province

	or = {
		# Can always conquer within core region
		and = {
			not = { has_global_flag = emf_mongols_in_persia }
			region = emf_region_mongol_core
		}

		# But must reach "persia" before unlocking full [scripted] conquest zone
		and = {
			has_global_flag = emf_mongols_in_persia
			region = emf_region_mongol_full
		}
	}

	owner = {
		top_liege = {
			nor = {
				# Respect truce
				reverse_has_truce = ROOT

				# Respect suzerainty
				ROOT = { pays_tribute_to = PREV }
			}
		}
	}
}
