
###############################################
#### AUTOMATIC EMPIRE DISINTEGRATION (AED) ####
###############################################

emf_empires_aed_valid_title_trigger = {
	tier = EMPEROR
	temporary = no
	rebel = no
	OR = {
		AND = {
			ROOT = { NOT = { realm_size = 100 } }
			NOR = {
				title = e_hip
				title = e_rebels
				title = e_pirates
				title = e_hre
				title = e_byzantium
				title = e_roman_empire
				title = e_latin_empire
				title = e_timurids
				title = e_mexikha
				title = e_seljuk_turks # Only until we convert title destruction to be time-delayed after fulfilling AED conditions
				AND = {
					OR = {
						title = e_golden_horde
						title = e_il-khanate
						title = e_chagatai
						title = e_mongol_empire
					}
					ROOT = { religion = tengri_pagan } # Hack to simulate time-delayed title destruction
				}
			}
		}
		AND = {
			ROOT = { NOT = { realm_size = 150 } }
			title = e_hre
		}
		AND = {
			OR = {
				title = e_byzantium
				title = e_latin_empire
			}
			NOT = { c_byzantion = { owner_under_ROOT = yes } }
		}
	}
}

###############################################
####       RESTORATION OF BYZANTIUM        ####
###############################################

# THIS = character whose culture/religion is to be evaluated
emf_empires_byz_valid_culture_religion_trigger = {
	OR = {
		culture_group = byzantine
		AND = {
			ai = no
			culture_group = south_slavic
		}
	}
	OR = {
		custom_tooltip = {
			text = emf_empires_ctt_byz_valid_religion
			hidden_tooltip = {
				OR = {
					religion = orthodox
					religion = bogomilist
					religion = paulician
					religion = monothelite
					religion = iconoclast
				}
			}
		}
		AND = {
			ai = no
			custom_tooltip = {
				text = emf_empires_ctt_byz_valid_religion_player
				hidden_tooltip = {
					OR = {
						religion = nestorian
						religion = messalian
						religion = miaphysite
						religion = monophysite
					}
				}
			}
		}
	}
}

emf_empires_mongol_core_prov_trigger = {
	owner = { NOT = { same_realm = ROOT } } # Border province
	region = emf_region_mongol_core
	owner = {
		top_liege = {
			NOR = {
				# Respect truce
				reverse_has_truce = ROOT
				# Respect suzerainty
				ROOT = { pays_tribute_to = PREV }
				# Respect alliances
				is_allied_with = ROOT
				# Respect NAPs
				has_non_aggression_pact_with = ROOT
			}
		}
	}
}
