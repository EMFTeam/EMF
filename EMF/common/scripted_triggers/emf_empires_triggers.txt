
# THIS = character being evaluated for fitting the Byzantium-restoration requirements
emf_empires_byz_valid_culture_religion_trigger = {
	OR = {
		culture_group = byzantine
		AND = {
			ai = no
			culture_group = south_slavic
		}
	}
	OR = {
		custom_tooltip = {
			text = emf_empires_ctt_byz_valid_religion
			hidden_tooltip = {
				OR = {
					religion = orthodox
					religion = bogomilist
					religion = paulician
					religion = monothelite
					religion = iconoclast
				}
			}
		}
		AND = {
			ai = no
			custom_tooltip = {
				text = emf_empires_ctt_byz_valid_religion_player
				hidden_tooltip = {
					OR = {
						religion = nestorian
						religion = messalian
						religion = miaphysite
						religion = monophysite
					}
				}
			}
		}
	}
}

emf_empires_mongol_core_prov_trigger = {
	owner = { NOT = { same_realm = ROOT } } # Border province
	region = emf_region_mongol_core
	owner = {
		top_liege = {
			NOR = {
				# Respect truce
				reverse_has_truce = ROOT
				# Respect suzerainty
				ROOT = { pays_tribute_to = PREV }
				# Respect alliances
				is_allied_with = ROOT
				# Respect NAPs
				has_non_aggression_pact_with = ROOT
			}
		}
	}
}
