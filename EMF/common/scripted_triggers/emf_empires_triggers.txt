
# THIS = character being evaluated for fitting the Byzantium-restoration requirements
emf_empires_byz_valid_culture_religion_trigger = {
	OR = {
		culture_group = byzantine
		AND = {
			ai = no
			culture_group = south_slavic
		}
	}
	OR = {
		religion = orthodox
		religion = bogomilist
		religion = paulician
		religion = monothelite
		religion = iconoclast
		AND = {
			ai = no
			OR = {
				religion = nestorian
				religion = messalian
				religion = miaphysite
				religion = monophysite
			}
		}
	}
}

# THIS = scope to consider for being same religion
emf_empires_no_empire_same_religion_trigger = {
	custom_tooltip = {
		text = emf_empires_ctt_no_empire_same_religion_trigger
		hidden_tooltip = {
			NOT = {
				any_independent_ruler = {
					tier = EMPEROR
					is_nomadic = no
					realm_size = 100
					religion = PREV
					any_demesne_title = {
						tier = EMPEROR
						is_titular = no
					}
				}
			}
		}
	}
}

emf_empires_mongol_core_prov_trigger = {
	owner = { NOT = { same_realm = ROOT } } # Border province
	region = emf_region_mongol_core
	owner = {
		top_liege = {
			NOR = {
				# Respect truce
				reverse_has_truce = ROOT
				# Respect suzerainty
				ROOT = { pays_tribute_to = PREV }
				# Respect alliances
				is_allied_with = ROOT
				# Respect NAPs
				has_non_aggression_pact_with = ROOT
			}
		}
	}
}
