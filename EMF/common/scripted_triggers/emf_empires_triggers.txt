
# For de jure empire formation/restoration via decision
emf_empires_create_empire_basic_trigger = {
	war = no
	is_adult = yes
	prisoner = no
	has_regent = no
	custom_tooltip = {
		text = emf_empires_ctt_can_afford_dj_empire_creation
		hidden_tooltip = {
			OR = {
				wealth = 600
				AND = {
					ROOT = { ai = yes }
					wealth = 0
				}
			}
		}
	}
	prestige = 1000
}


###############################################
#### AUTOMATIC EMPIRE DISINTEGRATION (AED) ####
###############################################

# THIS = empire title
emf_empires_aed_valid_title_trigger = {
	tier = EMPEROR
	is_landless_type_title = no
	is_nomadic = no
	temporary = no
	rebel = no
	NOR = { # Never, ever disintegrate these specific titles with AED:
		title = e_timurids
		title = e_mexikha
		title = e_seljuk_turks
	}
	OR = {
		# Minimum 100 realm_size rule that applies by default to almost all empires:
		NOR = {
			# Titles exempt even from any realm size rule:
			title = e_byzantium
			title = e_latin_empire
			# Titles which only use 100-size rule when they have at least High CA:
			AND = {
				OR = {
					title = e_hre
					title = e_hre_french
					title = e_roman_empire
				}
				OR = {
					has_law = crown_authority_3
					has_law = crown_authority_4
				}
			}
			owner = { realm_size = 100 }
		}
		# Empires with higher min. realm size requirements, unless they have at least High CA:
		AND = {
			NOR = {
				has_law = crown_authority_3
				has_law = crown_authority_4
			}
			OR = {
				AND = {
					OR = {
						title = e_hre
						title = e_hre_french
					}
					OR = {
						NOT = { owner = { realm_size = 200 } }
						AND = {
							NOT = { owner = { realm_size = 300 } }
							has_global_flag = SWMH
						}
					}
				}
				AND = {
					title = e_roman_empire
					OR = {
						NOT = { owner = { realm_size = 300 } }
						AND = {
							NOT = { owner = { realm_size = 400 } }
							has_global_flag = SWMH
						}
					}
				}
			}
		}
		# Empires which disintegrate solely based upon whether they control the Queen of Cities:
		AND = {
			OR = {
				title = e_byzantium
				title = e_latin_empire
			}
			NOT = { owner = { c_byzantion = { owner_under_PREV = yes } } }
		}
	}
}

###############################################
####       RESTORATION OF BYZANTIUM        ####
###############################################

# THIS = character whose culture/religion is to be evaluated
emf_empires_byz_valid_culture_religion_trigger = {
	OR = {
		culture_group = byzantine
		AND = {
			ai = no
			culture_group = south_slavic
		}
	}
	OR = {
		custom_tooltip = {
			text = emf_empires_ctt_byz_valid_religion
			hidden_tooltip = {
				OR = {
					religion = orthodox
					religion = bogomilist
					religion = paulician
					religion = monothelite
					religion = iconoclast
				}
			}
		}
		AND = {
			ai = no
			custom_tooltip = {
				text = emf_empires_ctt_byz_valid_religion_player
				hidden_tooltip = {
					OR = {
						religion = nestorian
						religion = messalian
						religion = miaphysite
						religion = monophysite
					}
				}
			}
		}
	}
}

emf_empires_mongol_core_prov_trigger = {
	owner = { NOT = { same_realm = ROOT } } # Border province
	region = emf_region_mongol_core
	owner = {
		top_liege = {
			NOR = {
				# Respect truce
				reverse_has_truce = ROOT
				# Respect suzerainty
				ROOT = { pays_tribute_to = PREV }
				# Respect alliances
				is_allied_with = ROOT
				# Respect NAPs
				has_non_aggression_pact_with = ROOT
			}
		}
	}
}
