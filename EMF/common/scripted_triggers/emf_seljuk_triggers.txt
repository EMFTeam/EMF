# -*- ck2.scripted_triggers -*-

emf_seljuk_dynasty = {
	dynasty = 613
}

emf_seljuk_valid_initial_invasion_target_ruler = {
	independent = yes # Don't declare war on revolters
	NOR = {
		is_primary_war_defender = yes
		is_nomadic = yes
		is_tribal = yes
	}
}

emf_seljuk_can_become_invader_base = {
	ai = yes
	is_alive = yes
	is_female = no
	is_adult = yes
	practical_age < 55
	prisoner = no
	is_incapable = no
}

emf_seljuk_can_become_invader = {
	emf_seljuk_can_become_invader_base = yes
	martial > 7
	is_weak_trigger = no
	is_dumb_trigger = no
	OR = {
		character = 3030 # Seljuk
		character = 3038 # Tughril-Beg Seljuk
		character = 3040 # Alp Arslan Seljuk
		trait = genius
		trait = quick
		trait = strong
		trait = brilliant_strategist
		calc_true_if = {
			amount = 2
			martial >= 12
			martial >= 16
			diplomacy >= 12
			diplomacy >= 16
			stewardship >= 16
			trait = ambitious
			trait = shrewd
			trait = robust
			trait = master_warrior
			trait = master_diplomat
		}
	}
}

# Main invasion mission: conquer and form de jure Persia, conquer further until realm borders or is within de jure Armenia
emf_seljuk_invasion_mission = {
	independent = yes
	trait = emf_seljuk_invader
	NOT = { has_global_flag = emf_seljuk_invasion_done }
}

emf_seljuk_invasion_mission_done = {
	OR = {
		AND = {
			emf_seljuk_dynasty = yes
			OR = {
				primary_title = {
					tier = EMPEROR
					is_de_jure_title = yes
				}
				has_random_dejure = yes
			}
			any_realm_province = {
				owner = { same_realm = PREVPREV }
				has_flag = emf_capital_connected
				region = emf_region_seljuk_invasion_tier_3
			}
		}
		k_rum = {
			OR = {
				owner = { emf_seljuk_dynasty = yes }
				any_previous_holder = { emf_seljuk_dynasty = yes }
			}
		}
	}
}

emf_seljuk_invader_is_valid = {
	year < 1100
	emf_seljuk_dynasty = yes
	independent = yes
	OR = {
		religion = sunni
		has_game_rule = { name = railroading value = passive }
	}
	is_feudal = yes
	is_landed = yes
	higher_tier_than = BARON
	mercenary = no
	holy_order = no
	ai = yes
	OR = {
		capital_scope = {
			OR = {
				region = emf_region_seljuk_invasion_tier_1
				region = emf_region_seljuk_invasion_tier_2
			}
		}
		any_realm_province = {
			owner = { same_realm = PREVPREV }
			has_flag = emf_capital_connected
			OR = {
				region = emf_region_seljuk_invasion_tier_1
				region = emf_region_seljuk_invasion_tier_2
				any_neighbor_province = {
					OR = {
						region = emf_region_seljuk_invasion_tier_1
						region = emf_region_seljuk_invasion_tier_2
					}
				}
			}
		}
	}
}

emf_seljuk_invasion_valid_target_province = {
	owner = { top_liege = { emf_seljuk_invasion_valid_target = yes } }
}

emf_seljuk_invasion_valid_target = {
	liege = { character = PREV } #independent = yes and any_independent_ruler exclude revolts
	NOR = {
		character = ROOT
		reverse_has_truce = ROOT
		AND = {
			ai = no # Will break NAPs and alliances with AI
			OR = {
				is_allied_with = ROOT
				has_non_aggression_pact_with = ROOT
			}
		}
		any_war = {
			OR = {
				AND = {
					any_attacker = { character = PREV }
					any_defender = { character = ROOT }
				}
				AND = {
					any_attacker = { character = ROOT }
					any_defender = { character = PREV }
				}
			}
		}
		AND = {
			ai = no
			OR = {
				reverse_has_opinion_modifier = { who = ROOT modifier = opinion_special_truce }
				ROOT = { pays_tribute_to = PREV }
			}
		}
	}
}

emf_seljuk_dynasty_member_has_enough_kids = {
	OR = {
		AND = {
			has_game_rule = { name = railroading value = default }
			OR = {
				# At least 3 dynastic sons
				any_child = {
					count = 3
					emf_seljuk_dynasty = yes
					is_female = no
				}
				AND = {
					# Exactly two dynastic sons & at least 7 kids total
					any_child_even_if_dead = { count = 7 }
					any_child = {
						count = 2
						emf_seljuk_dynasty = yes
						is_female = no
					}
					NOT = {
						any_child = {
							count = 3
							emf_seljuk_dynasty = yes
							is_female = no
						}
					}
				}
				AND = {
					# Exactly one dynastic son & at least 10 kids total
					any_child_even_if_dead = { count = 10 }
					any_child = {
						emf_seljuk_dynasty = yes
						is_female = no
					}
					NOT = {
						any_child = {
							count = 2
							emf_seljuk_dynasty = yes
							is_female = no
						}
					}
				}
				# Never more than 13 kids total
				any_child_even_if_dead = { count = 13 }
			}
		}
		AND = {
			has_game_rule = { name = railroading value = passive }
			OR = {
				# At least 2 dynastic sons
				any_child = {
					count = 2
					emf_seljuk_dynasty = yes
					is_female = no
				}
				AND = {
					# Exactly one dynastic son & at least 5 kids total
					any_child_even_if_dead = { count = 5 }
					any_child = {
						emf_seljuk_dynasty = yes
						is_female = no
					}
					NOT = {
						any_child = {
							count = 2
							emf_seljuk_dynasty = yes
							is_female = no
						}
					}
				}
				# Never more than 7 kids total
				any_child_even_if_dead = { count = 7 }
			}
		}
	}
}
