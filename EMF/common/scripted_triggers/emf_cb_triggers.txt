emf_cb_can_use_gui_trigger = {
	ROOT = {
		custom_tooltip = {
			text = emf_cb_ctt_no_peace_pledge
			hidden_tooltip = { NOT = { has_character_modifier = emf_peace_pledge } }
		}
	}
}
emf_cb_can_use_trigger = {
	ROOT = {
		OR = {
			ai = no
			AND = {
				NAND = {
					has_landed_title = e_mongol_empire
					NOR = {
						has_global_flag = emf_mongols_defeated
						has_character_flag = emf_horde_invasion_pass
					}
				}
				NOT = { has_character_flag = planning_coronation }
				NOT = { has_character_flag = emf_planning_tournament }
				NOT = { has_character_flag = emf_holding_tournament }
			}
		}
	}
}
emf_cb_can_use_title_trigger = {
	OR = {
		FROM = { ai = no }
		ROOT = { ai = no }
		ROOT = { is_merchant_republic = yes }
		FROM = {
			NAND = {
				is_merchant_republic = yes
				OR = {
					capital_scope = { county = { title = PREVPREVPREV } }
					capital_holding = { title = PREVPREV }
					primary_title = { title = PREVPREV }
					succ_law_title = { title = PREVPREV }
				}
			}
		}
	}
	FROM = {
		NAND = {
			is_nomadic = yes
			OR = {
				primary_title = { title = PREVPREV }
				clan_title = { title = PREVPREV }
			}
		}
	}
	holy_order = no
	mercenary = no
	emf_cb_can_use_magyar_title_trigger = yes
}
emf_cb_can_use_de_jure_title_trigger = {
	emf_cb_can_use_sahara_title_trigger = yes
	emf_cb_can_use_magyar_title_trigger = yes
	FROM = {
		OR = {
			ai = no
			is_merchant_republic = no
			ROOT = { is_merchant_republic = yes }
			ROOT = { ai = no }
			NOT = { capital_holding = { de_jure_liege_or_above = PREVPREV } }
		}
	}
}

# ROOT = claimant, THIS = claimed title
emf_cb_can_press_claim = {
	ROOT = {
		OR = {
			is_ruler = no
			primary_title = {
				OR = {
					is_primary_type_title = no # Static Mercs, the Pope, Holy Orders, etc
					NOT = { lower_tier_than = PREVPREV }
				}
			}
		}
	}
	OR = {
		ROOT = { has_strong_claim = PREV }
		AND = {
			ROOT = { has_weak_claim = PREV }
			emf_cb_is_weak_claim_pressable = yes
		}
	}
}

emf_cb_is_weak_claim_pressable = {
	OR = {
		is_pretender = yes # CAUTION: checks vs ROOT implicitly!
		is_contested = yes
		holder_scope = { has_regent = yes }
		holder_scope = {
			trait = uncrowned
			higher_real_tier_than = DUKE # JIC
		}
		AND = {
			succ_law_title = { has_law = agnatic_succession }
			ROOT = { is_female = no }
			holder_scope = { is_female = yes }
		}
		AND = {
			succ_law_title = { has_law = cognatic_succession }
			holder_scope = { is_female = yes }
		}
		AND = {
			succ_law_title = { has_law = enatic_succession }
			ROOT = { is_female = yes }
			holder_scope = { is_female = no }
		}
		AND = {
			succ_law_title = { has_law = enatic_cognatic_succession }
			holder_scope = { is_female = no }
		}
	}
}

# Shared `can_use` CB code for religious CBs
emf_cb_can_use_religious = {
	ROOT = {
		is_nomadic = no
		NOT = { same_realm = FROM }
		NOR = {
			mercenary = yes
			pacifist = yes
			religion = buddhist
		}
		OR = { # Allow reformed pagans
			NOT = { religion_group = pagan_group }
			is_reformed_religion = yes
		}
		OR = { # Hindus can only wage holy war on non-Indian rulers
			NOT = { religion = hindu }
			FROM = { NOT = { religion_group = indian_group } }
		}
		OR = {
			NOT = { religion_group = FROM }
			is_heresy_of = FROM
			is_parent_religion = FROM
			AND = {
				OR = {
					religion_group = muslim
					religion_group = pagan_group
					religion = paulician
				}
				NOT = { religion = FROM }
			}
		}
		NOT = { is_reformed_religion = FROM }
		top_liege = {
			religion_group = ROOT
			NOT = { religion = FROM }
			OR = {
				religion = ROOT
				AND = {
					is_heretic = no
					ROOT = { is_heretic = no }
				}
			}
			OR = {
				NOT = { religion_group = pagan_group }
				religion = ROOT
			}
		}
	}
}

# Standard `is_valid` CB code for religious CBs
emf_cb_is_valid_religious = {
	ROOT = {
		NOT = { same_realm = FROM }
		NOR = {
			mercenary = yes
			pacifist = yes
			religion = buddhist
		}
		OR = {
			NOT = { religion_group = FROM }
			is_heresy_of = FROM
			is_parent_religion = FROM
			AND = {
				OR = {
					religion_group = muslim
					religion_group = pagan_group
					religion = paulician
				}
				NOT = { religion = FROM }
			}
			NOT = { is_reformed_religion = FROM }
		}
		top_liege = {
			religion_group = ROOT
			NOT = { religion = FROM }
			OR = {
				religion = ROOT
				AND = {
					is_heretic = no
					ROOT = { is_heretic = no }
				}
			}
		}
	}
}
