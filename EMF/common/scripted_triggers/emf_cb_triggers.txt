
# In attacker scope, FROM is war defender
#
# True when defender does not pay tribute to attacker,
# when no liege of attacker either pays tribute to
# defender or defender pays tribute to them.
emf_cb_has_tribute_block = {
	or = {
		pays_tribute_to = FROM
		any_liege = {
			or = {
				pays_tribute_to = FROM
				FROM = {
					pays_tribute_to = PREV
				}
			}
		}
	}
}
