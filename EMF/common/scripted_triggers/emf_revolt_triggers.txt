# -*- ck2.scripted_triggers -*-

emf_can_province_join_religious_revolt = {
	religion = event_target:emf_revolt_origin
	NOT = { province = event_target:emf_revolt_origin }
	OR = {
		revolt_risk = 0.1
		has_province_modifier = relsub_unrest_modifier
	}
	any_province_holding = { NOT = { holding_type = nomad } }
}

emf_is_religious_revolt_related_to_relsub = {
	OR = {
		event_target:emf_revolt_prov1 = { has_province_modifier = relsub_unrest_modifier }
		event_target:emf_revolt_prov2 = { has_province_modifier = relsub_unrest_modifier }
		event_target:emf_revolt_prov3 = { has_province_modifier = relsub_unrest_modifier }
		event_target:emf_revolt_prov4 = { has_province_modifier = relsub_unrest_modifier }
		event_target:emf_revolt_prov5 = { has_province_modifier = relsub_unrest_modifier }
		event_target:emf_revolt_prov6 = { has_province_modifier = relsub_unrest_modifier }
		event_target:emf_revolt_prov7 = { has_province_modifier = relsub_unrest_modifier }
		event_target:emf_revolt_prov8 = { has_province_modifier = relsub_unrest_modifier }
		event_target:emf_revolt_prov9 = { has_province_modifier = relsub_unrest_modifier }
	}
}
