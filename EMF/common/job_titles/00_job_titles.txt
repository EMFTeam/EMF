#	attribute: This attribute is added to the ruler's for governance purposes
#	viceroy: This is the Viceroy title for de jure Kingdoms
#	allow: conditions for the character to hold the title
#	gain_effect: fires when the title is granted
#	lose_effect: fires when a title is lost if dismissal_trigger evaluates to false
#	The other fields are all character modifiers.

job_chancellor = {
	is_chancellor = yes
	attribute = diplomacy
	dignity = 0.33
	opinion_effect = 15
	
	monthly_salary = 0.1
	monthly_prestige = 0.5 # EMF
	
	allow = {
		or = {
			is_female = no
			and = {
				religion = cathar
				liege = { religion = cathar }
			}
			has_character_flag = special_chancellor
			liege = {
				primary_title = { # EMF: Absolute Cognatic opens up all council seats to women (except Lord Spiritual)
					or = {
						has_law = true_cognatic_succession
						has_law = enatic_cognatic_succession
						has_law = enatic_succession
					}
				}
			}
			and = {
				has_global_flag = gender_equality
				not = { liege = { religion_group = muslim } }
			}
		}
		or = {
			is_adult = yes
			has_character_flag = special_chancellor
		}
		prisoner = no
		not = { trait = incapable }
		not = { trait = in_hiding }
	}
	gain_effect = {
		emf_council_gain_effect = yes
	}
	lose_effect = {
		opinion = { who = FROM modifier = opinion_fired_from_council }
	}
	
	action = action_improve_relations
	action = action_fabricate_claims
	action = action_clan_sentiment # Replaces Fabricate Claims for Nomads
	action = action_sow_dissent
}

job_spiritual = {
	is_spiritual = yes
	attribute = learning
	dignity = 0.15
	opinion_effect = 15
	
	monthly_salary = 0.1
	monthly_piety = 0.1 # EMF
	monthly_prestige = 0.25 # EMF
	
	allow = {
		prisoner = no
		not = { trait = incapable }
		not = { trait = in_hiding }
		liege = {
			religion = ROOT
		}
		or = {
			and = {
				is_adult = yes
				or = {
					is_female = no
					and = {
						religion_group = pagan_group
						not = { religion = aztec }
						not = { religion = aztec_reformed }
						not = { religion = tengri_pagan }
						not = { religion = tengri_pagan_reformed }
					}
					religion = cathar
					religion = bogomilist
					and = { # EMF: Gender Equality
						has_global_flag = gender_equality
						not = { liege = { religion_group = muslim } }
					}
				}
				or = {
					not = { religion = hindu }
					trait = brahmin
				}
			}
			has_character_flag = special_spiritual
		}
		
		or = {
			is_theocracy = yes
			and = {
				liege = { is_tribal = yes }
				is_ruler = no
			}
			liege = { religion_group = muslim }
			and = {
				is_ruler = no
				has_autocephaly = no
			}
		}
		
		or = {
			not = { religion_group = muslim }
			not = { liege = { religion_group = muslim } }
			trait = detached_priest
			trait = martial_cleric
			trait = scholarly_theologian
			trait = mastermind_theologian
		}
	}
	gain_effect = {
		emf_council_gain_effect = yes
	}
	lose_effect = {
		opinion = { who = FROM modifier = opinion_fired_from_council }
	}
	dismiss_trigger = {
		FROM = {
			OR = {
				NOT = { religion_group = muslim }
				NOT = { religion = ROOT }
			}
		}
	}
	
	action = action_inquisition
	action = action_advance_cul_tech
	action = action_improve_rel_relations
	action = action_charity
	action = action_build_zeal # Tribal replacement for 'action_advance_cul_tech'
}
