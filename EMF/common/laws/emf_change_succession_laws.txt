succession_laws = {

	succ_adopt_gavelkind = {
		oldest_child_opinion = -5
		child_opinion = 15
		dynasty_opinion = 5
		vassal_opinion = 5
	
		potential = {
			has_global_flag = EMF
			NOT = { has_law = succ_gavelkind }
			emf_laws_succ_basic_potential_trigger = yes
			holder_scope = {
				OR = {
					is_tribal = no
					NOT = { religion_group = pagan_group }
				}
				OR = {
					is_feudal = yes
					is_tribal = yes
				}
				in_revolt = no
			}
			rebel = no
		}
		allow = {
			emf_laws_change_succ_allow_trigger = yes
		}
		effect = {
			emf_laws_change_succ_effect = yes
			tooltip = { succession = gavelkind }
			hidden_tooltip = {
				save_event_target_as = emf_succ_title
				
				#set opinion changes for heirs and pretenders
				current_heir = { save_event_target_as = emf_former_heir }
				any_character = {
					limit = { any_pretender_title = { title = event_target:emf_succ_title } }
					set_character_flag = emf_former_pretender
				}
				holder_scope = { character_event = { id = emf_laws.10 days = 1 } }
				
				#actually change the law
				add_law = succ_gavelkind

				holder_scope = {					
					#current heir is slightly upset (they have to share) if they were the previous heir
					event_target:emf_succ_title = {
						current_heir = {
							if = { 
								limit = { character = event_target:emf_former_heir }
								opinion = {
									who = PREVPREV
									modifier = opinion_succ_law_change
								}
							}
						}
					}
					#legitimate children of same dynasty are happy
					any_child = {
						limit = {
							dynasty = PREV
							emf_laws_any_succ_change_opinion_trigger = no
							NOR = {
								trait = bastard
								trait = monk
								trait = nun
								character = event_target:emf_former_heir
								has_character_flag = former_pretender
								any_heir_title = { title = event_target:emf_succ_title }
								any_pretender_title = { title = event_target:emf_succ_title }
							}
						}
						opinion = { who = PREV modifier = opinion_good_succ_law_change }
					}
					#vassals are slightly upset
					any_vassal = {
						limit = {
							emf_laws_any_succ_change_opinion_trigger = no
							OR = {
								primary_title = { de_jure_liege_or_above = event_target:emf_succ_title }
								event_target:emf_succ_title = { is_primary_holder_title = yes }
								event_target:emf_succ_title = { is_titular = yes }
							}
						}
						opinion = { who = PREV modifier = opinion_succ_law_change }
					}
				}
			}
		}
		revoke_allowed = {
			always = no
		}
		ai_will_revoke = {
			factor = 0
		}
		ai_will_do = {
			factor = 0
		}
	}
