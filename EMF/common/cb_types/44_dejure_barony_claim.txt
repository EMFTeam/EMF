		}
		
		any_direct_de_jure_vassal_title = {
			limit = {
				holder_scope = {
					OR = {
						character = FROM

						is_liege_or_above = event_target:target_taker

					}
				}
			}
			
			usurp_title_plus_barony_if_unlanded = { target = event_target:target_taker type = claim }
		}

		emf_cb_defeat_effect = yes
